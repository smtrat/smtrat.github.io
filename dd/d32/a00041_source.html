16 /**
17  * Base class for all backends.
18  * Offers appropriate hooks to model the whole workflow for backends where each job is executed individually.
19  * The run() method is called from ourside which first calls startTool() for every tool and then runs execute() for every pair of tool and benchmark.
20  * It also offers addResult() to store results and madeProgress() to provide a progress indication to the user.
21  * If a benchmark requires a completely different workflow, for example for a batch job, it should override the run() method.
22  */
23 class Backend {
24 private:

27 protected:

29  std::size_t mExpectedJobs;

35  Backend(): mExpectedJobs(0), mFinishedJobs(0), mLastPercent(0) {}

37  /**

50  /// Can be called to give information about the current progress, if available.
51  void madeProgress(std::size_t files = 1) {
52  mFinishedJobs += files;
53  std::size_t newPercent = mFinishedJobs * 100 / mExpectedJobs;
54  if (newPercent > mLastPercent) {
55  mLastPercent = newPercent;
56  BENCHMAX_LOG_INFO("benchmax", "Progress: " << mLastPercent << "% (" << mFinishedJobs << " / " << mExpectedJobs << ")");
57  }
58  }

60  virtual bool collect_results(const Jobs& jobs, bool check_finished) { return true; }
61  void sanitize_results(const Jobs& jobs) const {
62  for (const auto& j: jobs) {
63  auto res = mResults.get(j.first, j.second);
64  if (!res && j.first->canHandle(j.second)) {
65  BENCHMAX_LOG_WARN("benchmax", "Missing result for " << j.first->name() << " on " << j.second);
66  BENCHMAX_LOG_WARN("benchmax", "Chances are that something went wrong, please check this!");
67  }
68  }
69  }

72  XMLWriter xml(settings_benchmarks().output_file_xml);
73  mResults.store(xml, jobs);

81  if (collect_results(jobs, check_finished)) {
82  sanitize_results(jobs);
83  write_results(jobs);
84  } else {

88  /// Add a result.
89  void addResult(const Tool* tool, const fs::path& file, BenchmarkResult& result) {
90  tool->additionalResults(file, result);
91  if (result.time > settings_benchmarks().limit_time + 2*settings_benchmarks().grace_time) {
92  BENCHMAX_LOG_WARN("benchmax", "Computation took longer than it should: " << carl::settings::duration(result.time) << " > " << settings_benchmarks().limit_time << " + " << settings_benchmarks().grace_time);
93  BENCHMAX_LOG_WARN("benchmax", "Offending command: " << tool->name() << " " << file);
94  }
95  result.cleanup(settings_benchmarks().limit_time);
96  mResults.addResult(tool, file, result);
97  }
98  /// Run the list of tools against the list of benchmarks.
99  void run(const Jobs& jobs, bool wait_for_termination) {
100  mExpectedJobs = jobs.size();
101  BENCHMAX_LOG_INFO("benchmax", "Running " << mExpectedJobs << " now.");
102  if (!wait_for_termination) {
103  BENCHMAX_LOG_INFO("benchmax", "Running jobs synchronously.");
104  }
105 
106  for (const auto& tool: jobs.tools()) {
107  this->startTool(tool.get());
108  }
109  for (const auto& [tool, file]: jobs.randomized()) {
110  this->execute(tool, file);
111  }
112  this->finalize();
113  }
114 };
