62  boost::variant<VariablePos,bool> level(const FormulaT& constraint) const {
63  bool lowerLevelFound = false;
64  auto highestLevel = mVariables.first;
65  for (const auto& var : constraint.variables()) {
66  auto currentVarInOrdering = std::find(mVariables.first, mVariables.second, var);
67  if (currentVarInOrdering == mVariables.second) { // variable not found
68  if (mModel.find(var) != mModel.end()) { // level is lower than current range
69  lowerLevelFound = true;
70  } else { // level is higher
71  return true;
72  }
73  }
74  else if (highestLevel < currentVarInOrdering) {
75  highestLevel = currentVarInOrdering;
76  }
77  }
78  if (lowerLevelFound) return false;
79  return highestLevel;
80  }

91  boost::tribool addConstraint(const FormulaT& f) {
92  assert(f.getType() == carl::FormulaType::CONSTRAINT);
93 
94  FormulaT fnew(carl::model::substitute(f, mModel));
95  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Constraint " << f << " evaluated to " << fnew);
96  if (fnew.getType() == carl::FormulaType::CONSTRAINT) {
97  assert(fnew.variables().size() > 0);
98  auto lvl = level(fnew);
99  if (carl::variant_is_type<VariablePos>(lvl)) {
100  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Considering constraint " << f << " for level " << *boost::get<VariablePos>(lvl) << ".");
101  mConstraints[boost::get<VariablePos>(lvl)].push_back(fnew);
102  mEvaluatedConstraints[fnew] = f;
103  mFreeConstraintVars[boost::get<VariablePos>(lvl)].insert(fnew.variables().begin(), fnew.variables().end());
104  return true;
105  } else if (boost::get<bool>(lvl) == true) {
106  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Ignoring constraint " << f << " because it has more unassigned variables than in the current range.");
107  return true;
108  } else {
109  assert(boost::get<bool>(lvl) == false);
110  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Constraint " << f << " did not fully evaluate under the current model");
111  return boost::indeterminate;
112  }
113  } else if (fnew.isTrue()) {
114  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Ignoring " << f << " which simplified to true.");
115  return true;
116  } else {
117  assert(fnew.isFalse());
118  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Conflict: " << f << " simplified to false.");
119  return false;
120  }
121  assert(false);
122  }

124  boost::tribool addMVBound(const FormulaT& f) {
125  assert(f.getType() == carl::FormulaType::VARCOMPARE);
126 
127  // A VariableComparison is of the form y ~ RAN or y ~ MVRoot(i,p) where p is in variables x_1,...,x_n
128  // and x_1,...,x_n are of lower level according to the current variable ordering.
129  // Thus, if y is not in the variable range, then:
130  // * If y is of lower level (according to the variable ordering): Then, we can evaluate it and it
131  // evaluates either to true (and can be ignored) or false (conflict found).
132  // * If y is of higher level: Then, the bound can be ignored.
133  // If y is in the variable range, then:
134  // * If the MVRoot or RAN evaluates to a Rational, we can simplify it to a Constraint y ~ Rational,
135  // * otherwise, this method fails.
136 
137  VariablePos lvl = std::find(mVariables.first, mVariables.second, f.variableComparison().var());
138  if (lvl == mVariables.second) {
139  if (mModel.find(f.variableComparison().var()) != mModel.end()) {
140  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Evaluating " << f);
141  FormulaT fnew(carl::model::substitute(f, mModel));
142  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "-> " << fnew);
143  if (fnew.isTrue()) {
144  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Bound evaluated to true, we can ignore it.");
145  return true;
146  } else if (fnew.isFalse()) {
147  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Conflict: " << f << " simplified to false.");
148  return false;
149  } else { // not fully evaluated => not possible, as all variables are assigned
150  assert(false);
151  }
152  } else {
153  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Ignoring bound " << f << " of higher level");
154  return true;
155  }
156  } else { // the bound's level is potentially in the range to be checked
157  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Evaluating " << f);
158  FormulaT fnew(carl::model::substitute(f, mModel));
159  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "-> " << fnew);
160  if (fnew.isTrue()) {
161  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Bound evaluated to true, we can ignore it.");
162  return true;
163  } else if (fnew.isFalse()) {
164  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Conflict: " << f << " simplified to false.");
165  return false;
166  } else {
167  assert(fnew.getType() == carl::FormulaType::VARCOMPARE);
168 
169  ModelValue value = fnew.variableComparison().value();
170  if (value.isSubstitution()) {
171  auto res = value.asSubstitution()->evaluate(mModel);
172  value = res;
173  }
174  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Evaluated to " << value);
175 
176  if (value.isRational()) {
177  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Value is Rational, can convert to Constraint");
178  auto rel = fnew.variableComparison().negated() ? carl::inverse(fnew.variableComparison().relation()) : fnew.variableComparison().relation();
179  ConstraintT constr(Poly(fnew.variableComparison().var()) - value.asRational(), rel);
180  FormulaT fnewnew = FormulaT(constr);
181  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Considering constraint " << fnewnew);
182  mConstraints[lvl].push_back(fnewnew);
183  mEvaluatedConstraints[fnewnew] = f;
184  mFreeConstraintVars[lvl].insert(fnew.variableComparison().var());
185  return true;
186  } else {
187  return boost::indeterminate;
188  }
189  }
190  }
191  assert(false);
192  return boost::indeterminate;
193  }

195  boost::optional<AssignmentOrConflict> findAssignment(const VariablePos excludeVar) const {
196  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Look for assignment on level " << *(excludeVar-1));
197 
198  // set all variables to zero that do not occur in the given constraints
199  Model model;
200  bool freeVariables = false;
201  for (auto varIter = mVariables.first; varIter != excludeVar; varIter++) {
202  bool found = false;
203  for (auto iter = mVariables.first; iter != excludeVar; iter++) {
204  if (mFreeConstraintVars.at(iter).find(*varIter) != mFreeConstraintVars.at(iter).end()) {
205  found = true;
206  break;
207  }
208  }
209  if (!found) {
210  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Variable " << *varIter << " does not occur in constraint set, assigning to 0");
211  model.assign(*varIter, Rational(0));
212  } else {
213  freeVariables = true;
214  }
215  }
216 
217  if (!freeVariables) {
218  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "No free variables left, returning " << model);
219  return AssignmentOrConflict(modelToAssignment(model));
220  } else {
221  SMTModule smtModule;
222  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Calling SMT backend");
223  smtModule.push();
224  for (auto iter = mVariables.first; iter != excludeVar; iter++) {
225  for (const auto& constraint : mConstraints.at(iter)) {
226  smtModule.add(constraint);
227  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "-> Consider " << constraint);
228  }
229  }
230  Answer answer = smtModule.check();
231  if (answer == UNKNOWN || answer == ABORTED) {
232  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Backend could not solve instance");
233  return boost::none;
234  } else if (answer == SAT) {
235  assert(smtModule.model().size() > 0);
236  model.update(smtModule.model());
237  // assign all remaining vars to 0, if any
238  for (auto iter = mVariables.first; iter != excludeVar; iter++) {
239  if (model.find(*iter) == model.end()) {
240  model.assign(*iter, Rational(0));
241  }
242  }
243  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Found assignment " << model);
244  return AssignmentOrConflict(modelToAssignment(model));
245  } else if (answer == UNSAT) {
246  assert(smtModule.infeasibleSubsets().size() > 0);
247  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "No assignment found, conflicting core (after evaluation under current model) is " << smtModule.infeasibleSubsets()[0]);
248  const auto& infSubset = smtModule.infeasibleSubsets()[0];
249  FormulasT infCore;
250  for (const auto& feval : infSubset) {
251  infCore.push_back(mEvaluatedConstraints.at(feval));
252  }
253  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Conflicting core is " << infCore);
254  return AssignmentOrConflict(std::move(infCore));
255  } else {
256  assert(false);
257  }
258  }
259  assert(false);
260  return boost::none;
261  }

263  boost::optional<AssignmentOrConflict> findAssignment() const {
264  for (auto curVar = mVariables.first; curVar != mVariables.second; curVar++) {
265  auto res = findAssignment(curVar+1);
266  if (res) {
267  // if a conflict has been found OR we looked at the last variable, we return the result
268  if (carl::variant_is_type<FormulasT>(*res) || curVar == mVariables.second - 1) {
269  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Found result");
270  return res;
271  }
272  } else {
273  return boost::none;
274  }
275  }
276  assert(false);
277  return boost::none;
278  }

281 template<class Settings>
282 struct AssignmentFinder {
283  boost::optional<AssignmentOrConflict> operator()(const mcsat::Bookkeeping& data, carl::Variable var) const {
284  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Looking for an assignment for " << var << " with lookahead " << Settings::lookahead);
285 
286  static_assert(Settings::lookahead > 0);
287 
288  VariablePos varPos = std::find(data.variableOrder().begin(), data.variableOrder().end(), var);
289  VariablePos varPosEnd = varPos;
290  for (int i = 0; i < Settings::lookahead && varPosEnd != data.variableOrder().end(); i++) ++varPosEnd;
291  assert(varPos != varPosEnd);
292 
293  AssignmentFinder_detail af(std::make_pair(varPos, varPosEnd), data.model());
294 
295  for (const auto& c: data.constraints()) {
296  SMTRAT_LOG_TRACE("smtrat.mcsat.smtaf", "Adding Constraint " << c);
297  boost::tribool res = af.addConstraint(c);
298  if (indeterminate(res)) {
299  SMTRAT_LOG_TRACE("smtrat.mcsat.smtaf", "Constraint " << c << " cannot be handled!");
300  return boost::none;
301  } else if(!res){
302  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "No Assignment, built conflicting core " << c << " under model " << data.model());
303  return AssignmentOrConflict(FormulasT({c}));
304  }
305  }
306 
307  for (const auto& c: data.mvBounds()) {
308  SMTRAT_LOG_TRACE("smtrat.mcsat.smtaf", "Adding MVBound " << c);
309  boost::tribool res = af.addMVBound(c);
310  if (indeterminate(res)) {
311  SMTRAT_LOG_TRACE("smtrat.mcsat.smtaf", "MVBound " << c << " cannot be handled!");
312  return boost::none;
313  } else if(!res){
314  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "No Assignment, built conflicting core " << c << " under model " << data.model());
315  return AssignmentOrConflict(FormulasT({c}));
316  }
317  }
318 
319  SMTRAT_LOG_DEBUG("smtrat.mcsat.smtaf", "Calling AssignmentFinder...");
320  if (Settings::advance_level_by_level) {
321  return af.findAssignment();
322  } else {
323  return af.findAssignment(varPosEnd);
324  }
325  assert(false);
326  }
327 };
