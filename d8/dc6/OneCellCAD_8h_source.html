270 /**
271  * Find the index of the highest variable (wrt. the ordering
272  * in 'variableOrder') that occurs with positive degree in 'poly'.
273  * Although 'level' is a math concept that starts counting at 1
274  * we start counting at 0 and represent "no level/variable" as std::nullopt
275  * because it simplifies using the level directly as an index into arrays
276  * or vectors.
277  * Examples:
278  * - polyLevel(2) == nullopt wrt. any variable order
279  * - polyLevel(0*x+2) == nullopt wrt. any variable order
280  * - polyLevel(x+2) == 0 wrt. [x < y < z]
281  * - polyLevel(x+2) == 1 wrt. [y < x < z]
282  * - polyLevel(x+2) == 2 wrt. [y < z < x]
283  * - polyLevel(x*y+2) == 1 wrt. [x < y < z] because of y
284  * - polyLevel(x*y+2) == 1 wrt. [y < x < z] because of x
285  * - polyLevel(x*y+2) == 2 wrt. [x < z < y] because of y
286  * Preconditions:
287  * - 'poly.gatherVariables()' must be a subset of 'variableOrder'.
288  */
289 inline std::optional<std::size_t> levelOf(
290  const std::vector<carl::Variable>& variableOrder,
291  const Poly& poly) {
292  // precondition:
293  assert(isSubset(carl::variables(poly).as_vector(), variableOrder));
294 
295  // Algorithm:
296  // Iterate through each variable inside 'variableOrder' in ascending order
297  // and remove it from 'polyVariables'. The last variable in 'polyVariables' before
298  // it becomes empty is the highest appearing variable in 'poly'.
299 
300  // 'gatherVariables()' collects only variables with positive degree
301  auto polyVariables = carl::variables(poly).as_set();
302 
303  if (polyVariables.empty()) {
304  return std::nullopt; // for const-polys like '2'
305  }
306 
307  for (std::size_t level = 0; level < variableOrder.size(); ++level) {
308  polyVariables.erase(variableOrder[level]);
309  // Last variable before 'polyVariables' becomes empty is the highest variable.
310  if (polyVariables.empty())
311  return level;
312  }
313  throw("Poly contains variable not found in variableOrder");
314  return std::nullopt;
315 }

468  * Shrink the component of the 'cell' at the 'boundCandidate's level around
469  * the given point if the 'boundCandidate' is a tighter bound. Don't
470  * recursively shrink lower level components of the cell.
471  * Note that a cell's sector may collapse into a section, which is why we
472  * need to pass in a cell and not only its sector.
473  * Note that this shrink function is always "successful", even if the cell
474  * was not shrunk.
475  * @param boundCandidate Must be a non-const irreducible polynomial that does
476  * not vanish early.
477  */
478  void shrinkSingleComponent(
479  const std::size_t polyLevel,
480  const Poly& poly,
481  CADCell& cell) {
482  // This function is called "Update" in [brown15]
483  // precondition:
484  assert(isNonConstIrreducible(poly));
485  assert(!vanishesEarly(polyLevel, poly));
486  if (std::holds_alternative<Section>(cell[polyLevel]))
487  return; // canot shrink further
488 
489  Sector& sector = std::get<Sector>(cell[polyLevel]);
490  const RAN pointComp = point[polyLevel]; // called alpha_k in [brown15]
491 
492  SMTRAT_LOG_DEBUG("smtrat.cad", "Shrink cell sector at lvl " << polyLevel);
493  SMTRAT_LOG_TRACE("smtrat.cad", "Sector: " << sector);
494  SMTRAT_LOG_TRACE("smtrat.cad", "Poly: " << poly);
495  SMTRAT_LOG_TRACE("smtrat.cad", "Lvl-Var: " << variableOrder[polyLevel]);
496  SMTRAT_LOG_DEBUG("smtrat.cad", "PointComp: " << pointComp);
497  //SMTRAT_LOG_TRACE("smtrat.cad", "Point: " << point.prefixPoint(polyLevel + 1));
498  // Isolate real isolatedRoots of level-k 'poly' after plugin in a level-(k-1) point.
499  // Poly must not vanish under this prefixPoint!
500  auto isolatedRoots =
501  isolateLastVariableRoots(polyLevel, poly);
502  if (isolatedRoots.empty()) {
503  SMTRAT_LOG_TRACE("smtrat.cad", "No isolatable isolatedRoots");
504  return;
505  }
506  SMTRAT_LOG_TRACE("smtrat.cad", "Isolated roots: " << isolatedRoots);
507 
508  if (sector.lowBound) {
509  SMTRAT_LOG_TRACE("smtrat.cad", "Existing low bound: " << (*(sector.lowBound)).isolatedRoot);
510  assert((*(sector.lowBound)).isolatedRoot < pointComp);
511  }
512  if (sector.highBound) {
513  SMTRAT_LOG_TRACE("smtrat.cad", "Existing high bound: " << (*(sector.highBound)).isolatedRoot);
514  assert((*(sector.highBound)).isolatedRoot > pointComp);
515  }
516 
517  // Search for closest isolatedRoots/boundPoints to pointComp, i.e.
518  // someRoot ... < closestLower <= pointComp <= closestUpper < ... someOtherRoot
519  std::optional<RAN> closestLower;
520  std::optional<RAN> closestUpper;
521 
522  std::size_t rootIdx = 0, lowerRootIdx, upperRootIdx;
523  carl::Variable rootVariable = variableOrder[polyLevel];
524  for (const auto& root : isolatedRoots) {
525  rootIdx++;
526  if (root < pointComp) {
527  SMTRAT_LOG_TRACE("smtrat.cad", "Smaller: " << root);
528  if (!closestLower || *closestLower < root) {
529  closestLower = root;
530  lowerRootIdx = rootIdx;
531  }
532  } else if (root == pointComp) {
533  SMTRAT_LOG_TRACE("smtrat.cad", "Equal: " << root);
534  // Sector collapses into a section
535  cell[polyLevel] = Section{asRootExpr(rootVariable, poly, rootIdx), root};
536  SMTRAT_LOG_TRACE("smtrat.cad", "Sector collapses: " << (Section{asRootExpr(rootVariable, poly, rootIdx), root}));
537  return;
538  } else { // pointComp < root
539  SMTRAT_LOG_TRACE("smtrat.cad", "Bigger: " << root);
540  if (!closestUpper || root < *closestUpper) {
541  closestUpper = root;
542  upperRootIdx = rootIdx;
543  }
544  }
545  }
546 
547  if (closestLower) {
548  if (!sector.lowBound || (*(sector.lowBound)).isolatedRoot < closestLower) {
549  sector.lowBound = Section{asRootExpr(rootVariable, poly, lowerRootIdx), *closestLower};
550  SMTRAT_LOG_TRACE("smtrat.cad", "New section: "
551  << " " << sector);
552  }
553  }
554 
555  if (closestUpper) {
556  if (!sector.highBound || closestUpper < (*(sector.highBound)).isolatedRoot) {
557  sector.highBound = Section{asRootExpr(rootVariable, poly, upperRootIdx), *closestUpper};
558  SMTRAT_LOG_TRACE("smtrat.cad", "New section: "
559  << " " << sector);
560  }
561  }

564  /**
565  * Find the smallest index m such that in the set S of all m-th partial
566  * derivatives of the given poly, such that there is a derivative that does
567  * not vanish early [brown15].
568  * Note that m > 0 i.e, this function never just returns the given poly,
569  * which is the 0th partial derivative of itself.
570  * @param poly must not be a const-poly like 2, otherwise this function
571  * does not terminate.
572  * @return Actually only returns the set S
573  */
574  std::vector<Poly> partialDerivativesLayerWithSomeNonEarlyVanishingPoly(
575  const std::vector<carl::Variable>& variableOrder,
576  const carl::RealAlgebraicPoint<Rational>& point,
577  const TagPoly2& mainPoly) {
578  assert(!mainPoly.poly.isConstant());
579  // We search for this set of partial derivatives "layer by layer".
580  // The layer of 0-th partial derivatives is the mainPoly itself.
581  std::vector<Poly> layerOfDerivatives;
582  layerOfDerivatives.emplace_back(mainPoly.poly);
583  bool foundSomeNonEarlyVanishingDerivative = false;
584 
585  do {
586  std::vector<Poly> nextLayer;
587  for (const auto& poly : layerOfDerivatives) {
588  // Derive poly wrt to each variable (variables with idx 0 to 'mainPoly.level')
589  for (std::size_t varIdx = 0; varIdx <= mainPoly.level; varIdx++) {
590  const auto derivative = carl::derivative(poly, variableOrder[varIdx]);
591  if (carl::isZero(derivative))
592  continue;
593  nextLayer.emplace_back(derivative);
594  if (foundSomeNonEarlyVanishingDerivative)
595  continue; // avoid expensive vanishing check
596 
597  if (derivative.isConstant() ||
598  !vanishesEarly(mainPoly.level, mainPoly.poly))
599  foundSomeNonEarlyVanishingDerivative = true;
600  // still need to compute all remaining nextLayer-polys
601  }
602  }
603  layerOfDerivatives = std::move(nextLayer);
604  } while (!foundSomeNonEarlyVanishingDerivative);
605 
606  return layerOfDerivatives;
607  }
