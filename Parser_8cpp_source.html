103 void SMTLIBParser::declareConst(const std::string& name, const carl::Sort& sort) {
104  if (state->handler->printInstruction()) handler->regular() << "(declare-const " << name << " " << sort << ")" << std::endl;
105  assert(state->isSymbolFree(name));
106  switch (TypeOfTerm::get(sort)) {
107  case ExpressionType::BOOLEAN: {
108  if (state->var_bool.sym.find(name) != nullptr) {
109  SMTRAT_LOG_WARN("smtrat.parser", "A boolean variable with name '" << name << "' has already been defined.");
110  }
111  carl::Variable var = carl::freshBooleanVariable(name);
112  state->var_bool.sym.add(name, var);
113  break;
114  }
115  case ExpressionType::THEORY: {
116  if (state->var_theory.sym.find(name) != nullptr) {
117  SMTRAT_LOG_WARN("smtrat.parser", "A theory variable with name '" << name << "' has already been defined.");
118  }
119  carl::Variable var = carl::freshVariable(name, carl::SortManager::getInstance().getType(sort));
120  state->var_theory.sym.add(name, var);
121  break;
122  }
123  case ExpressionType::UNINTERPRETED: {
124  SMTRAT_LOG_ERROR("smtrat.parser", "Only variables of type \"Bool\", \"Int\" or \"Real\" are allowed!");
125  break;
126  }
127  default:
128  SMTRAT_LOG_ERROR("smtrat.parser", "Only variables of type \"Bool\", \"Int\" or \"Real\" are allowed!");
129  }

132 void SMTLIBParser::declareFun(const std::string& name, const std::vector<carl::Sort>& args, const carl::Sort& sort) {
133  if (state->handler->printInstruction()) handler->regular() << "(declare-fun " << name << " () " << sort << ")" << std::endl;
134  assert(state->isSymbolFree(name));
135  switch (TypeOfTerm::get(sort)) {
136  case ExpressionType::BOOLEAN: {
137  if (args.size() == 0) {
138  carl::Variable var = carl::freshBooleanVariable(name);
139  state->var_bool.sym.add(name, var);
140  callHandler(&InstructionHandler::declareFun, var);
141  } else {
142  state->funmap_ufbool.add(name, carl::newUninterpretedFunction(name, args, sort));
143  }
144  break;
145  }
146  case ExpressionType::THEORY: {
147  if (args.size() == 0) {
148  carl::Variable var = carl::freshVariable(name, carl::SortManager::getInstance().getType(sort));
149  state->var_theory.sym.add(name, var);
150  callHandler(&InstructionHandler::declareFun, var);
151  } else {
152  state->funmap_uftheory.add(name, carl::newUninterpretedFunction(name, args, sort));
153  }
154  break;
155  }
156  case ExpressionType::UNINTERPRETED: {
157  if (args.size() == 0) {
158  carl::Variable var = carl::freshVariable(name, carl::VariableType::VT_UNINTERPRETED);
159  auto v = carl::UVariable(var, sort);
160  state->var_uninterpreted.sym.add(name, v);
161  callHandler(&InstructionHandler::declareFun, var);
162  } else {
163  auto uf = carl::newUninterpretedFunction(name, args, sort);
164  state->funmap_uf.add(name, uf);
165  }
166  break;
167  }
168  case ExpressionType::BITVECTOR: {
169  if (args.size() == 0) {
170  carl::Variable var = carl::freshVariable(name, carl::VariableType::VT_BITVECTOR);
171  auto indices = carl::SortManager::getInstance().getIndices(sort);
172  if (indices == nullptr || indices->size() != 1) {
173  SMTRAT_LOG_ERROR("smtrat.parser", "The bitvector sort \"BitVec\" must always be used with exactly one index, for example \"(_ BitVec 32)\".");
174  } else {
175  auto v = carl::BVVariable(var, (*indices)[0]);
176  state->var_bitvector.sym.add(name, v);
177  callHandler(&InstructionHandler::declareFun, var);
178  }
179  } else {
180  SMTRAT_LOG_ERROR("smtrat.parser", "Uninterpreted functions over bitvectors are not supported yet.");
181  }
182  }
183  default:
184  SMTRAT_LOG_ERROR("smtrat.parser", "Only functions of with a defined return type are allowed!");
185  }

194 void SMTLIBParser::defineFun(const std::string& name, const std::vector<carl::Variable>& args, const carl::Sort& sort, const Argument& term) {
195  if (state->handler->printInstruction()) handler->regular() << "(define-fun " << name << " () " << term << ")" << std::endl;
196  state->popScope();
197  switch (TypeOfTerm::get(sort)) {
198  case ExpressionType::BOOLEAN:
199  if (TypeOfTerm::get(term) != ExpressionType::BOOLEAN) {
200  SMTRAT_LOG_ERROR("smtrat.parser", "The return type of \"" << name << "\" was given as Bool, but the parsed expression is a polynomial.");
201  return;
202  }
203  if (args.size() == 0) {
204  state->bind_bool.sym.add(name, boost::get<FormulaT>(term));
205  } else {
206  state->funmap_bool.add(name, std::make_tuple(name, args, boost::get<FormulaT>(term)));
207  }
208  break;
209  case ExpressionType::THEORY:
210  if (TypeOfTerm::get(term) != ExpressionType::THEORY) {
211  SMTRAT_LOG_ERROR("smtrat.parser", "The return type of \"" << name << "\" was given as a theory type, but the parsed expression is a formula.");
212  return;
213  }
214  for (const carl::Variable& v: args) {
215  if (TypeOfTerm::get(v) != ExpressionType::THEORY) {
216  SMTRAT_LOG_ERROR("smtrat.parser", "The argument " << carl::VariablePool::getInstance().getName(v) << " of " << name << " is Bool. For theory functions, only theory arguments are supported.");
217  return;
218  }
219  }
220  if (args.size() == 0) {
221  state->bind_theory.sym.add(name, boost::get<Poly>(term));
222  } else {
223  state->funmap_theory.add(name, std::make_tuple(name, args, boost::get<Poly>(term)));
224  }
225  break;
226  case ExpressionType::UNINTERPRETED:
227  SMTRAT_LOG_ERROR("smtrat.parser", "Functions of uninterpreted type are not allowed!");
228  break;
229  case ExpressionType::BITVECTOR:
230  SMTRAT_LOG_ERROR("smtrat.parser", "Functions of bitvector type are not allowed!");
231  break;
232  default:
233  SMTRAT_LOG_ERROR("smtrat.parser", "Unsupported function return type.");
234  }
