60  switch (rhs) {
61  case carl::Relation::GREATER:
62  case carl::Relation::GEQ: return carl::Relation::GEQ;
63  case carl::Relation::LEQ:
64  case carl::Relation::LESS: return carl::Relation::LEQ;
65  default: return carl::Relation::NEQ;
