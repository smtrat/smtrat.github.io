247  // Less than for literals in a lemma
248  struct lemma_lt
249  {
250  SATModule& solver;
251  lemma_lt(SATModule& solver) : solver(solver) {}
252  int levelOf(Minisat::Var v) {
253  if (solver.bool_value(v) != l_Undef) {
254  SMTRAT_LOG_TRACE("smtrat.sat.lemma_lt", "Level of " << v << ": " << solver.trailIndex(v) << " (trail index from boolean assignment)");
255  return solver.trailIndex(v);
256  } else {
257  assert(Settings::mc_sat);
258  auto res = solver.mMCSAT.assignedAtTrailIndex(v);
259  SMTRAT_LOG_TRACE("smtrat.sat.lemma_lt", "Index of " << v << ": " << res);
260  return res;
261  }
262  }
263  bool operator () (Minisat::Lit x, Minisat::Lit y) {
264  SMTRAT_LOG_TRACE("smtrat.sat.lemma_lt", "Doing comparison " << x << " < " << y << "?");
265  if (x == y) return false;
266 
267  /* We want the following order:
268  * First: Undef < True < False
269  * Second: Larger level < Smaller level
270  */
271  Minisat::lbool x_value = solver.value(x);
272  Minisat::lbool y_value = solver.value(y);
273  if (x_value == l_Undef) {
274  if (y_value == l_Undef) {
275  // arbitrary
276  SMTRAT_LOG_TRACE("smtrat.sat.lemma_lt", "Both unassigned, using arbitrary order: " << (x < y));
277  return x < y;
278  } else {
279  // x < y
280  SMTRAT_LOG_TRACE("smtrat.sat.lemma_lt", x << " unassigned but " << y << " assigned, hence true");
281  return true;
282  }
283  }
284  if (y_value == l_Undef) {
285  // y < x
286  SMTRAT_LOG_TRACE("smtrat.sat.lemma_lt", x << " assigned but " << y << " unassigned, hence false");
287  return false;
288  }
289  assert(x_value != l_Undef && y_value != l_Undef);
290  if (x_value != y_value) {
291  return x_value == l_True;
292  SMTRAT_LOG_TRACE("smtrat.sat.lemma_lt", "Both assigned but differently: " << (x_value == l_True));
293  }
294  assert(x_value == y_value);
295  int x_level = levelOf(var(x));
296  SMTRAT_LOG_TRACE("smtrat.sat.lemma_lt", "Level of " << x << ": " << x_level);
297  int y_level = levelOf(var(y));
298  SMTRAT_LOG_TRACE("smtrat.sat.lemma_lt", "Level of " << y << ": " << y_level);
299  SMTRAT_LOG_TRACE("smtrat.sat.lemma_lt", "Comparing levels: " << (x_level > y_level));
300  return x_level > y_level;
301  }

938  inline Minisat::lbool theoryValue( Minisat::Var x ) const {
939  // MCSAT-specific:
940  // There are currently two ways of propagating literals evaluating in the theory
941  // to the Boolean level: Semantic propagations inserted as decisions explicitly
942  // in the search loop and this value function. This value function should have no
943  // effect most of the times as semantic propagations handle most cases. The only case
944  // where theoryValue is neccessary is during backtracking, when semantic propagations
945  // which are not inserted at the earliest possible point are already backtracked but
946  // the corerspondiong literals still evaluate to a value.
947  // Since the trail is left inconsistent after an theory decision, it is important for
948  // the correctness of the procedure that the Boolean level is considered first here,
949  // as Boolean conflict resolution (which will fix those inconsistencies) relies on that.
950  // Note that the inconsistency mentioned here is not between semantic propagations and
951  // Boolean decisions/propagations but between theory values and Boolean decisions/propagations.
952  // In this sense, the trail is kept consistent throughout the procedure.
953  Minisat::lbool res = assigns[x];
954  if (res == l_Undef) {
955  if (mBooleanConstraintMap.size() <= x) return l_Undef;
956  if (mBooleanConstraintMap[x].first == nullptr) return l_Undef;
957  res = mMCSAT.evaluateLiteral(Minisat::mkLit(x, false));
958  }
959  //SMTRAT_LOG_DEBUG("smtrat.sat", x << " -> " << res);
960  return res;
961  }
