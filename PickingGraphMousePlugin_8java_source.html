58  {
59  if( pickedVertex!=null )
60  {
61  Point p = me.getPoint();
62  Point2D graphPoint = visualization.getRenderContext().getMultiLayerTransformer().inverseTransform( p );
63  Point2D graphDown = visualization.getRenderContext().getMultiLayerTransformer().inverseTransform( mouseDown );
64  Layout<Vertex,Edge> layout = visualization.getGraphLayout();
65  double dx = graphPoint.getX()-graphDown.getX();
66  double dy = graphPoint.getY()-graphDown.getY();
67  PickedState<Vertex> pickedVertexState = visualization.getPickedVertexState();
68  for( Vertex v : pickedVertexState.getPicked() )
69  {
70  if( v!=null )
71  {
72  Point2D vp = layout.transform( v );
73  vp.setLocation( vp.getX()+dx, vp.getY()+dy );
74  layout.setLocation( v, vp );
75  }
76  }
77  mouseDown = p;
78  }
79  else
80  {
81  Point mouseOut = me.getPoint();

176  private boolean notTooClose( Point2D p, Point2D q, double min )
177  {
178  return !( Math.abs( p.getX()-q.getX() )<min && Math.abs( p.getY()-q.getY() )<min );

186  private void pickContainedVertices( VisualizationViewer<Vertex,Edge> visualization, Point2D mouseDown, Point2D mouseOut, boolean clear )
187  {
188  Layout<Vertex,Edge> layout = visualization.getGraphLayout();
189  PickedState<Vertex> pickedVertexState = visualization.getPickedVertexState();
190  Rectangle2D pickRectangle = new Rectangle2D.Double();
191  pickRectangle.setFrameFromDiagonal( mouseDown, mouseOut );
192  if( pickedVertexState!=null )
193  {
194  if( clear )
195  {
196  pickedVertexState.clear();
197  }
198  GraphElementAccessor<Vertex,Edge> pickSupport = visualization.getPickSupport();
199  Collection<Vertex> picked = pickSupport.getVertices( layout, pickRectangle );
200  for( Vertex v : picked )
201  {
202  pickedVertexState.pick( v, true );
203  }
204  }
