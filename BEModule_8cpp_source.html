53  FormulaT BEModule<Settings>::extractBounds(const FormulaT& formula)
54  {
55  if (formula.getType() != carl::FormulaType::OR && formula.getType() != carl::FormulaType::AND) return formula;
56  bool conjunction = formula.getType() == carl::FormulaType::AND;
57 
58  FormulaT::ConstraintBounds cb;
59  collectBounds(cb, formula, conjunction);
60  if (cb.empty()) return formula;
61 
62  SMTRAT_LOG_DEBUG("smtrat.be", "Extracted bounds " << cb);
63  if (conjunction) {
64  FormulasT f;
65  FormulaT::swapConstraintBounds(cb, f, conjunction);
66  f.push_back(formula);
67  return FormulaT(carl::FormulaType::AND, std::move(f));
68  } else {
69  for (const auto& poly: cb) {
70  if (!poly.first.isVariable()) continue;
71  std::vector<Choice> choices;
72  for (const auto& entry: poly.second) {
73  if (entry.second.first != carl::Relation::EQ) break;
74  choices.emplace_back(poly.first.getSingleVariable(), entry.second.second);
75  }
76  if (choices.size() != poly.second.size()) continue;
77  for (const auto& c: choices) {
78  auto it = mReplacements.find(c);
79  if (it == mReplacements.end()) {
80  mReplacements.emplace(c, carl::freshBooleanVariable());
81  }
82  }
83  }
84  }
85  return formula;

98  FormulaT BEModule<Settings>::applyReplacements(const FormulaT& f) const {
99  std::vector<carl::Variable> variables;
100  std::map<FormulaT, FormulaT> repl;
101  for (const auto& r: mReplacements) {
102  carl::Variable v = std::get<0>(r.first);
103  FormulaT form = std::get<1>(r.first);
104 
105  variables.push_back(v);
106  repl.emplace(form, FormulaT(r.second));
107  }
108  carl::FormulaSubstitutor<FormulaT> subs;
109  FormulaT res = subs.substitute(f, repl);
110  carl::Variables remainingVars;
111  res.allVars(remainingVars);
112  FormulasT impl;
113  for (const auto& v: variables) {
114  if (remainingVars.count(v) > 0) {
115  for (const auto& r: mReplacements) {
116  if (v != std::get<0>(r.first)) continue;
117  FormulaT form = std::get<1>(r.first);
118  impl.push_back(FormulaT(carl::FormulaType::IMPLIES, {FormulaT(r.second), form}));
119  }
120  }
121  }
122  if (impl.empty()) return res;
123  impl.push_back(res);
124  return FormulaT(carl::FormulaType::AND, std::move(impl));
125  }
