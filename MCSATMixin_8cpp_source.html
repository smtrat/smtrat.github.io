12 bool MCSATMixin::backtrackTo(Minisat::Lit literal) {
13  std::size_t lvl = level();
14  while (lvl > 0) {
15  if (get(lvl).decisionLiteral == literal) break;
16  lvl--;
17  }
18  SMTRAT_LOG_TRACE("smtrat.sat.mcsat", "Backtracking until " << literal << " on level " << lvl);
19  if (lvl == 0 || lvl == level()) {
20  SMTRAT_LOG_TRACE("smtrat.sat.mcsat", "Nothing to backtrack for " << literal);
21  return false;
22  }
23 
24  while (level() > lvl) {
25  popLevel();
26  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Backtracking theory assignment for " << currentVariable());
27 
28  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Model " << model());
29 
30  if (current().decisionLiteral != Minisat::lit_Undef) {
31  mBackend.popAssignment(currentVariable());
32  }
33  current().decisionLiteral = Minisat::lit_Undef;
34  }
35  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Next theory variable is " << currentVariable());
36  return true;
37 }

50 boost::optional<FormulaT> MCSATMixin::isDecisionPossible(Minisat::Lit lit) {
51  auto var = Minisat::var(lit);
52  if (!mGetter.isTheoryAbstraction(var)) return boost::none;
53  const auto& f = mGetter.reabstractLiteral(lit);
54  auto res = mBackend.isInfeasible(currentVariable(), f);
55  if (res == boost::none) {
56  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Decision " << lit << " is possible");
57  } else {
58  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Decision " << lit << " is impossible due to " << *res);
59  return mBackend.explain(currentVariable(), *res, FormulaT(carl::FormulaType::FALSE));
60  }
61  return boost::none;
62 }

64 void MCSATMixin::updateCurrentLevel(carl::Variable var) {
65  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Updating current level for " << var);
66  assert(level() <= mTheoryStack.size());
67  if (level() == mTheoryStack.size()) {
68  mTheoryStack.emplace_back();
69  current().variable = var;
70  } else {
71  assert(current().variable == var);
72  }
73 
74  // Check undecided variables whether they became univariate
75  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Undecided Variables: " << mUndecidedVariables);
76  for (auto vit = mUndecidedVariables.begin(); vit != mUndecidedVariables.end();) {
77  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Looking at " << *vit);
78  if (theoryLevel(*vit) != level()) {
79  ++vit;
80  continue;
81  }
82  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Associating " << *vit << " with " << var << " at " << level());
83  current().univariateVariables.push_back(*vit);
84  vit = mUndecidedVariables.erase(vit);
85  }
86  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "-> " << mUndecidedVariables);
87 }

101 void MCSATMixin::pushLevel(carl::Variable var) {
102  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Pushing new level with " << var);
103  // Future levels are cached and maybe should be discarded
104  if (level() != mTheoryStack.size() - 1) {
105  // Next level already has the right variable
106  if (current().variable == var) return;
107  // Discard levels until the current one
108  while (level() != mTheoryStack.size() - 1) {
109  removeLastLevel();
110  }
111  }
112  mCurrentLevel++;
113  // Push new level
114  updateCurrentLevel(var);
115 }
