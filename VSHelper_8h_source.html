65  inline bool generateZeros(const ConstraintT& constraint, const carl::Variable& eliminationVar, std::function<void(SqrtEx&& sqrtExpression, ConstraintsT&& sideConditions)> yield_result) {
66  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Generating zeros of constraint " << constraint);
67 
68  // TODO test with factorization
69 
70  if (!constraint.hasVariable(eliminationVar)) {
71  return true;
72  }
73 
74  std::vector<Poly> factors;
75  ConstraintsT sideConditions;
76 
77  if(constraint.hasFactorization()) {
78  for(const auto& iter : constraint.factorization()) {
79  carl::Variables factorVars;
80  iter.first.gatherVariables( factorVars );
81  if(factorVars.find( eliminationVar ) != factorVars.end()) {
82  factors.push_back( iter.first );
83  }
84  else {
85  ConstraintT cons = ConstraintT( iter.first, carl::Relation::NEQ );
86  if( cons != ConstraintT( true ) ) {
87  assert( cons != ConstraintT( false ) );
88  sideConditions.insert( cons );
89  }
90  }
91  }
92  }
93  else {
94  factors.push_back( constraint.lhs() );
95  }
96 
97  for(const auto& factor : factors) {
98  VarPolyInfo varInfo = factor.getVarInfo<true>( eliminationVar );
99  const auto& coeffs = varInfo.coeffs();
100  assert( !coeffs.empty() );
101 
102  switch( coeffs.rbegin()->first ) {
103  case 0:
104  {
105  assert(false);
106  break;
107  }
108  case 1: // degree = 1
109  {
110  Poly constantCoeff;
111  auto iter = coeffs.find( 0 );
112  if( iter != coeffs.end() ) constantCoeff = iter->second;
113  // Create state ({b!=0} + oldConditions, [x -> -c/b]):
114  ConstraintT cons = ConstraintT( coeffs.rbegin()->second, carl::Relation::NEQ );
115  if( cons != ConstraintT( false ) ) {
116  ConstraintsT sideCond = sideConditions;
117  if( cons != ConstraintT( true ) ) {
118  sideCond.insert( cons );
119  }
120  SqrtEx sqEx = SqrtEx( -constantCoeff, Poly(), coeffs.rbegin()->second, Poly() );
121  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Generated zero " << sqEx << " with side conditions " << sideCond);
122  yield_result(std::move(sqEx), std::move(sideCond));
123  }
124  break;
125  }
126  case 2: // degree = 2
127  {
128  Poly constantCoeff;
129  auto iter = coeffs.find( 0 );
130  if( iter != coeffs.end() ) constantCoeff = iter->second;
131  Poly linearCoeff;
132  iter = coeffs.find( 1 );
133  if( iter != coeffs.end() ) linearCoeff = iter->second;
134  Poly radicand = carl::pow(linearCoeff, 2) - Rational( 4 ) * coeffs.rbegin()->second * constantCoeff;
135  ConstraintT cons11 = ConstraintT( coeffs.rbegin()->second, carl::Relation::EQ );
136  if( cons11 != ConstraintT( false ) ) {
137  // Create state ({a==0, b!=0} + oldConditions, [x -> -c/b]):
138  ConstraintT cons12 = ConstraintT( linearCoeff, carl::Relation::NEQ );
139  if( cons12 != ConstraintT( false ) ) {
140  ConstraintsT sideCond = sideConditions;
141  if( cons11 != ConstraintT( true ) )
142  sideCond.insert( cons11 );
143  if( cons12 != ConstraintT( true ) )
144  sideCond.insert( cons12 );
145  SqrtEx sqEx = SqrtEx( -constantCoeff, Poly(), linearCoeff, Poly() );
146  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Generated zero " << sqEx << " with side conditions " << sideCond);
147  yield_result(std::move(sqEx), std::move(sideCond));
148  }
149  }
150  ConstraintT cons21 = ConstraintT( radicand, carl::Relation::GEQ );
151  if( cons21 != ConstraintT( false ) ) {
152  ConstraintT cons22 = ConstraintT( coeffs.rbegin()->second, carl::Relation::NEQ );
153  if( cons22 != ConstraintT( false ) ) {
154  ConstraintsT sideCond = sideConditions;
155  if( cons21 != ConstraintT( true ) ) {
156  sideCond.insert( cons21 );
157  }
158  if( cons22 != ConstraintT( true ) ) {
159  sideCond.insert( cons22 );
160  }
161 
162  // Create ({a!=0, b^2-4ac>=0} + oldConditions, [x -> (-b-sqrt(b^2-4ac))/2a]):
163  SqrtEx sqExA = SqrtEx( -linearCoeff, Poly(-1), Rational( 2 ) * coeffs.rbegin()->second, radicand );
164  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Generated zero " << sqExA << " with side conditions " << sideCond);
165  yield_result(std::move(sqExA), ConstraintsT(sideCond));
166 
167  // Create ({a!=0, b^2-4ac>=0} + oldConditions, [x -> (-b+sqrt(b^2-4ac))/2a]):
168  SqrtEx sqExB = SqrtEx( -linearCoeff, Poly(1), Rational( 2 ) * coeffs.rbegin()->second, radicand );
169  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Generated zero " << sqExB << " with side conditions " << sideCond);
170  yield_result(std::move(sqExB), std::move(sideCond));
171  }
172  }
173  break;
174  }
175  //degree > 2 (> 3)
176  default:
177  {
178  // degree to high
179  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Degree of factor " << factor << " too high");
180  return false;
181  break;
182  }
183  }
184  }
185  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Generated zeros for " << constraint << " finished");
186  return true;
187  }

192  inline boost::optional<bool> compareSqrtEx(const SqrtEx& exprA, const SqrtEx& exprB, const Model& model) {
193  static carl::Variable subVar1 = carl::freshRealVariable("__cmpVar1");
194  static carl::Variable subVar2 = carl::freshRealVariable("__cmpVar2");
195  static ConstraintT subConstraint(Poly(subVar1) - subVar2, carl::Relation::LESS);
196 
197  // calculate subRes := (x<0)[(exprA-exprB)//x]
198  smtrat::vs::Substitution subSub1(subVar1, exprA, smtrat::vs::Substitution::Type::NORMAL, carl::PointerSet<smtrat::vs::Condition>());
199  smtrat::vs::Substitution subSub2(subVar2, exprB, smtrat::vs::Substitution::Type::NORMAL, carl::PointerSet<smtrat::vs::Condition>());
200  smtrat::vs::DisjunctionOfConstraintConjunctions subRes1;
201  if (!substituteHelper(subConstraint, subSub1, subRes1)) {
202  return false;
203  }
204  std::vector<FormulaT> disjF;
205  for (const auto& conj : subRes1) {
206  std::vector<FormulaT> conjF;
207  for (const auto& constr : conj) {
208  smtrat::vs::DisjunctionOfConstraintConjunctions subRes2;
209  if (!substituteHelper(constr, subSub2, subRes2)) {
210  return false;
211  }
212  conjF.push_back(doccToFormula(subRes2));
213  }
214  disjF.emplace_back(carl::FormulaType::AND, std::move(conjF));
215  }
216  FormulaT subRes(carl::FormulaType::OR, std::move(disjF));
217 
218  // evaluate subRes
219  carl::ModelValue<Rational, Poly> evalRes = carl::model::evaluate(subRes, model);
220  assert(evalRes.isBool());
221 
222  return evalRes.asBool();
223  }

225  inline bool rationalLessThanSqrtEx(const Rational& rat, const SqrtEx& expr, const Model& model) {
226  if (expr.factor() == Poly(Rational(0))) {
227  Poly lhs = expr.constantPart() / expr.denominator();
228  carl::ModelValue<Rational, Poly> eval = carl::model::evaluate(lhs, model);
229  return rat < eval.asRational();
230  }
231  else {
232  Poly rs = expr.denominator() * expr.factor();
233  carl::ModelValue<Rational, Poly> evalRs = carl::model::evaluate(rs, model);
234  Poly lhs = ( rat * expr.denominator() - expr.constantPart() ) / expr.factor();
235  carl::ModelValue<Rational, Poly> eval = carl::model::evaluate(lhs, model);
236  carl::ModelValue<Rational, Poly> evalRad = carl::model::evaluate(expr.radicand(), model);
237 
238  if (evalRs.asRational() > Rational(0)) {
239  return eval.asRational() < Rational(0) || eval.asRational()*eval.asRational() < evalRad.asRational();
240  } else if (evalRs.asRational() < Rational(0)) {
241  return eval.asRational() > Rational(0) && eval.asRational()*eval.asRational() > evalRad.asRational();
242  } else {
243  return false;
244  }
245  }
246  }

248  inline bool sqrtExLessThanRational(const SqrtEx& expr, const Rational& rat, const Model& model) {
249  if (expr.factor() == Poly(Rational(0))) {
250  Poly lhs = expr.constantPart() / expr.denominator();
251  carl::ModelValue<Rational, Poly> eval = carl::model::evaluate(lhs, model);
252  return rat > eval.asRational();
253  }
254  else {
255  Poly rs = expr.denominator() * expr.factor();
256  carl::ModelValue<Rational, Poly> evalRs = carl::model::evaluate(rs, model);
257  Poly lhs = ( rat * expr.denominator() - expr.constantPart() ) / expr.factor();
258  carl::ModelValue<Rational, Poly> eval = carl::model::evaluate(lhs, model);
259  carl::ModelValue<Rational, Poly> evalRad = carl::model::evaluate(expr.radicand(), model);
260 
261  if (evalRs.asRational() > Rational(0)) {
262  return eval.asRational() > Rational(0) && eval.asRational()*eval.asRational() > evalRad.asRational();
263  } else if (evalRs.asRational() < Rational(0)) {
264  return eval.asRational() < Rational(0) || eval.asRational()*eval.asRational() < evalRad.asRational();
265  } else {
266  return false;
267  }

271  inline bool generateZeros(const MultivariateRootT& mvroot, const Model& model, std::function<void(SqrtEx&& sqrtExpression, ConstraintsT&& sideConditions)> yield_result) {
272  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Generating zeros of VariableComparison with MultivariateRoot " << mvroot);
273 
274  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Disabled!");
275  return false;
276 
277  // get zeros of mvroot.var() in mvroot.poly()
278  ConstraintT constraint(mvroot.poly(), carl::Relation::EQ);
279  std::vector<std::pair<SqrtEx, ConstraintsT>> zeros;
280  bool result = generateZeros(constraint, mvroot.var(), [&](SqrtEx&& sqrtExpression, ConstraintsT&& sideConditions) {
281  zeros.push_back(std::make_pair(std::move(sqrtExpression), std::move(sideConditions)));
282  });
283  if (!result) {
284  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Could not generate zeros of polynomial of MultivariateRoot");
285  return false;
286  }
287 
288  // remove non-existing zeros from vector
289  zeros.erase( std::remove_if(zeros.begin(), zeros.end(), [&](const auto& zero) {
290  for (const auto& cond : zero.second) {
291  carl::ModelValue<Rational, Poly> evalRes = carl::model::evaluate(FormulaT(cond), model);
292  assert(evalRes.isBool());
293  if (!evalRes.asBool()) {
294  return true;
295  }
296  }
297  return false;
298  }), zeros.end() );
299 
300  // make sure that zero exists
301  assert(zeros.size() > 0 && mvroot.k() <= zeros.size());
302 
303  // sort zeros according to current model
304  bool failed = false;
305  std::sort (zeros.begin(), zeros.end(), [&](const auto& zero1, const auto& zero2) {
306  auto res = compareSqrtEx(zero1.first, zero2.first, model);
307  if (res == boost::none) {
308  failed = true;
309  return false;
310  }
311  else {
312  return res.value();
313  }
314  });
315  if (failed) {
316  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Could not sort zeros of polynomial of MultivariateRoot");
317  return false;
318  }
319 
320  // choose ith zero
321  auto res = zeros[mvroot.k()-1];
322  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Generated zero " << res.first << " with side conditions " << res.second);
323  yield_result(std::move(res.first), std::move(res.second));
324  return true;
325  }

329  if (ran.isNumeric()) {
330  // should have been simplified to a ConstraintT before...
331  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Not implemented for numeric content");
332  assert(false);
333  return false;
334  }
335  else if (ran.isInterval()) {
336  // get zeros of ran.getIRPolynomial().mainVar() in ran.getIRPolynomial()

362  return false;
363  }

365  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Not implemented for Thom encoding");
366  return false;

417  // add minus infinity
418  smtrat::vs::Substitution sub (eliminationVar, smtrat::vs::Substitution::MINUS_INFINITY, carl::PointerSet<smtrat::vs::Condition>());

435  });
436

439  return false;
440  }
441  }
442 
443  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Generated test candidates successfully");

468  static bool substitute(const VariableComparisonT& varcomp, const smtrat::vs::Substitution& substitution, const Model& model, FormulaT& result) {
469  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Substitute " << substitution << " into VariableComparison " << varcomp);
470 
471  static carl::Variable subVar1 = carl::freshRealVariable("__subVar1");
472  static carl::Variable subVar2 = carl::freshRealVariable("__subVar2");
473 
474  if (std::holds_alternative<MultivariateRootT>(varcomp.value()) && std::get<MultivariateRootT>(varcomp.value()).poly().has(substitution.variable())) {
475  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Substitution variable occurs in MVRoot's polynomial");
476  // Note that this does not occur if the theory variable order is fixed
477  // TODO can this case be handled?
478  return false;
479  }
480 
481  if (varcomp.var() == substitution.variable()) {
482  carl::Relation varcompRelation = varcomp.negated() ? carl::inverse(varcomp.relation()) : varcomp.relation();
483 
484  if (substitution.type() == smtrat::vs::Substitution::NORMAL || substitution.type() == smtrat::vs::Substitution::PLUS_EPSILON) {
485  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Substitution is of type NORMAL or PLUS_EPSILON");
486 
487  // convert MVRoot/RAN to SqrtExpression with side conditions
488  SqrtEx zero;
489  ConstraintsT scs;
490  if (!generateZeros(varcomp, substitution.variable(), model, [&](SqrtEx&& sqrtExpression, ConstraintsT&& sideConditions) {
491  zero = sqrtExpression;
492  scs = sideConditions;
493  })) {
494  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Could not generate zero for contained MVRoot/RAN");
495  return false;
496  }
497 
498  // calculate subVar1-subVar2 ~ 0 [substitution.term()//subVar1][zero//subVar2]
499  ConstraintT subConstraint(Poly(subVar1) - subVar2, varcompRelation);
500  smtrat::vs::Substitution subSub1(subVar1, substitution.term(), substitution.type(), carl::PointerSet<smtrat::vs::Condition>());
501  smtrat::vs::Substitution subSub2(subVar2, zero, smtrat::vs::Substitution::Type::NORMAL, carl::PointerSet<smtrat::vs::Condition>());
502 
503  smtrat::vs::DisjunctionOfConstraintConjunctions subRes1;
504  if (!substituteHelper(subConstraint, subSub1, subRes1)) {
505  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Substitution failed");
506  assert(false);
507  return false;
508  }
509  std::vector<FormulaT> disjF;
510  for (const auto& conj : subRes1) {
511  std::vector<FormulaT> conjF;
512  for (const auto& constr : conj) {
513  smtrat::vs::DisjunctionOfConstraintConjunctions subRes2;
514  if (!substituteHelper(constr, subSub2, subRes2)) {
515  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Substitution failed");
516  assert(false);
