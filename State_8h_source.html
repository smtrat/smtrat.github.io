39  struct unsignedTripleCmp
40  {
41  bool operator ()( UnsignedTriple n1, UnsignedTriple n2 ) const
42  {
43  if( n1.first > n2.first )
44  return true;
45  else if( n1.first == n2.first )
46  {
47  if( n1.first != 1 )
48  return n1.second.first > n2.second.first;
49  else
50  {
51  if( n1.second.second < n2.second.second )
52  return true;
53  else if( n1.second.second == n2.second.second )
54  return n1.second.first > n2.second.first;
55  return false;
56  }
57  }
58  else
59  return false;
60  }
61  };
