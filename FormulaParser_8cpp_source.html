71 FormulaT FormulaParser::mkConstraint(const Poly& lhs, const Poly& rhs, carl::Relation rel) {
72  Poly p = lhs - rhs;
73  std::set<carl::Variable> pVars = p.gatherVariables();
74  std::vector<carl::Variable> vars;
75  while (!pVars.empty()) {
76  auto it = state->mTheoryItes.find(*pVars.begin());
77  pVars.erase(pVars.begin());
78  if (it != state->mTheoryItes.end()) {
79  std::get<1>(it->second).gatherVariables(pVars);
80  std::get<2>(it->second).gatherVariables(pVars);
81  vars.push_back(it->first);
82  }
83  }
84  std::size_t n = vars.size();
85  if (n == 0) {
86  // There are no ITEs.
87  ConstraintT cons = ConstraintT(p, rel);
88  return FormulaT(cons);
89  } else if (n < 4) {
90  // There are only a few ITEs, hence we expand them here directly to 2^n cases.
91  // 2^n Polynomials with values substituted.
92  std::vector<Poly> polys({p});
93  // 2^n Formulas collecting the conditions.
94  std::vector<FormulasT> conds(1 << n);
95  unsigned repeat = 1 << (n-1);
96  for (auto v: vars) {
97  auto t = state->mTheoryItes[v];
98  std::vector<Poly> ptmp;
99  for (auto& p: polys) {
100  // Substitute both possibilities for this ITE.
101  ptmp.push_back(p.substitute(v, std::get<1>(t)));
102  ptmp.push_back(p.substitute(v, std::get<2>(t)));
103  }
104  std::swap(polys, ptmp);
105  // Add the conditions at the appropriate positions.
106  FormulaT f[2]= { std::get<0>(t), FormulaT(carl::FormulaType::NOT, std::get<0>(t)) };
107  for (size_t i = 0; i < (size_t)(1 << n); i++) {
108  conds[i].insert(f[0]);
109  if ((i+1) % repeat == 0) std::swap(f[0], f[1]);
110  }
111  repeat /= 2;
112  }
113  // Now combine everything: (and (=> (and conditions) constraint) ...)
114  FormulasT subs;
115  for (unsigned i = 0; i < polys.size(); i++) {
116  subs.insert(FormulaT(carl::FormulaType::IMPLIES, FormulaT(carl::FormulaType::AND, conds[i]), FormulaT(polys[i], rel)));
117  }
118  auto res = FormulaT(carl::FormulaType::AND, subs);
119  return res;
120  } else {
121  // There are many ITEs, we keep the auxiliary variables.
122  for (auto v: vars) {
123  auto t = state->mTheoryItes[v];
124  FormulaT consThen = FormulaT(std::move(carl::makePolynomial<Poly>(v) - std::get<1>(t)), carl::Relation::EQ);
125  FormulaT consElse = FormulaT(std::move(carl::makePolynomial<Poly>(v) - std::get<2>(t)), carl::Relation::EQ);
126 
127  state->mTheoryIteBindings.emplace(FormulaT(carl::FormulaType::IMPLIES,std::get<0>(t), consThen));
128  state->mTheoryIteBindings.emplace(FormulaT(carl::FormulaType::IMPLIES,FormulaT(carl::FormulaType::NOT,std::get<0>(t)), consElse));
129  }
130  return FormulaT(p, rel);
131  }
