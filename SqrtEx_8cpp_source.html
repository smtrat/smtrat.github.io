281  bool SqrtEx::evaluate( smtrat::Rational& _result, const smtrat::EvalRationalMap& _evalMap, int _rounding ) const
282  {
283  smtrat::Poly radicandEvaluated = radicand().substitute( _evalMap );
284  assert( radicandEvaluated.isConstant() );
285  smtrat::Rational radicandValue = radicandEvaluated.constantPart();
286  assert( radicandValue >= 0 );
287  smtrat::Poly factorEvaluated = factor().substitute( _evalMap );
288  assert( factorEvaluated.isConstant() );
289  smtrat::Rational factorValue = factorEvaluated.constantPart();
290  smtrat::Poly constantPartEvaluated = constantPart().substitute( _evalMap );
291  assert( constantPartEvaluated.isConstant() );
292  smtrat::Rational constantPartValue = constantPartEvaluated.constantPart();
293  smtrat::Poly denomEvaluated = denominator().substitute( _evalMap );
294  assert( denomEvaluated.isConstant() );
295  smtrat::Rational denomValue = denomEvaluated.constantPart();
296  assert( !carl::isZero( denomValue ) );
297  // Check whether the resulting assignment is integer.
298  bool rounded = true;
299  smtrat::Rational sqrtExValue;
300  if( !carl::sqrt_exact( radicandValue, sqrtExValue ) )
301  {
302  assert( _rounding != 0 );
303  rounded = false;
304  assert( factorValue != 0 );
305  double dbSqrt = sqrt( carl::toDouble( radicandValue ) );
306  sqrtExValue = carl::rationalize<smtrat::Rational>( dbSqrt ) ;
307  // As there is no rational number representing the resulting square root we have to round.
308  if( _rounding < 0 ) // If the result should round down in this case.
309  {
310  if( factorValue > 0 && (sqrtExValue*sqrtExValue) > radicandValue )
311  {
312  // The factor of the resulting square root is positive, hence force rounding down.
313  dbSqrt = std::nextafter( dbSqrt, -INFINITY );
314  sqrtExValue = carl::rationalize<smtrat::Rational>( dbSqrt );
315  assert( !((sqrtExValue*sqrtExValue) > radicandValue) );
316  }
317  else if( factorValue < 0 && (sqrtExValue*sqrtExValue) < radicandValue )
318  {
319  // The factor of the resulting square root is negative, hence force rounding up.
320  dbSqrt = std::nextafter( dbSqrt, INFINITY );
321  sqrtExValue = carl::rationalize<smtrat::Rational>( dbSqrt );
322  assert( !((sqrtExValue*sqrtExValue) < radicandValue) );
323  }
324  }
325  else if( _rounding > 0 ) // If the result should round up in this case.
326  {
327  if( factorValue < 0 && (sqrtExValue*sqrtExValue) > radicandValue )
328  {
329  // The factor of the resulting square root is negative, hence force rounding down.
330  dbSqrt = std::nextafter( dbSqrt, -INFINITY );
331  sqrtExValue = carl::rationalize<smtrat::Rational>( dbSqrt );
332  assert( !((sqrtExValue*sqrtExValue) > radicandValue) );
333  }
334  else if( factorValue > 0 && (sqrtExValue*sqrtExValue) < radicandValue )
335  {
336  // The factor of the resulting square root is positive, hence force rounding up.
337  dbSqrt = std::nextafter( dbSqrt, INFINITY );
338  sqrtExValue = carl::rationalize<smtrat::Rational>( dbSqrt );
339  assert( !((sqrtExValue*sqrtExValue) < radicandValue) );
340  }
341  }
342  }
343  _result = (constantPartValue + factorValue * sqrtExValue) / denomValue;
344  return rounded;
345  }

365  SqrtEx SqrtEx::subBySqrtEx( const smtrat::Poly& _substituteIn, const carl::Variable& _varToSubstitute, const SqrtEx& _substituteBy )
366  {
367  if( !_substituteIn.has( _varToSubstitute ) )
368  return SqrtEx( _substituteIn );
369  /*
370  * We have to calculate the result of the substitution:
371  *
372  * q+r*sqrt{t}
373  * (a_n*x^n+...+a_0)[------------ / x]
374  * s
375  * being:
376  *
377  * sum_{k=0}^n (a_k * (q+r*sqrt{t})^k * s^{n-k})
378  * ----------------------------------------------
379  * s^n
380  */
381  smtrat::VarPolyInfo varInfo = _substituteIn.getVarInfo<true>( _varToSubstitute );
382  const auto& coeffs = varInfo.coeffs();
383  // Calculate the s^k: (0<=k<=n)
384  auto coeff = coeffs.begin();
385  carl::uint lastDegree = varInfo.maxDegree();
386  vector<smtrat::Poly> sk;
387  sk.push_back( smtrat::ONE_POLYNOMIAL );
388  for( carl::uint i = 1; i <= lastDegree; ++i )
389  {
390  // s^i = s^l * s^{i-l}
391  sk.push_back( sk.back() * _substituteBy.denominator() );
392  }
393  // Calculate the constant part and factor of the square root of (q+r*sqrt{t})^k
394  vector<smtrat::Poly> qk;
395  qk.push_back( _substituteBy.constantPart() );
396  vector<smtrat::Poly> rk;
397  rk.push_back( _substituteBy.factor() );
398  // Let (q+r*sqrt{t})^l be (q'+r'*sqrt{t})
399  // then (q+r*sqrt{t})^l+1 = (q'+r'*sqrt{t}) * (q+r*sqrt{t}) = ( q'*q+r'*r't + (q'*r+r'*q) * sqrt{t} )
400  for( carl::uint i = 1; i < lastDegree; ++i )
401  {
402  // q'*q+r'*r't
403  qk.push_back( qk.back() * _substituteBy.constantPart() + rk.back() * _substituteBy.factor() * _substituteBy.radicand() );
404  // q'*r+r'*q
405  rk.push_back( rk.back() * _substituteBy.constantPart() + qk.at( i - 1 ) * _substituteBy.factor() );
406  }
407  // Calculate the result:
408  smtrat::Poly resFactor = smtrat::ZERO_POLYNOMIAL;
409  smtrat::Poly resConstantPart = smtrat::ZERO_POLYNOMIAL;
410  if( coeff->first == 0 )
411  {
412  resConstantPart += sk.back() * coeff->second;
413  ++coeff;
414  }
415  for( ; coeff != coeffs.end(); ++coeff )
416  {
417  resConstantPart += coeff->second * qk.at( coeff->first - 1 ) * sk.at( lastDegree - coeff->first );
418  resFactor += coeff->second * rk.at( coeff->first - 1 ) * sk.at( lastDegree - coeff->first );
419  }
420  return SqrtEx( resConstantPart, resFactor, sk.back(), _substituteBy.radicand() );
421  }
