66 RedProc RedProc_new(const char *reduce) {
67  pid_t processId;
68 
69 #ifdef LOGFILE
70  if (!logfile) logfile = RedMsg_openLogfile("libreduce.log",logfile);
71 #endif
72 
73 #ifdef DEBUG
74  if (!debugfile) debugfile = stderr;
75 #endif
76 
77  process = (RedProc)malloc(sizeof(struct oRedProc));
78 
79  RedProc_initSockets(process);
80 
81  signal(SIGCHLD,RedProc_SigChld);
82 
83  (void)setsid(); /* become the leader of a new process group */
84 
85  if ((processId = fork())) { /* I am not the child */
86 
87  if (processId < 0) { /* failure */
88  perror("libreduce: cannot fork()");
89  exit(errno);
90  }
91 
92  RedMsg_cfprintf(logfile,"successfully forked %d\n\n",processId);
93 
94  RedMsg_cfprintf(debugfile,
95  "-- reduce_init: parent process alive - fork()=%d\n",
96  processId);
97 
98  process->processId = processId;
99 
100  return RedProc_parent(process);
101 
102  } else { /* I am the child */
103 
104  RedMsg_cfprintf(debugfile,
105  "-- reduce_init: child process alive - fork()=%d\n",
106  processId);
107 
108  RedProc_child(process,reduce);
109  }
110 
111  return (RedProc)0; /* just to make GCC happy */
112 }
