42  bool CubeLIAModule<Settings>::addCore( ModuleInput::const_iterator _subformula )
43  {
44  if( _subformula->formula().getType() == carl::FormulaType::CONSTRAINT && !_subformula->formula().propertyHolds( carl::PROP_CONTAINS_REAL_VALUED_VARS ) )
45  {
46  const ConstraintT& constraint = _subformula->formula().constraint();
47  if( constraint.lhs().isLinear() && constraint.relation() != carl::Relation::NEQ && constraint.relation() != carl::Relation::EQ )
48  {
49  auto iter = mCubifications.find( _subformula->formula() );
50  if( iter == mCubifications.end() )
51  {
52  // For all variables in the constraint, which do not yet have a real relaxation, create one.
53  carl::Variables vars = constraint.lhs().gatherVariables();
54  for( carl::Variable::Arg var : vars )
55  {
56  if( var.type() == carl::VariableType::VT_INT || var.type() == carl::VariableType::VT_BOOL )
57  {
58  auto iter = mIntToRealVarMap.find( var );
59  if( iter == mIntToRealVarMap.end() )
60  {
61  carl::Variable realVar = carl::freshRealVariable();
62  mIntToRealVarMap[var] = carl::makePolynomial<Poly>(realVar);
63  mRealToIntVarMap[realVar] = var;
64  }
65  }
66  }
67  // Create the real relaxation of the constraint's left-hand side.
68  Poly realRelax = carl::substitute(constraint.lhs(), mIntToRealVarMap);
69  #ifdef DEBUG_CUBELIAMODULE
70  std::cout << "mIntToRealVarMap: " << mIntToRealVarMap << std::endl;
71  std::cout << "Real relaxation of " << constraint.lhs() << " is " << realRelax << std::endl;
72  #endif
73  // Find the 1-norm of the left-hand side's coefficients.
74  Rational norm = 0;
75  for( auto& term : realRelax.getTerms() )
76  {
77  if( !term.isConstant() )
78  {
79  norm += carl::abs( term.coeff() );
80  }
81  }
82  // Add half of the found 1-norm to the real relaxation of the constraint's left-hand side.
83  realRelax += norm/Rational(2);
84  // Store this "cubification".
85  FormulaT cubification( realRelax, constraint.relation() );
86  auto ret = mLRAFormula->add( cubification, false );
87  assert( ret.second );
88  #ifdef DEBUG_CUBELIAMODULE
89  std::cout << "Add to internal LRAModule: " << cubification << std::endl;
90  #endif
91  mLRA.inform( cubification );
92  mLRA.add( ret.first );
93  mCubifications.emplace( _subformula->formula(), Cubification( cubification, ret.first ) );
94  }
95  else
96  {
97  // If the cubification has already been created, update the usage counter.
98  if( iter->second.mUsages == 0 )
99  {
100  // If the cubification is now active again, add it to the internal LRAModule.
101  auto ret = mLRAFormula->add( iter->second.mCubification );
102  #ifdef DEBUG_CUBELIAMODULE
103  std::cout << "Add to internal LRAModule: " << iter->second.mCubification << std::endl;
104  #endif
105  assert( ret.second );
106  mLRA.add( ret.first );
107  assert( iter->second.mPosition == mLRAFormula->end() );
108  iter->second.mPosition = ret.first;
109  }
110  ++iter->second.mUsages;
111  }
112  }
113  }
114  addReceivedSubformulaToPassedFormula( _subformula );
115  return true;
116  }

154  template<class Settings>
155  Answer CubeLIAModule<Settings>::checkCore()
156  {
157  #ifdef DEBUG_CUBELIAMODULE
158  print();
159  #endif
160  Answer ans;
161  if( !rReceivedFormula().isRealConstraintConjunction() )
162  {
163  #ifdef DEBUG_CUBELIAMODULE
164  std::cout << "Call internal LRAModule:" << std::endl;
165  mLRA.print();
166  #endif
167  mLRA.clearLemmas();
168  mLRAFormula->updateProperties();
169  ans = mLRA.check( false, mFullCheck, objective() );
170  switch( ans )
171  {
172  case SAT:
173  {
174  clearModel();
175  // Get the model of mLRA
176  mLRA.updateModel();
177  const Model& relaxedModel = mLRA.model();
178  auto iter = mRealToIntVarMap.begin();
179  for( auto& ass : relaxedModel )
180  {
181  assert( iter != mRealToIntVarMap.end() );
182  // Round the result to the next integer
183  mModel.emplace_hint( mModel.end(), iter->second, carl::round( ass.second.asRational() ) );
184  ++iter;
185  }
186  // Check if the rounded model satisfies the received formula
187  bool receivedFormulaSatisfied = true;
188  for( const FormulaWithOrigins& fwo : rReceivedFormula() )
189  {
190  unsigned res = satisfies( mModel, fwo.formula() );
191  switch( res )
192  {
193  case 0:
194  case 2:
195  receivedFormulaSatisfied = false;
196  default:
197  break;
198  }
199  if( !receivedFormulaSatisfied )
200  break;
201  }
202  // If we found a model, we know that the formula is satisfiable, otherwise, we have to call the backends on the received formula
203  if( receivedFormulaSatisfied )
204  {
205  mModelUpdated = true;
206  return SAT;
207  }
208  clearModel();
209  break;
210  }
211  case UNSAT:
212  {
213  if( Settings::exclude_unsatisfiable_cube_space )
214  {
215  // Exclude the space for which mLRA has detected unsatisfiability
216  for( auto& infsubset : mLRA.infeasibleSubsets() )
217  {
218  FormulasT formulas;
219  for( auto& formula : infsubset )
220  formulas.push_back( formula.negated() );
221  addLemma( FormulaT( carl::FormulaType::OR, formulas ) );
222  }
223  }
224  break;
225  }
226  default:
227  assert( false );
228  }
229  }
230  #ifdef DEBUG_CUBELIAMODULE
231  std::cout << "Call Backends:" << std::endl;
232  #endif
233  // Run backends on received formula
234  ans = runBackends();
235  if( ans == UNSAT )
236  getInfeasibleSubsets();
237  else if( ans == SAT )
238  mModelUpdated = false;
239  return ans;
240  }
241 }
