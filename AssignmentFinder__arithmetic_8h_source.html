33  bool isUnivariate(const FormulaT& f) const {
34  return mcsat::constraint_type::isUnivariate(f, mModel, mVar);
35  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", "is " << f << " univariate in " << mVar << "?");
36  carl::Variables vars;
37  f.arithmeticVars(vars);
38  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", "Collected " << vars);
39  auto it = vars.find(mVar);
40  if (it == vars.end()) return false;
41  vars.erase(it);
42  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", "Checking whether " << mModel << " covers " << vars);
43  return mModel.contains(vars);
44  }
45  bool satisfies(const FormulaT& f, const RAN& r) const {
46  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", f << ", " << mModel << ", " << mVar << ", " << r);
47  Model m = mModel;
48  m.assign(mVar, r);
49  auto res = carl::model::evaluate(f, m);
50  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", "Evaluating " << f << " on " << m << " -> " << res);
51  if (!res.isBool()) return true;
52  assert(res.isBool());
53  return res.asBool();
54  }

56  bool compare_assignments(std::size_t lhs, std::size_t rhs) const {
57  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", lhs << " < " << rhs << "?");
58  const auto& l = mRI.sampleFrom(lhs);
59  const auto& r = mRI.sampleFrom(rhs);
60  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", l << " < " << r << "?");
61  if (l.isIntegral() != r.isIntegral()) return l.isIntegral();
62  if (l.isNumeric() != r.isNumeric()) return l.isNumeric();
63  if (l.size() != r.size()) return l.size() < r.size();
64  if (l.abs() != r.abs()) return l.abs() < r.abs();
65  return l < r;
66  }

68  ModelValue selectAssignment(const Covering& cover) const {
69  std::vector<std::size_t> samples;
70  for (auto b: cover.satisfyingSamples()) {
71  samples.push_back(b);
72  }
73  assert(samples.size() > 0);
74  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Finding minimum from " << samples);
75  auto min = std::min_element(samples.begin(), samples.end(), [this](auto lhs, auto rhs){ return this->compare_assignments(lhs, rhs); });
76  return mRI.sampleFrom(*min);
77  }

82  bool addConstraint(const FormulaT& f) {
83  assert(f.getType() == carl::FormulaType::CONSTRAINT);
84  auto category = mcsat::constraint_type::categorize(f, mModel, mVar);
85  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", f << " is " << category << " under " << mModel << " w.r.t. " << mVar);
86  switch (category) {
87  case mcsat::ConstraintType::Constant:
88  assert(f.isTrue() || f.isFalse());
89  if (f.isFalse()) return false;
90  break;
91  case mcsat::ConstraintType::Assigned: {
92  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", "Checking fully assigned " << f);
93  FormulaT fnew = carl::model::substitute(f, mModel);
94  if (fnew.isTrue()) {
95  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", "Ignoring " << f << " which simplified to true.");
96  return true;
97  } else {
98  assert(fnew.isFalse());
99  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Conflict: " << f << " simplified to false.");
100  return false;
101  }
102  break;
103  }
104  case mcsat::ConstraintType::Univariate:
105  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Considering univariate constraint " << f);
106  break;
107  case mcsat::ConstraintType::Unassigned:
108  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Considering unassigned constraint " << f << " (which may still become univariate)");
109  return true;
110  break;
111  }
112  FormulaT fnew(carl::model::substitute(f, mModel));
113  std::vector<RAN> list;
114  if (fnew.getType() == carl::FormulaType::CONSTRAINT) {
115  const auto& poly = fnew.constraint().lhs();
116  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", "Real roots of " << poly << " in " << mVar << " w.r.t. " << mModel);
117  auto roots = carl::model::tryRealRoots(poly, mVar, mModel);
118  if (roots) {
119  list = *roots;
120  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", "-> " << list);
121  } else {
122  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Failed to compute roots because polynomial is not univariate or became zero.");
123  mMVBounds.emplace_back(f);
124  return true;
125  }
126  } else if (fnew.isTrue()) {
127  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", "Ignoring " << f << " which simplified to true.");
128  return true;
129  } else {
130  assert(fnew.isFalse());
131  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Conflict: " << f << " simplified to false.");
132  return false;
133  }
134 
135  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Adding " << list << " with " << fnew);
136  mRI.add(list);
137  mRootMap.emplace(f, std::make_pair(std::move(list), fnew));
138  return true;
139  }

141  bool addMVBound(const FormulaT& f) {
142  assert(f.getType() == carl::FormulaType::VARCOMPARE);
143  if (!isUnivariate(f)) {
144  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Ignoring non-univariate bound " << f);
145  return true;
146  }
147  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Adding univariate bound " << f);
148  FormulaT fnew(carl::model::substitute(f, mModel));
149  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "-> " << fnew);
150  if (fnew.isTrue()) {
151  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Bound evaluated to true, we can ignore it.");
152  return true;
153  } else if (fnew.isFalse()) {
154  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Conflict: " << f << " simplified to false.");
155  return false;
156  }
157  assert(fnew.getType() == carl::FormulaType::VARCOMPARE);
158  ModelValue value = fnew.variableComparison().value();
159  if (value.isSubstitution()) {
160  // Prevent memory error due to deallocation of shared_ptr before copying value from shared_ptr.
161  auto res = value.asSubstitution()->evaluate(mModel);
162  value = res;
163  }
164  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Evaluated to " << value);
165  if (!value.isRational() && !value.isRAN()) {
166  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Value is neither rational nor RAN, cannot generate roots from it");
167  if (value.isBool()) {
168  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "But it is bool");
169  assert(false);
170  }
171  mMVBounds.emplace_back(fnew);
172  return true;
173  }
174  std::vector<RAN> list;
175  if (value.isRational()) list.emplace_back(value.asRational());
176  else list.push_back(value.asRAN());
177  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Adding " << list << " with " << fnew);
178  mRI.add(list);
179  mRootMap.emplace(f, std::make_pair(std::move(list), fnew));
180  return true;
181  }

183  Covering computeCover() {
184  mRI.process();
185  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", mRI);
186  Covering cover(mRI.size() * 2 + 1);
187  for (const auto& c: mRootMap) {
188  carl::Bitset b;
189  const auto& roots = c.second.first; // sorted
190  const auto& constraint = c.second.second;
191  std::size_t last = 0;
192  for (const auto& r: roots) {
193  std::size_t cur = mRI[r];
194  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", constraint << " vs " << mRI.sampleFrom(2*cur));
195  if (!satisfies(constraint, mRI.sampleFrom(2*cur))) {
196  // Refutes interval left of this root
197  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", constraint << " refutes " << mRI.sampleFrom(2*cur) << " -> " << last << ".." << (2*cur));
198  b.set_interval(last, 2*cur);
199  }
200  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", constraint << " vs " << mRI.sampleFrom(2*cur+1));
201  if (!satisfies(constraint, r)) {
202  // Refutes root
203  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", constraint << " refutes " << r << " -> " << 2*cur+1);
204  b.set(2*cur+1, 2*cur+1);
205  }
206  last = 2*cur + 2;
207  }
208  SMTRAT_LOG_TRACE("smtrat.mcsat.assignmentfinder", constraint << " vs " << mRI.sampleFrom(last));
209  if (!satisfies(constraint, mRI.sampleFrom(last))) {
210  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", constraint << " refutes " << mRI.sampleFrom(last) << " which is " << mRI.sampleFrom(roots.size()*2));
211  // Refutes interval right of largest root
212  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", constraint << " refutes " << mRI.sampleFrom(roots.size()*2) << " -> " << last << ".." << (mRI.size()*2));
213  b.set_interval(last, mRI.size()*2);
214  }
215  if (b.any()) {
216  cover.add(c.first, b);
217  }
218  }

220  // but cannot be found by the assignment finder. Thus, satisfying cells may be excluded.
221  // Currently, these bounds are disabled in the BaseBackend/Bookkeeping, but they may be handled using
222  // the CAD.
223  assert(mMVBounds.empty());
224  /*
225  for (const auto& c: mMVBounds) {
226  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Computing cover for " << c);
227  carl::Bitset b;
228  for (std::size_t i = 0; i < mRI.size() * 2 + 1; ++i) {
229  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", c << " vs " << mRI.sampleFrom(i));
230 
231  Model m = mModel;
232  m.assign(mVar, mRI.sampleFrom(i));
233  auto res = carl::model::evaluate(c, m);
234  if (!res.isBool()) {
235  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", c << " is inconclusive on " << mRI.sampleFrom(i));
236  } else if (!res.asBool()) {
237  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", c << " refutes " << mRI.sampleFrom(i));
238  b.set(i);
239  }
240  }
241  if (b.any()) {
242  cover.add(c, b);
243  }
244  }
245  */
246  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", cover);
247  return cover;

250  AssignmentOrConflict findAssignment() {
251  Covering cover = computeCover();
252  if (cover.conflicts()) {
253  FormulasT conflict;
254  cover.buildConflictingCore(conflict);
255  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "No Assignment, built conflicting core " << conflict << " under model " << mModel);
256  return conflict;
257  } else {
258  ModelValue assignment = selectAssignment(cover);
259  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Assignment: " << mVar << " = " << assignment << " from interval " << cover.satisfyingInterval());
260  assert(assignment.isRAN());
261  if (assignment.asRAN().isNumeric()) {
262  assignment = assignment.asRAN().value();
263  }
264  SMTRAT_LOG_DEBUG("smtrat.mcsat.assignmentfinder", "Assignment: " << mVar << " = " << assignment);
265  ModelValues res;
266  res.push_back(std::make_pair(mVar,assignment));
267  return std::move(res);
268  }
269  }
