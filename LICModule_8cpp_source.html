185  template<class Settings>
186  bool LICModule<Settings>::isSuitable(const ConstraintT& c, TermT& src, std::vector<TermT>& dest, Coefficient& coeff) {
187  SMTRAT_LOG_FUNC("smtrat.lic", c);
188  bool invert = false;
189  src = TermT();
190  dest.clear();
191  coeff.strict = false;
192  coeff.r = 0;
193 
194  switch (c.relation()) {
195  case carl::Relation::EQ: break;
196  case carl::Relation::NEQ: return false;
197  case carl::Relation::LESS:
198  invert = true;
199  coeff.strict = true;
200  break;
201  case carl::Relation::LEQ:
202  invert = true;
203  break;
204  case carl::Relation::GREATER:
205  coeff.strict = true;
206  break;
207  case carl::Relation::GEQ: break;
208  }
209  Poly p = c.lhs();
210  if (invert) p = -p;
211 
212  for (const auto& term: p) {
213  if (term.isConstant()) {
214  coeff.r += term.coeff();
215  continue;
216  }
217  if (isZero(term)) {
218  SMTRAT_LOG_WARN("smtrat.lic", "Term " << term << " is zero. We'll ignore it.");
219  continue;
220  }
221  if (isSemiPositive(term)) {
222  if (!src.isZero()) {
223  SMTRAT_LOG_TRACE("smtrat.lic", "No: Already has a source, but " << term << " was found.");
224  return false;
225  }
226  src = term;
227  } else if (isSemiNegative(term)) {
228  dest.push_back(-term);
229  } else {
230  SMTRAT_LOG_TRACE("smtrat.lic", "No: Has indefinite term " << term << ".");
231  return false;
232  }
233  }
234  if (dest.empty()) {
235  SMTRAT_LOG_TRACE("smtrat.lic", "No: No destinations were found.");
236  return false;
237  }
238  SMTRAT_LOG_TRACE("smtrat.lic", "Yes: " << src << " -> " << dest << " with coefficient " << coeff << ".");
239  return true;
240  }

289  template<class Settings>
290  Answer LICModule<Settings>::analyzeCycle(const Graph& g, const Vertex& root) {
291  Coefficient sum;
292  Vertex cur = root;
293  std::size_t component = g[cur].component;
294  std::vector<Vertex> cycle; // Variables on the cycle
295  std::vector<Vertex> edges; // Edges on the cycle
296  std::vector<Vertex> others; // Variables adjacent to the cycle
297  while (true) {
298  cycle.push_back(cur);
299  // Get edge node
300  Vertex edgeNode = nextInComponent(g, cur, component);
301  edges.push_back(edgeNode);
302  // Get next variable node
303  Vertex targetNode = nextInComponent(g, edgeNode, component, &others);
304  sum += g[edgeNode].coeff;
305  if (targetNode == root) break;
306  cur = targetNode;
307  }
308  if (carl::isZero(sum.r)) {
309  FormulaT origin = buildCycleFormula(g, edges);
310  if (sum.strict) {
311  // Sum is zero but there is a strict inequality -> UNSAT
312  mInfeasibleSubsets.emplace_back(origin.subformulas().begin(), origin.subformulas().end());
313  return False;
314  } else {
315  // Sum is zero and all inequalities are weak
316  // -> Variables on the cycle
317  for (const auto& v: others) {
318  FormulaT lemma(Poly(g[v].term), carl::Relation::EQ);
319  SMTRAT_LOG_DEBUG("smtrat.lic", "Deducted " << lemma);
320  addSubformulaToPassedFormula(lemma, origin);
321  }
322  for (std::size_t i = 1; i < cycle.size(); i++) {
323  const TermT& a = g[cycle[i-1]].term;
324  const TermT& b = g[cycle[i]].term;
325  FormulaT lemma(a - b + g[edges[i-1]].coeff.r, carl::Relation::EQ);
326  SMTRAT_LOG_DEBUG("smtrat.lic", "Deducted " << lemma);
327  addSubformulaToPassedFormula(lemma, origin);
328  }
329  return True;
330  }
331  } else if (sum.r > 0) {
332  std::cout << "What to do? sum > 0" << std::endl;
333  } else if (sum.r < 0) {
334  FormulaT origin = buildCycleFormula(g, edges);
335  mInfeasibleSubsets.emplace_back(origin.subformulas().begin(), origin.subformulas().end());
336  return False;
337  }
338  return True;
339  }
