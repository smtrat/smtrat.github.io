123  bool isSymbolFree(const std::string& name, bool output = true) {
124  std::stringstream out;
125  if (name == "true" || name == "false") out << "\"" << name << "\" is a reserved keyword.";
126  else if (variables.find(name) != variables.end()) out << "\"" << name << "\" has already been defined as a variable.";
127  else if (bindings.find(name) != bindings.end()) out << "\"" << name << "\" has already been defined as a binding to \"" << bindings[name] << "\".";
128  else if (constants.find(name) != constants.end()) out << "\"" << name << "\" has already been defined as a constant.";
129  else if (declared_functions.find(name) != declared_functions.end()) out << "\"" << name << "\" has already been declared as a function.";
130  else if (defined_functions.find(name) != defined_functions.end()) out << "\"" << name << "\" has already been defined as a function.";
131  else if (defined_indexed_functions.find(name) != defined_indexed_functions.end()) out << "\"" << name << "\" has already been defined as a function.";
132  else if (defined_user_functions.find(name) != defined_user_functions.end()) out << "\"" << name << "\" has already been defined as a function by the user.";
133  else return true;
134  if (output) SMTRAT_LOG_ERROR("smtrat.parser", out.str());
135  return false;
136  }
