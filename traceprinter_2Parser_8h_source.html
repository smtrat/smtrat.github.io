116  Parser() {
117  token = qi::as_string[qi::lexeme[ *(~qi::char_("<>()") | qi::blank) ]];
118  token.name("token");
119  tpl = token >> "<" >> expressionlist >> ">";
120  tpl.name("template");
121  function = token >> "(" >> expressionlist >> ")";
122  function.name("function");
123  expression = tpl | function | token;
124  expression.name("expression");
125  expressionlist = expression % ", ";
126  linePrefix = "==" >> +qi::digit >> "==" >> *qi::blank;
127  linePrefix.name("line prefix");
128  lineAddr = (qi::lit("at 0x") | qi::lit("by 0x")) >> +qi::alnum;
129  lineAddr.name("line address");
130  lineContent = qi::as_string[qi::no_skip[ *(qi::char_ - qi::eol) ]];
131  lineContent.name("line content");
132  line = linePrefix >> (
133  (lineAddr >> ": " >> expression)
134  | lineContent
135  );
136  line.name("line");
