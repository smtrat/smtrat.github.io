40  FormulaT OPBConverterModule<Settings>::convertPbConstraintToConstraintFormula(const FormulaT& formula) {
41  assert(formula.getType() == carl::FormulaType::PBCONSTRAINT);
42 
43  const PBConstraintT& pbConstraint = formula.pbConstraint();
44  // extract the parts we are working with
45  const std::vector<std::pair<Rational, carl::Variable>>& pbLhs = pbConstraint.getLHS();
46  const Rational& rhs = pbConstraint.getRHS();
47 
48  Poly lhs;
49  for (const auto& term : pbLhs) {
50  auto it = mVariablesCache.find(term.second);
51  if (it == mVariablesCache.end()) {
52  // We haven't seen this variable, yet. Create a new map entry for it.
53  mVariablesCache[term.second] = carl::freshBooleanVariable();
54  }
55 
56  const carl::Variable& booleanVariable = mVariablesCache[term.second];
57  lhs += Poly(term.first) * Poly(booleanVariable);
58  }
59 
60  // the new constraint, based on the pbConstraint
61  ConstraintT constraint(lhs - Poly(rhs), pbConstraint.getRelation());
62 
63  SMTRAT_LOG_INFO("smtrat.pbc", "converted PBConstraint " << pbConstraint
64  << " to arithmetic constraint "
65  << constraint);
66 
67  return FormulaT(constraint);
68  }

76  // Hence, for compatibility, we convert the Formula to the correct type.

78  } else {
79  // pass through.
