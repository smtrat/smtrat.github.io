384  result_type P_dispatch(const FormulaT& formula) {
385  // make sure that ConcreteType is a complete type
386  constexpr std::size_t require_complete_type = sizeof(ConcreteType);
387  (void)require_complete_type;
388 
389  switch(formula.getType()) {
390  case carl::OR: {
391  P_enter_or(formula, SMTRAT_DISP_TAG(enter_or));
392 
393  set_type results;
394 
395  for(const FormulaT& subformula : formula.subformulas()) {
396  results.insert(P_dispatch(subformula));
397  }
398 
399  return P_visit_or(formula, std::move(results), SMTRAT_DISP_TAG(visit_or));
400  }
401 
402  case carl::AND: {
403  P_enter_and(formula, SMTRAT_DISP_TAG(enter_and));
404 
405  set_type results;
406 
407  for(const FormulaT& subformula : formula.subformulas()) {
408  results.insert(P_dispatch(subformula));
409  }
410 
411  return P_visit_and(formula, std::move(results), SMTRAT_DISP_TAG(visit_and));
412  }
413 
414  case carl::XOR: {
415  P_enter_xor(formula, SMTRAT_DISP_TAG(enter_xor));
416 
417  multiset_type results;
418 
419  for(const FormulaT& subformula : formula.subformulas()) {
420  results.insert(P_dispatch(subformula));
421  }
422 
423  return P_visit_xor(formula, std::move(results), SMTRAT_DISP_TAG(visit_xor));
424  }
425 
426  case carl::IFF: {
427  P_enter_iff(formula, SMTRAT_DISP_TAG(enter_iff));
428 
429  set_type results;
430 
431  for(const FormulaT& subformula : formula.subformulas()) {
432  results.insert(P_dispatch(subformula));
433  }
434 
435  return P_visit_iff(formula, std::move(results), SMTRAT_DISP_TAG(visit_iff));
436  }
437 
438  case carl::NOT: {
439  P_enter_not(formula, SMTRAT_DISP_TAG(enter_not));
440  result_type result(P_dispatch(formula.subformula()));
441  return P_visit_not(formula, std::move(result), SMTRAT_DISP_TAG(visit_not));
442  }
443 
444  case carl::TRUE: {
445  P_enter_true(formula, SMTRAT_DISP_TAG(enter_true));
446  return P_visit_true(formula, SMTRAT_DISP_TAG(visit_true));
447  }
448 
449  case carl::FALSE: {
450  P_enter_false(formula, SMTRAT_DISP_TAG(enter_false));
451  return P_visit_false(formula, SMTRAT_DISP_TAG(visit_false));
452  }
453 
454  case carl::BOOL: {
455  P_enter_bool(formula, SMTRAT_DISP_TAG(enter_bool));
456  return P_visit_bool(formula, SMTRAT_DISP_TAG(visit_bool));
457  }
458 
459  case carl::UEQ: {
460  P_enter_ueq(formula, SMTRAT_DISP_TAG(enter_ueq));
461  return P_visit_ueq(formula, SMTRAT_DISP_TAG(visit_ueq));
462  }
463 
464  case carl::CONSTRAINT: {
465  P_enter_constraint(formula, SMTRAT_DISP_TAG(enter_constraint));
466  return P_visit_constraint(formula, SMTRAT_DISP_TAG(visit_constraint));
467  }
468 
469  case carl::EXISTS: {
470  P_enter_exists(formula, SMTRAT_DISP_TAG(enter_exists));
471  result_type result(P_dispatch(formula.quantifiedFormula()));
472  return P_visit_exists(formula, std::move(result), SMTRAT_DISP_TAG(visit_exists));
473  }
474 
475  case carl::FORALL: {
476  P_enter_forall(formula, SMTRAT_DISP_TAG(enter_forall));
477  result_type result(P_dispatch(formula.quantifiedFormula()));
478  return P_visit_forall(formula, std::move(result), SMTRAT_DISP_TAG(visit_forall));
479  }
480 
481  case carl::IMPLIES: {
482  P_enter_implication(formula, SMTRAT_DISP_TAG(enter_implication));
483  result_type premise(P_dispatch(formula.premise()));
484  result_type conclusion(P_dispatch(formula.conclusion()));
485  return P_visit_implication(formula, std::move(premise), std::move(conclusion), SMTRAT_DISP_TAG(visit_implication));
486  }
487 
488  case carl::ITE: {
489  P_enter_ite(formula, SMTRAT_DISP_TAG(enter_ite));
490  result_type condition(P_dispatch(formula.condition()));
491  result_type true_case(P_dispatch(formula.firstCase()));
492  result_type false_case(P_dispatch(formula.secondCase()));
493  return P_visit_ite(formula, std::move(condition), std::move(true_case), std::move(false_case), SMTRAT_DISP_TAG(visit_ite));
494  }
495

498  P_enter_default(formula);
499  return P_visit_default(formula);
500  }
501  }

832  void P_dispatch(const FormulaT& formula) {
833  // make sure that ConcreteType is a complete type
834  constexpr std::size_t require_complete_type = sizeof(ConcreteType);
835  (void)require_complete_type;
836 
837  switch(formula.getType()) {
838  case carl::OR: {
839  P_enter_or(formula, SMTRAT_DISP_TAG(enter_or));
840 
841  for(const FormulaT& subformula : formula.subformulas()) {
842  P_dispatch(subformula);
843  }
844 
845  P_visit_or(formula, SMTRAT_DISP_TAG(visit_or));
846  return;
847  }
848 
849  case carl::AND: {
850  P_enter_and(formula, SMTRAT_DISP_TAG(enter_and));
851 
852  for(const FormulaT& subformula : formula.subformulas()) {
853  P_dispatch(subformula);
854  }
855 
856  P_visit_and(formula, SMTRAT_DISP_TAG(visit_and));
857  return;
858  }
859 
860  case carl::XOR: {
861  P_enter_xor(formula, SMTRAT_DISP_TAG(enter_xor));
862 
863  for(const FormulaT& subformula : formula.subformulas()) {
864  P_dispatch(subformula);
865  }
866 
867  P_visit_xor(formula, SMTRAT_DISP_TAG(visit_xor));
868  return;
869  }
870 
871  case carl::IFF: {
872  P_enter_iff(formula, SMTRAT_DISP_TAG(enter_iff));
873 
874  for(const FormulaT& subformula : formula.subformulas()) {
875  P_dispatch(subformula);
876  }
877 
878  P_visit_iff(formula, SMTRAT_DISP_TAG(visit_iff));
879  return;
880  }
881 
882  case carl::NOT: {
883  P_enter_not(formula, SMTRAT_DISP_TAG(enter_not));
884  P_dispatch(formula.subformula());
885  P_visit_not(formula, SMTRAT_DISP_TAG(visit_not));
886  return;
887  }
888 
889  case carl::TRUE: {
890  P_enter_true(formula, SMTRAT_DISP_TAG(enter_true));
891  P_visit_true(formula, SMTRAT_DISP_TAG(visit_true));
892  return;
893  }
894 
895  case carl::FALSE: {
896  P_enter_false(formula, SMTRAT_DISP_TAG(enter_false));
897  P_visit_false(formula, SMTRAT_DISP_TAG(visit_false));
898  return;
899  }
900 
901  case carl::BOOL: {
902  P_enter_bool(formula, SMTRAT_DISP_TAG(visit_bool));
903  P_visit_bool(formula, SMTRAT_DISP_TAG(visit_bool));
904  return;
905  }
906 
907  case carl::UEQ: {
908  P_enter_ueq(formula, SMTRAT_DISP_TAG(enter_ueq));
909  P_visit_ueq(formula, SMTRAT_DISP_TAG(visit_ueq));
910  return;
911  }
912 
913  case carl::CONSTRAINT: {
914  P_enter_constraint(formula, SMTRAT_DISP_TAG(enter_constraint));
915  P_visit_constraint(formula, SMTRAT_DISP_TAG(visit_constraint));
916  return;
917  }
918 
919  case carl::EXISTS: {
920  P_enter_exists(formula, SMTRAT_DISP_TAG(enter_exists));
921  P_dispatch(formula.quantifiedFormula());
922  P_visit_exists(formula, SMTRAT_DISP_TAG(visit_exists));
923  return;
924  }
925 
926  case carl::FORALL: {
927  P_enter_forall(formula, SMTRAT_DISP_TAG(enter_forall));
928  P_dispatch(formula.quantifiedFormula());
929  P_visit_forall(formula, SMTRAT_DISP_TAG(visit_forall));
930  return;
931  }
932 
933  case carl::IMPLIES: {
934  P_enter_implication(formula, SMTRAT_DISP_TAG(enter_implication));
935  P_dispatch(formula.premise());
936  P_dispatch(formula.conclusion());
937  P_visit_implication(formula, SMTRAT_DISP_TAG(visit_implication));
938  return;
939  }
940 
941  case carl::ITE: {
942  P_enter_ite(formula, SMTRAT_DISP_TAG(enter_ite));
943  P_dispatch(formula.condition());
944  P_dispatch(formula.firstCase());
945  P_dispatch(formula.secondCase());
946  P_visit_ite(formula, SMTRAT_DISP_TAG(visit_ite));
947  return;
948  }
949 
950  default: {
951  assert(false && "Unexpected formula encountered!");
952  P_enter_default(formula);
953  P_visit_default(formula);
954  return;
955  }
956  }
957  }
958  #undef SMTRAT_DISP_TAG
959  };
