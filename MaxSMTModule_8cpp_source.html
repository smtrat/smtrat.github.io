75  Answer MaxSMTModule<Settings>::checkCore()
76  {
77  SMTRAT_LOG_INFO("smtrat.maxsmt", "Running MAXSMT.");
78 
79  // check sat for hard constraints. If UNSAT for hard-clauses, we are UNSAT, for unknown we are lost as well.
80  Answer ans = mBackend.check();
81  SMTRAT_LOG_DEBUG("smtrat.maxsmt", "Checking satisfiability of hard clauses... Got " << ans);
82  if (ans != Answer::SAT) return ans;
83 
84  // populate all known soft clauses
85  for (auto softFormula : mpManager->weightedFormulas()) {
86  addSoftFormula(softFormula.first);
87  }
88 
89  SMTRAT_LOG_INFO("smtrat.maxsmt", "Trying to maximize weight for following soft formulas " << softclauses);
90 
91  std::vector<FormulaT> satiesfiedSoftClauses;
92  if (Settings::ALGORITHM == MAXSATAlgorithm::FU_MALIK_INCREMENTAL) {
93  SMTRAT_LOG_INFO("smtrat.maxsmt", "Running FUMALIK Algorithm.");
94  satiesfiedSoftClauses = runFuMalik();
95  } else if (Settings::ALGORITHM == MAXSATAlgorithm::LINEAR_SEARCH) {
96  SMTRAT_LOG_INFO("smtrat.maxsmt", "Running Linear Search Algorithm.");
97  satiesfiedSoftClauses = runLinearSearch();
98  } else if (Settings::ALGORITHM == MAXSATAlgorithm::MSU3) {
99  SMTRAT_LOG_INFO("smtrat.maxsmt", "Running MSU3 Algorithm.");
100  satiesfiedSoftClauses = runMSU3();
101  } else {
102  SMTRAT_LOG_ERROR("smtrat.maxsmt", "The selected MAXSATAlgorithm is not implemented.");
103  assert(false);
104  }
105 
106  SMTRAT_LOG_DEBUG("smtrat.maxsmt", "Found maximal set of satisfied soft clauses: " << satiesfiedSoftClauses);
107 
108  // at this point we can be sure to be SAT. Worst case is that all soft-constraints are false.
109  return Answer::SAT;
110

113  template<class Settings>
114  std::vector<FormulaT> MaxSMTModule<Settings>::runFuMalik()
115  {
116  getInfeasibleSubsets();
117  assert(mInfeasibleSubsets.size() == 0 && "hard clauses must be SAT and we haven't experienced any UNSAT yet.");
118 
119  // a set of assignments we need to keep track of the enabled clauses
120  std::map<carl::Variable, FormulaT> assignments;
121  std::map<FormulaT, FormulaT> extendedClauses;
122 
123  std::vector<FormulaT> newSoftClauses;
124 
125  // now add all soft clauses with a blocking variable which we assert to false in order to enable all soft clauses
126  // NB! if we added the soft clauses directly to the backend we would need to remove them later on which is not what we want
127  // in an incremental approach
128  for (const FormulaT& clause : softclauses) {
129  carl::Variable blockingVar = carl::freshBooleanVariable();
130 
131  // remember the blockingVar associated to clause
132  blockingVars[clause] = blockingVar;
133 
134  // add the clause v blockingVar to backend
135  FormulaT clauseWithBlockingVar(carl::FormulaType::OR, FormulaT(blockingVar), clause);
136  extendedClauses[clauseWithBlockingVar] = clause;
137  mBackend.add(clauseWithBlockingVar);
138 
139  // we may not add or remove from softclauses, hence we save a intermediate result which new soft clauses to add
140  newSoftClauses.push_back(clauseWithBlockingVar);
141 
142  // enable the clause related to blocking var
143  assignments[blockingVar] = !FormulaT(blockingVar);
144  formulaPositionMap[assignments[blockingVar]] = addToLocalBackend(assignments[blockingVar]);
145  }
146 
147  for (FormulaT& f : newSoftClauses) {
148  addSoftFormula(f); // need to add internally otherwise we assume that this would be a hard clause
149  SMTRAT_LOG_DEBUG("smtrat.maxsmt.fumalik", "Added new clause to backend " << f);
150  }
151 
152  // TODO implement weighted case according to https://www.seas.upenn.edu/~xsi/data/cp16.pdf
153  for (;;) {
154  std::vector<carl::Variable> relaxationVars;
155 
156  // try to solve
157  //Answer ans = runBackends();
158  Answer ans = mBackend.check();
159  SMTRAT_LOG_DEBUG("smtrat.maxsmt.fumalik", "Checked satisfiability of current soft/hardclause mixture got... " << ans);
160  if (ans == Answer::SAT) return gatherSatisfiedSoftClauses();
161 
162  for (auto it : getUnsatCore()) {
163 
164  if (!isSoft(it)) continue; // hard clauses are ignored here since we can not "relax".
165 
166  relaxationVars.push_back(carl::freshBooleanVariable()); // r
167  carl::Variable blockingRelaxationVar = carl::freshBooleanVariable(); // b_r
168 
169  // build new clause to add to formula
170  assert(extendedClauses.find(it) != extendedClauses.end());
171  FormulaT replacementClause = FormulaT(carl::FormulaType::OR, extendedClauses[it], FormulaT(relaxationVars.back()), FormulaT(blockingRelaxationVar));
172  addSoftFormula(replacementClause);
173 
174  extendedClauses[replacementClause] = it;
175  blockingVars[replacementClause] = blockingRelaxationVar;
176  assignments[blockingRelaxationVar] = !FormulaT(blockingRelaxationVar);
177 
178  SMTRAT_LOG_DEBUG("smtrat.maxsat.fumalik", "adding clause to backend: " << replacementClause);
179  mBackend.add(replacementClause);
180  formulaPositionMap[assignments[blockingRelaxationVar]] = addToLocalBackend(assignments[blockingRelaxationVar]);
181 
182  // disable original clause
183  carl::Variable prevBlockingVar = blockingVars[extendedClauses[it]];
184  const auto& prevAssignment = assignments.find(prevBlockingVar);
185  assert(prevAssignment != assignments.end() && "Could not find an assignment which we must have made before.");
186 
187  mBackend.remove(formulaPositionMap[prevAssignment->second]);
188  assignments.erase(prevAssignment);
189 
190  SMTRAT_LOG_DEBUG("smtrat.maxsat.fumalik", "adding clause to backend: " << FormulaT(prevBlockingVar));
191  mBackend.add(FormulaT(prevBlockingVar));
192 
193  }
194 
195  Poly relaxationPoly;
196  for (carl::Variable& var : relaxationVars) {
197  relaxationPoly = relaxationPoly + var;
198  }
199 
200  // \sum r \ in relaxationVars : r <= 1
201  FormulaT pbEncoding = FormulaT(ConstraintT(relaxationPoly - Rational(1),carl::Relation::LEQ));
202  mBackend.add(pbEncoding);
203  // addSubformulaToPassedFormula(pbEncoding); // translate/solve this in backend!
204  SMTRAT_LOG_DEBUG("smtrat.maxsmt.fumalik", "Adding cardinality constraint to solver: " << pbEncoding);
205  }
206 
207  return gatherSatisfiedSoftClauses();
208  }

216  template<class Settings>
217  std::vector<FormulaT> MaxSMTModule<Settings>::runLinearSearch()
218  {
219  // add all soft clauses with relaxation var
220  // for i = 0; i < soft.size; i++:
221  // check sat for \sum relaxation var <= i to determine if we have to disable some constraint
222  // if sat return;
223 
224  std::vector<carl::Variable> relaxationVars;
225  for (const auto& clause : softclauses) {
226  carl::Variable relaxationVar = carl::freshBooleanVariable();
227  mBackend.add(FormulaT(carl::FormulaType::OR, clause, FormulaT(relaxationVar)));
228 
229  relaxationVars.push_back(relaxationVar);
230  }
231 
232  Poly relaxationPoly;
233  for (carl::Variable& var : relaxationVars) {
234  relaxationPoly = relaxationPoly + var;
235  }
236 
237  // initially all constraints must be enabled
238  ConstraintT initialRelaxationConstraint(relaxationPoly - Rational(0),carl::Relation::LEQ);
239  ModuleInput::iterator previousRelaxationConstraint = addToLocalBackend(FormulaT(initialRelaxationConstraint));
240 
241  for (unsigned i = 1; i < softclauses.size(); i++) {
242  // check sat for max i disables clauses
243  SMTRAT_LOG_DEBUG("smtrat.maxsmt.linear", "Trying to check SAT for " << i - 1 << " disabled soft constraints...");
244 
245  Answer ans = mBackend.check();
246  if (ans == Answer::SAT) return gatherSatisfiedSoftClauses();
247 
248  mBackend.remove(previousRelaxationConstraint);
249 
250  ConstraintT relaxationConstraint(relaxationPoly - Rational(i),carl::Relation::LEQ);
251  previousRelaxationConstraint = addToLocalBackend(FormulaT(relaxationConstraint));
252  }
253 
254  // from here, none of the soft clauses can be satisfied
255  return gatherSatisfiedSoftClauses();
256  }

258  template<class Settings>
259  std::vector<FormulaT> MaxSMTModule<Settings>::runMSU3()
260  {
261 
262  // initialise data structures
263  for (const auto& clause : softclauses) {
264  // add clause backend and remember where we put it so we can easily remove it later
265  ModuleInput::iterator it = addToLocalBackend(clause);
266  formulaPositionMap[clause] = it;
267  }
268 
269  std::vector<carl::Variable> relaxationVars;
270  std::vector<FormulaT> relaxedConstraints;
271 
272  for (unsigned i = 0; i < softclauses.size(); i++) {
273  // rebuild polynomial since we add relaxation vars incrementally
274  Poly relaxationPoly;
275  for (carl::Variable& var : relaxationVars) {
276  relaxationPoly = relaxationPoly + var;
277  }
278 
279  ConstraintT relaxationConstraint(relaxationPoly - Rational(i),carl::Relation::LEQ);
280  SMTRAT_LOG_DEBUG("smtrat.maxsmt.msu3", FormulaT(relaxationConstraint));
281  ModuleInput::iterator relaxationFormula = addToLocalBackend(FormulaT(relaxationConstraint));
282 
283  SMTRAT_LOG_DEBUG("smtrat.maxsmt.msu3", "Checking SAT for up to " << i << " disabled constraints.");
284  Answer ans = mBackend.check();
285  if (ans == Answer::SAT) return gatherSatisfiedSoftClauses();
286 
287  // We extract directly from the backend because newly added formulas get deleted from the infeasible subset!
288  FormulasT unsatisfiableCore = getUnsatCore();
289 
290  SMTRAT_LOG_DEBUG("smtrat.maxsmt.msu3", "Got unsat core " << unsatisfiableCore);
291 
292  for (const auto& f : unsatisfiableCore) { // for all soft constraints in unsat core...
293  if (!isSoft(f)) continue;
294 
295  // check whether we already relaxed f
296  bool alreadyRelaxed = std::find(relaxedConstraints.begin(), relaxedConstraints.end(), f) != relaxedConstraints.end();
297  if (alreadyRelaxed) continue;
298 
299  carl::Variable relaxationVar = carl::freshBooleanVariable();
300  mBackend.remove(formulaPositionMap[f]); // first erase non-relaxed clause
301  addToLocalBackend(FormulaT(carl::FormulaType::OR, f, FormulaT(relaxationVar))); // ...then add relaxed clause
302 
303  relaxedConstraints.push_back(f);
304  relaxationVars.push_back(relaxationVar);
305  }
306 
307  mBackend.remove(relaxationFormula); // remove cardinality constraint again
308 
309  }
310 
311  // from here, none of the soft clauses can be satisfied
312  return gatherSatisfiedSoftClauses();
313  }
