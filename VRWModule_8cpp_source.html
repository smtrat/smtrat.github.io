86  Answer VRWModule::isConsistent()
87  {
88  std::list<std::pair<FormulaT::iterator, bool> > notNecessary;
89  do
90  {
91  notNecessary = mMatchingGraph.findIrrelevantConstraints(mpPassedFormula->end());
92  for(std::list<std::pair<FormulaT::iterator, bool> >::const_iterator it = notNecessary.begin(); it != notNecessary.end(); ++it)
93  {
94  removeSubformulaFromPassedFormula( it->first, it->second, !it->second);
95  assert(it->first != mpPassedFormula->end());
96  }
97  } while(!notNecessary.empty());
98 
99  Answer ans = Unknown;
100 
101  if(mpPassedFormula->size() > 0) {
102  mMatchingGraph.assertConstraints();

104  }
105  else

107  ans = True;
108  }
109 
110  if( ans == False )

112  // print();
113  getInfeasibleSubsets();
114  }
