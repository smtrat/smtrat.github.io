188  @Override
189  public void mouseReleased( MouseEvent me )
190  {
191  if( checkModifiers( me ) )
192  {
193  Point p = me.getPoint();
194  try
195  {
196  graph = (StrategyGraph) visualization.getGraphLayout().getGraph();
197  if( startEdge!=null )
198  {
199  if( (pickedEdge = graph.getEdge( p ))!=null && startEdge!=pickedEdge )
200  {
201  graph.pushPriority( startEdge, pickedEdge );
202  visualization.fireStateChanged();
203  visualization.repaint();
204  }
205  }
206  else
207  {
208  Layout<Vertex,Edge> layout = visualization.getModel().getGraphLayout();
209  GraphElementAccessor<Vertex,Edge> pickSupport = visualization.getPickSupport();
210  if( pickSupport!=null )
211  {
212  pickedVertex = pickSupport.getVertex( layout, p.getX(), p.getY() );
213  if( pickedVertex!=null && startVertex!=null && pickedVertex!=startVertex && pickedVertex!=graph.getRoot() )
214  {
215  if( !graph.getSuccessors( startVertex ).contains( pickedVertex ) )
216  {
217  pickedEdge = new Edge( new Condition(), true );
218  if( graph.addEdge( pickedEdge, startVertex, pickedVertex, EdgeType.DIRECTED ) )
219  {
220  EditStrategyGraphDialog.showAddEdgeDialog( gui, pickedEdge );
221  }
222  }
223  else
224  {
225  throw new StrategyGraph.UnsupportedStrategyGraphOperationException( StrategyGraph.UnsupportedStrategyGraphOperationException.PARALLEL_EDGES_ARE_NOT_ALLOWED );
226  }
227  }
228  else if( pickedVertex!=null && pickedVertex!=startVertex && pickedVertex==graph.getRoot() )
229  {
230  throw new StrategyGraph.UnsupportedStrategyGraphOperationException( StrategyGraph.UnsupportedStrategyGraphOperationException.EDGES_CANNOT_POINT_TO_START_VERTEX );
231  }
232  }
233  }
234  }
235  catch( StrategyGraph.UnsupportedStrategyGraphOperationException ex )
236  {
237  JOptionPane.showMessageDialog( gui, ex.getMessage(), "Information", JOptionPane.INFORMATION_MESSAGE );
238  }
239  visualization.setCursor( Cursor.getPredefinedCursor( Cursor.DEFAULT_CURSOR ) );
240  startEdge = null;
241  startVertex = null;
242  mouseDown = null;
243  visualization.removePostRenderPaintable( edgePaintable );
244  visualization.removePostRenderPaintable( arrowPaintable );
245  visualization.repaint();
246  }

316  private void transformArrowShape( Point2D mouseDown, Point2D mouseOut )
317  {
318  float x1 = (float) mouseDown.getX();
319  float y1 = (float) mouseDown.getY();
320  float x2 = (float) mouseOut.getX();
321  float y2 = (float) mouseOut.getY();
322 
323  AffineTransform xform = AffineTransform.getTranslateInstance( x2, y2 );
324 
325  float dx = x2-x1;
326  float dy = y2-y1;
327  float thetaRadians = (float) Math.atan2( dy, dx );
328  xform.rotate( thetaRadians );
329  arrowShape = xform.createTransformedShape( rawArrowShape );
330  }

336  private void transformEdgeShape( Point2D mouseDown, Point2D mouseOut )
337  {
338  float x1 = (float) mouseDown.getX();
339  float y1 = (float) mouseDown.getY();
340  float x2 = (float) mouseOut.getX();
341  float y2 = (float) mouseOut.getY();
342 
343  AffineTransform xform = AffineTransform.getTranslateInstance( x1, y1 );
344 
345  float dx = x2-x1;
346  float dy = y2-y1;
347  float thetaRadians = (float) Math.atan2( dy, dx );
348  xform.rotate( thetaRadians );
349  float dist = (float) Math.sqrt( dx*dx+dy*dy );
350  xform.scale( dist/rawEdgeShape.getBounds().getWidth(), 1.0 );
351  edgeShape = xform.createTransformedShape( rawEdgeShape );
352  }
