100 void RedAns_sendReduce(RedProc process,const char *command) {
101  char ch;
102  int ii;
103 
104  RedMsg_cfprintf(debugfile,
105  "-> RedAns_sendReduce: entering, command=%s\n",
106  command);
107 
108  if (command == (char *)NULL) {
109  ch = EOT;
110  write(process->meToReduce[1],&ch,1);
111  ch = LF;
112  write(process->meToReduce[1],&ch,1);
113  } else {
114  for (ii=0; command[ii] != 0; ii++) {
115  ch = command[ii] & 0x7f;
116  RedMsg_cfprintf(process->rlgFile,"%c",ch);
117  write(process->meToReduce[1],&ch,1);
118 #ifdef PSLFIX
119  if ((ii+1) % SYSMAXBUFFER == 0) {
120  RedAns_delete(RedAns_readUntilPrompt(process));

122  "-- RedAns_sendReduce: skipped prompt (SYSMAXBUFFER=%d)\n",
123  SYSMAXBUFFER);
124  }
125 #endif
126  }
127  ch = LF;
128  RedMsg_cfprintf(process->rlgFile,"%c",ch);
129  write(process->meToReduce[1],&ch,1);

178 void RedAns_dfa(RedProc process,StrBuf *pstatcbuf,int *psymbolic,
179  StrBuf *ppretext,StrBuf *presult,StrBuf *pposttext,
180  StrBuf *pnextprompt) {
181  unsigned char buffer[READBUFSIZE];
182  int ncharread,ii;
183  int status=PRETEXT,ch;
184 
185  RedMsg_cfprintf(debugfile,"-> RedAns_dfa: entering\n");
186  RedMsg_cfprintf(debugfile,"-- RedAns_dfa: status=PRETEXT\n");
187 
188  while(status != FINISHED) {
189  ncharread = read(process->reduceToMe[0],buffer,READBUFSIZE);
190  for (ii=0; ii < ncharread; ii++) {
191  ch = buffer[ii];
192  if (ISCOLOR(ch)) {
193  status = ch;
194  RedAns_dfaLog(process,status);
195  } else if (status == RESULT) {
196  RedMsg_cfprintf(process->rlgFile,"%c",ch);
197  // if (ch >= 0x20) /* in particular skip LF */
198  if (ch >= 0x07) /* skip prompt colorings */
199  *presult = StrBuf_addChar(*presult,ch);
200  } else if (FASTMODE) {
201  continue;
202  } else if (status == STATCOUNTER) {
203  RedMsg_cfprintf(process->rlgFile,"%c",ch);
204  *pstatcbuf = StrBuf_addChar(*pstatcbuf,ch);
205  } else if (status == PRETEXT) {
206  RedMsg_cfprintf(process->rlgFile,"%c",ch);
207  *ppretext = StrBuf_addChar(*ppretext,ch);
208  } else if (status == MODE) {
209  char chstr[2];
210  RedMsg_cfprintf(process->rlgFile,"%c",ch);
211  chstr[0] = ch;
212  chstr[1] = 0;
213  sscanf(chstr,"%d",psymbolic);
214  } else if (status == POSTTEXT) {
215  RedMsg_cfprintf(process->rlgFile,"%c",ch);
216  if (ch >= 0x20)
217  *pposttext = StrBuf_addChar(*pposttext,ch);
218  } else if (status == PROMPT) {
219  RedMsg_cfprintf(process->rlgFile,"%c",ch);
220  *pnextprompt = StrBuf_addChar(*pnextprompt,ch);
221  }
222  }
223  }
224  RedMsg_cfprintf(debugfile,"<- RedAns_dfa: leaving\n");
225 }
