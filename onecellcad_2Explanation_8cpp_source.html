15 inline carl::RealAlgebraicPoint<smtrat::Rational> asRANPoint(
16  const mcsat::Bookkeeping& data) {
17  std::vector<carl::RealAlgebraicNumber<smtrat::Rational>> point;
18  for (const auto variable : data.assignedVariables()) {
19  const auto& modelValue = data.model().evaluated(variable);
20  assert(modelValue.isRational() || modelValue.isRAN());
21  modelValue.isRational() ? point.emplace_back(modelValue.asRational()) : point.emplace_back(modelValue.asRAN());
22  }
23  return carl::RealAlgebraicPoint<smtrat::Rational>(std::move(point));

50 boost::optional<mcsat::Explanation>
51 Explanation::operator()(const mcsat::Bookkeeping& trail, // current assignment state
52  carl::Variable var,
53  const FormulasT& trailLiterals) const {
54  assert(trail.model().size() == trail.assignedVariables().size());
55 
56 #ifdef SMTRAT_DEVOPTION_Statistics
57  mStatistics.explanationCalled();
58 #endif
59 
60  #if not (defined USE_COCOA || defined USE_GINAC)
61  // OneCellCAD needs carl::irreducibleFactors to be implemented
62  #warning OneCellCAD may be incorrect as USE_COCOA is disabled
63  #endif
64 
65  // compute compatible complete variable ordering
66  std::vector varOrder(trail.assignedVariables());
67  for (const auto& v : trail.variables()) {
68  if (std::find(trail.assignedVariables().begin(), trail.assignedVariables().end(), v) == trail.assignedVariables().end()) {
69  varOrder.push_back(v);
70  }
71  }
72 
73  // Temp. workaround, should at least one theory-var should be assigned, otherwise no OneCell construct possible
74  // TODO remove as soon, mcsat backend handles this case.
75  if (trail.assignedVariables().size() == 0) {
76  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", " OneCellExplanation called with 0 theory-assignment");
77  FormulasT explainLiterals;
78  for (const auto& trailLiteral : trailLiterals)
79  explainLiterals.emplace_back(trailLiteral.negated());
80  return boost::variant<FormulaT, ClauseChain>(FormulaT(carl::FormulaType::OR, std::move(explainLiterals)));
81  }
82  assert(trail.assignedVariables().size() > 0);
83 
84  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", "Starting an explanation");
85  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", trail);
86  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", "Number of assigned vars: " << trail.model().size());
87  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", "Trail literals: " << trailLiterals); //<< " Implied literal: " << impliedLiteral);
88  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", "Ascending variable order: " << varOrder
89  << " and eliminate down from: " << var);
90 
91  std::vector<Poly> polys; // extract from trailLiterals
92  for (const auto& constraint : nlsat::helper::convertToConstraints(trailLiterals))
93  polys.emplace_back(constraint.lhs()); // constraints have the form 'poly < 0' with <, = etc.
94 
95  const auto& trailVariables = trail.assignedVariables();
96  std::vector<carl::Variable> fullProjectionVarOrder = varOrder;
97  std::vector<carl::Variable> oneCellCADVarOrder;
98  for (std::size_t i = 0; i < trailVariables.size(); i++)
99  oneCellCADVarOrder.emplace_back(fullProjectionVarOrder[i]);
100 
101  // std::vector<carl::Variable> fullProjectionVarOrder(trailVariables.size());
102  // std::vector<carl::Variable> oneCellCADVarOrder;
103  // fullProjectionVarOrder.reserve(varOrder.size());
104  // oneCellCADVarOrder.reserve(trailVariables.size());
105  //
106  // for (auto variable : varOrder) {
107  // if (std::find(trailVariables.begin(), trailVariables.end(), variable) == trailVariables.end())
108  // fullProjectionVarOrder.push_back(variable);
109  // else
110  // oneCellCADVarOrder.push_back(variable);
111  // }
112  // std::copy_n(oneCellCADVarOrder.begin(), oneCellCADVarOrder.size(), fullProjectionVarOrder.begin());
113 
114  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", "FullProjVarOrder: " << fullProjectionVarOrder);
115  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", "OneCellVarOrder: " << oneCellCADVarOrder);
116  std::size_t oneCellMaxLvl = trail.model().size() - 1; // -1, b/c we count first var = poly level 0
117  std::vector<std::vector<onecellcad::TagPoly>> projectionLevels(fullProjectionVarOrder.size()); // init all levels with empty vector
118  onecellcad::categorizeByLevel(
119  projectionLevels,
120  fullProjectionVarOrder,
121  onecellcad::nonConstIrreducibleFactors(toTagPoly(polys)));
122 
123  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", "Polys at levels before full CAD projection:\n"
124  << projectionLevels);
125  // Project higher level polys down to "normal" level
126  auto maxLevel = fullProjectionVarOrder.size() - 1;
127  for (std::size_t currentLvl = maxLevel; currentLvl > oneCellMaxLvl; currentLvl--) {
128  auto currentVar = fullProjectionVarOrder[currentLvl];
129  assert(currentLvl >= 1);
130  auto nextLowerVar = fullProjectionVarOrder[currentLvl - 1];
131  auto projectionFactors = onecellcad::singleLevelFullProjection(currentVar, nextLowerVar, projectionLevels[currentLvl]);
132  onecellcad::categorizeByLevel(
133  projectionLevels,
134  fullProjectionVarOrder,
135  onecellcad::nonConstIrreducibleFactors(projectionFactors));
136  projectionLevels[currentLvl].clear();
137  SMTRAT_LOG_TRACE("smtrat.mcsat.nlsat", "Polys at levels after a CAD projection at level: " << currentLvl << ":\n"
138  << projectionLevels);
139  }
140  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", "Polys at levels after full CAD projection:\n"
141  << projectionLevels);
142 
143  std::vector<onecellcad::TagPoly2> oneCellPolys;
144  // SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", "OneCellMaxLvl: " << oneCellMaxLvl << " ProjectionLvls: " << projectionLevels);
145  for (std::size_t currentLvl = 0; currentLvl <= oneCellMaxLvl; currentLvl++) {
146  for (const auto& poly : projectionLevels[currentLvl])
147  oneCellPolys.emplace_back(onecellcad::TagPoly2{poly.tag, poly.poly, currentLvl});
148  }
149  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", "All polys for OneCell construction: " << oneCellPolys);
150 
151  auto cellOpt = onecellcad::OneCellCAD(oneCellCADVarOrder, asRANPoint(trail).prefixPoint(oneCellMaxLvl + 1)).pointEnclosingCADCell(oneCellPolys);
152  if (!cellOpt) {
153  SMTRAT_LOG_WARN("smtrat.mcsat.nlsat", "OneCell construction failed");
154  return boost::none;
155  }
156 
157  auto cell = *cellOpt;
158  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", "Constructed cell: " << cell);
159 
160  // If we have trail literals: L_M, ... , L_M,
161  // an implied literal L
162  // and cell boundary atoms : A, .., A
163  // We construct the formula '(A & ... & A & L_M & ... & L_M) => L'
164  // in its clausal form 'E := (-A v ... v -A v -L_M v ... v -L_M v L)'
165  // as our explanation.
166  FormulasT explainLiterals;
167 
168  for (const auto& trailLiteral : trailLiterals)
169  explainLiterals.emplace_back(trailLiteral.negated());
170 
171  for (std::size_t i = 0; i < cell.size(); i++) {
172  auto& cellComponent = cell[i];
173  auto cellVariable = oneCellCADVarOrder[i];
174  if (std::holds_alternative<onecellcad::Section>(cellComponent)) {
175  auto section = std::get<onecellcad::Section>(cellComponent).boundFunction;
176  // Need to use poly with its main variable replaced by the special MultivariateRootT::var().
177  auto param = std::make_pair(section.poly(), section.k());
178  explainLiterals.emplace_back(nlsat::helper::buildEquality(cellVariable, param).negated());
179  } else {
180  auto sectorLowBound = std::get<onecellcad::Sector>(cellComponent).lowBound;
181  if (sectorLowBound) {
182  // Need to use poly with its main variable replaced by the special MultivariateRootT::var().
183  auto param = std::make_pair(sectorLowBound->boundFunction.poly(), sectorLowBound->boundFunction.k());
184  explainLiterals.emplace_back(nlsat::helper::buildAbove(cellVariable, param).negated());
185  }
186  auto sectorHighBound = std::get<onecellcad::Sector>(cellComponent).highBound;
187  if (sectorHighBound) {
188  // Need to use poly with its main variable replaced by the special MultivariateRootT::var().
189  auto param = std::make_pair(sectorHighBound->boundFunction.poly(), sectorHighBound->boundFunction.k());
190  explainLiterals.emplace_back(nlsat::helper::buildBelow(cellVariable, param).negated());
191  }
192  }
193  }
194 
195  // if (!impliedLiteral.isFalse())
196  // explainLiterals.emplace_back(impliedLiteral);
197 
198  SMTRAT_LOG_DEBUG("smtrat.mcsat.nlsat", "Explain literals: " << explainLiterals);
199 #ifdef SMTRAT_DEVOPTION_Statistics
200  mStatistics.explanationSuccess();
201 #endif
202  return boost::variant<FormulaT, ClauseChain>(FormulaT(carl::FormulaType::OR, std::move(explainLiterals)));
203 }
