25  std::size_t complexity(const ConstraintT& c) const {
26  return c.maxDegree() * c.variables().size() * c.lhs().size();

28  bool operator()(const ConstraintT& lhs, const ConstraintT& rhs) const {
29  auto cl = complexity(lhs);
30  auto cr = complexity(rhs);
31  if (cl != cr) return cl < cr;
32  return lhs < rhs;

92  std::size_t add(const ConstraintT& c) {
93  SMTRAT_LOG_DEBUG("smtrat.cad.constraints", "Adding " << c << " to " << std::endl << *this);
94  bool isBound = mBounds.addBound(c, c);
95  assert(!mVariables.empty());
96  std::size_t id = 0;
97  if (BT == Backtracking::ORDERED) {
98  id = mConstraintIts.size();
99  mConstraintIts.push_back(mConstraintMap.end());
100  mConstraintLevels.emplace_back(0);
101 
102  mActiveConstraintMap.emplace(c, id);
103  auto r = mConstraintMap.emplace(c, id);
104  assert(r.second);
105  mConstraintIts[id] = r.first;
106  } else if (BT == Backtracking::HIDE) {
107  auto it = mConstraintMap.find(c);
108  if (it != mConstraintMap.end()) {
109  id = it->second;
110  mActiveConstraintMap.emplace(c, id);
111  mConstraintIts[id] = it;
112  } else {
113  id = mIDPool.get();
114  if (id >= mConstraintIts.size()) {
115  mConstraintIts.resize(id+1, mConstraintMap.end());
116  mConstraintLevels.resize(id+1);
117  }
118  mActiveConstraintMap.emplace(c, id);
119  auto r = mConstraintMap.emplace(c, id);
120  assert(r.second);
121  mConstraintIts[id] = r.first;
122  }
123  } else {
124  id = mIDPool.get();
125  if (id >= mConstraintIts.size()) {
126  mConstraintIts.resize(id+1, mConstraintMap.end());
127  mConstraintLevels.resize(id+1);
128  }
129  mActiveConstraintMap.emplace(c, id);
130  auto r = mConstraintMap.emplace(c, id);
131  assert(r.second);
132  mConstraintIts[id] = r.first;
133  }

156  carl::Bitset remove(const ConstraintT& c) {
157  SMTRAT_LOG_DEBUG("smtrat.cad.constraints", "Removing " << c << " from " << std::endl << *this);
158  bool isBound = mBounds.removeBound(c, c);
159  auto it = mConstraintMap.find(c);
160  assert(it != mConstraintMap.end());
161  std::size_t id = it->second;
162  carl::Bitset res = {id};
163  mSatByBounds.reset(id);
164  mUnsatByBounds.reset(id);
165  assert(mConstraintIts[id] == it);
166  if (BT == Backtracking::ORDERED) {
167  SMTRAT_LOG_TRACE("smtrat.cad.constraints", "Removing " << id << " in ordered mode");
168  std::stack<ConstraintT> cache;
169  // Remove constraints added after c
170  while (mConstraintIts.back()->second > id) {
171  SMTRAT_LOG_TRACE("smtrat.cad.constraints", "Preliminary removal of " << mConstraintIts.back()->first);
172  cache.push(mConstraintIts.back()->first);
173  res |= remove(mConstraintIts.back()->first);
174  }
175  // Remove c
176  SMTRAT_LOG_TRACE("smtrat.cad.constraints", "Actual removal of " << mConstraintIts.back()->first);
177  callCallback(mRemoveCallback, mConstraintIts.back()->first, mConstraintIts.back()->second, isBound);
178  mActiveConstraintMap.erase(mConstraintIts.back()->first);
179  mConstraintMap.erase(mConstraintIts.back());
180  mConstraintIts.pop_back();
181  assert(mConstraintIts.size() == id);
182  // Add constraints removed before
183  while (!cache.empty()) {
184  SMTRAT_LOG_TRACE("smtrat.cad.constraints", "Readding of " << cache.top());
185  add(cache.top());
186  cache.pop();
187  }
188  } else if(BT == Backtracking::HIDE) {
189  SMTRAT_LOG_TRACE("smtrat.cad.constraints", "Removing " << id << " in unordered mode");
190  callCallback(mRemoveCallback, c, id, isBound);
191  mActiveConstraintMap.erase(it->first);
192  mConstraintIts[id] = mConstraintMap.end();
193  } else {
194  SMTRAT_LOG_TRACE("smtrat.cad.constraints", "Removing " << id << " in unordered mode");
195  callCallback(mRemoveCallback, c, id, isBound);
196  mActiveConstraintMap.erase(it->first);
197  mConstraintMap.erase(it);
198  mConstraintIts[id] = mConstraintMap.end();
199  mIDPool.free(id);
200  }
201  SMTRAT_LOG_DEBUG("smtrat.cad.constraints", "Removed " << c << " from " << std::endl << *this);
202  return res;
203  }

212  bool checkForTrivialConflict(std::vector<FormulaSetT>& mis) const {

222  for (const auto& c: mConstraintIts) {

225  switch (c->first.consistentWith(intervalmap)) {
226  case 0: {

228  mis.emplace_back();
229  for (const auto& b: bounds().getOriginsOfBounds()) {
230  mis.back().emplace(b);
231  }
232  mis.back().emplace(c->first);
233  return true;
234  }

236  }
