57  inline
58  std::set<ConstraintT> convertToConstraints(std::vector<FormulaT> constraintAtoms) {
59  std::set<ConstraintT> cons;
60  for (const auto& cAtom: constraintAtoms) {
61  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Adding " << cAtom << " to " << cons);
62  if (cAtom.getType() == carl::FormulaType::CONSTRAINT) {
63  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Adding " << cAtom);
64  cons.emplace(cAtom.constraint());
65  } else if (cAtom.getType() == carl::FormulaType::VARCOMPARE) {
66  // Note that we only add the polynomials here and don't really care about the relation
67  // var ~ rootexpr(poly)
68  // -> poly to ensure that the root exists
69  // -> var - poly to ensure that the relation still holds
70  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Adding bound " << cAtom << " -> " << cAtom.variableComparison().definingPolynomial());
71  cons.emplace(cAtom.variableComparison().definingPolynomial(), cAtom.variableComparison().relation());
72  cons.emplace(Poly(cAtom.variableComparison().var()) - cAtom.variableComparison().definingPolynomial(), cAtom.variableComparison().relation());
73  } else if (cAtom.getType() == carl::FormulaType::VARASSIGN) {
74  SMTRAT_LOG_WARN("smtrat.nlsat", "Variable assignment " << cAtom << " should never get here!");
75  assert(false);
76  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Adding assignment " << cAtom);
77  const VariableComparisonT& vc = cAtom.variableAssignment();
78  cons.emplace(vc.definingPolynomial(), carl::Relation::EQ);
79  } else {
80  SMTRAT_LOG_ERROR("smtrat.nlsat", "Unsupported formula type: " << cAtom);
81  assert(false);
82  }
83  SMTRAT_LOG_DEBUG("smtrat.nlsat", "-> " << cons);
84  }
85  return cons;
86  }

110  void generateBoundsFor(FormulasT& boundsAsAtoms, carl::Variable var, std::size_t level, const Model& model) const {
111  auto val = mModel.evaluated(var);
112  assert(val.isRational() || val.isRAN());
113  RAN value = val.isRational() ? RAN(val.asRational()) : val.asRAN();
114  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Generating bounds for " << var << " = " << value);
115  boost::optional<std::pair<RAN,FormulaT>> lower;
116  boost::optional<std::pair<RAN,FormulaT>> upper;
117 
118  for (std::size_t pid = 0; pid < mProjection.size(level); pid++) {
119  const auto& poly = mProjection.getPolynomialById(level, pid);
120  if (carl::isZero(carl::model::substitute(poly, model))) continue;
121  auto list = carl::model::realRoots(poly, model);
122  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Looking at " << poly << " with roots " << list);
123  // Find the closest roots/rootIdx around value.
124  std::size_t rootID = 1;
125  for (const auto& root: list) {
126  // Need to use poly with its main variable replaced by the special MultivariateRootT::var().
127  auto param = std::make_pair(Poly(carl::UnivariatePolynomial<Poly>(MultivariateRootT::var(), poly.coefficients())), rootID);
128  SMTRAT_LOG_TRACE("smtrat.nlsat", root << " -> " << param);
129  if (root < value) {
130  if (!lower || (root > lower->first)) {
131  lower = std::make_pair(root, helper::buildAbove(var, param));
132  SMTRAT_LOG_TRACE("smtrat.nlsat", "new lower bound: " << lower->second);
133  }
134  } else if (root == value) {
135  lower = std::make_pair(root, helper::buildEquality(var, param));
136  upper = *lower;
137  SMTRAT_LOG_TRACE("smtrat.nlsat", "new exact root: " << lower->second);
138  } else {
139  if (!upper || (root < upper->first)) {
140  upper = std::make_pair(root, helper::buildBelow(var, param));
141  SMTRAT_LOG_TRACE("smtrat.nlsat", "new upper bound: " << upper->second);
142  }
143  }
144  rootID++;
145  }
146  }
147  if (lower) {
148  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Lower bound:" << lower->second);
149  boundsAsAtoms.push_back(lower->second);
150  }
151  if (upper && lower != upper) {
152  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Upper bound:" << upper->second);
153  boundsAsAtoms.push_back(upper->second);
154  }
155  }

194  void generateExplanation(const FormulaT& impliedAtom, std::vector<FormulasT>& explainAtomsinLvls) const {
195  // Model with assignments for variables we already have constructed bounds for
196  Model submodel; // Start empty
197  explainAtomsinLvls.resize(mCADConstraints.vars().size());
198 
199  // Start from the bottom to generate bound constraints.
200  for (int level = mCADConstraints.vars().size() - 1; level >= 0; level--) {
201  carl::Variable varAtLvl = mCADConstraints.vars()[level];
202  if (mModel.find(varAtLvl) == mModel.end()) continue;
203  generateBoundsFor(explainAtomsinLvls[level], varAtLvl, level+1, submodel);
204  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Cell bounds for " << varAtLvl << ": " << explainAtomsinLvls[level]);
205  submodel.emplace(varAtLvl, mModel.evaluated(varAtLvl));
206  }
207 
208  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Collecting constraints from " << mConstraints);
209  for (const auto& constraintAtom: mConstraints) {
210  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Considering " << constraintAtom);
211  if (constraintAtom == impliedAtom.negated()) {
212  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Skipping " << constraintAtom);
213  continue;
214  }
215  explainAtomsinLvls.back().emplace_back(constraintAtom);
216  }
217  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Final: " << explainAtomsinLvls.back() << " -> " << impliedAtom);
218  }

223  mcsat::Explanation getExplanation(const FormulaT& impliedAtom) const {
224  std::vector<FormulasT> explainAtomsInLevels;
225  generateExplanation(impliedAtom, explainAtomsInLevels);
226  // Convert e1,..,en => I into clause -e1,...,-en, I
227  FormulasT explainClauseLiterals;
228  for (const auto& explainAtoms: explainAtomsInLevels) {
229  for (const auto& explainAtom: explainAtoms) {
230  explainClauseLiterals.emplace_back(explainAtom.negated());
231  }
232  }
233  if (!impliedAtom.isTrue()) explainClauseLiterals.emplace_back(impliedAtom);
234  return FormulaT(carl::FormulaType::OR, std::move(explainClauseLiterals));
235  }
