110  template<typename ConstraintIt>
111  bool evaluateSample(Sample& sample, const ConstraintIt& constraint, Assignment& assignment) const {
112  std::size_t cid = constraint.second;
113  SMTRAT_LOG_TRACE("smtrat.cad.lifting", "Evaluating " << cid << " / " << constraint.first.lhs() << " " << constraint.first.relation() << " 0 on " << assignment);
114  SMTRAT_LOG_TRACE("smtrat.cad.lifting", "Sample was evaluated: " << sample.evaluatedWith() << " / " << sample.evaluationResult());
115  if (sample.evaluatedWith().test(cid)) {
116  return sample.evaluationResult().test(cid);
117  }
118  auto res = carl::RealAlgebraicNumberEvaluation::evaluate(constraint.first.lhs(), assignment);
119  bool evalResult = carl::evaluate(res, constraint.first.relation());
120  SMTRAT_LOG_TRACE("smtrat.cad.lifting", "Evaluating " << constraint.first.lhs() << " " << constraint.first.relation() << " 0 on " << assignment << " -> " << evalResult);
121  sample.evaluatedWith().set(cid, true);
122  sample.evaluationResult().set(cid, evalResult);
123  return evalResult;
124  }

126  std::vector<Assignment> enumerateSolutions() {

129  while (mLifting.hasFullSamples()) {

132  SMTRAT_LOG_TRACE("smtrat.cad", "Checking full sample " << m);
133  assert(m.size() == it.depth());
134  bool sat = true;
135  for (const auto& c: mConstraints.ordered()) {
136  Assignment a = m;

142  res.emplace_back(m);

144  }

183  auto a = mLifting.extractSampleMap(it);

210  ConflictGraph generateConflictGraph() const {
211  ConflictGraph cg(mConstraints.size());
212  for (auto it = mLifting.getTree().begin_preorder(); it != mLifting.getTree().end_preorder();) {
213  if (it->hasConflictWithConstraint()) {
214  // Skip subtrees of already conflicting samples
215  SMTRAT_LOG_TRACE("smtrat.cad.mis", "Adding sample " << *it);
216  cg.addSample(*it);
217  it.skipChildren();
218  } else {
219  it++;
220  }
221  for (std::size_t id = 0; id < mConstraints.size(); id++) {
222  if (!mConstraints.valid(id)) {
223  SMTRAT_LOG_TRACE("smtrat.cad.mis", "Invalid constraint: " << id << ", cur graph:" << std::endl << cg);
224  }
225  assert(mConstraints.valid(id) || cg.coveredSamples(id) == 0);
226  }
227  }
228  SMTRAT_LOG_DEBUG("smtrat.cad.mis", "Resulting conflict graph " << cg);
