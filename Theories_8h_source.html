237  types::TermType handleITE(const std::vector<types::TermType>& arguments) {
238  if (settings_parser().disable_theory) return FormulaT();
239  types::TermType result;
240  if (arguments.size() != 3) {
241  SMTRAT_LOG_ERROR("smtrat.parser", "Failed to construct ITE expression, only exactly three arguments are allowed, but \"" << arguments << "\" were given.");
242  HANDLE_ERROR
243  return result;
244  }
245  FormulaT ifterm;
246  conversion::VariantConverter<FormulaT> converter;
247  if (!converter(arguments[0], ifterm)) {
248  SMTRAT_LOG_ERROR("smtrat.parser", "Failed to construct ITE expression, the first argument must be a formula, but \"" << arguments[0] << "\" was given.");
249  HANDLE_ERROR
250  return result;
251  }
252  if (ifterm.isTrue()) return arguments[1];
253  if (ifterm.isFalse()) return arguments[2];
254  TheoryError te;
255  for (auto& t: theories) {
256  if (t.second->handleITE(ifterm, arguments[1], arguments[2], result, te(t.first))) return result;
257  }
258  SMTRAT_LOG_ERROR("smtrat.parser", "Failed to construct ITE \"" << ifterm << "\" ? \"" << arguments[1] << "\" : \"" << arguments[2] << "\": " << te);
259  HANDLE_ERROR
260  return result;
261  }

299  bool instantiateUserFunction(const UserFunctionInstantiator& function, const std::vector<types::TermType>& arguments, types::TermType& result, TheoryError& errors) {
300  if (settings_parser().disable_theory) {
301  result = FormulaT();
302  return true;
303  }
304  if (function.arguments.size() != arguments.size()) {
305  errors.next() << "Function was expected to have " << function.arguments.size() << " arguments, but was instantiated with " << arguments.size() << ".";
306  return false;
307  }
308  result = function.definition;
309  for (const auto& aux: function.auxiliaries) {
310  TheoryError te;
311  bool wasRefreshed = false;
312  types::VariableType repl;
313  for (auto& t: theories) {
314  if (t.second->refreshVariable(aux, repl, te(t.first))) {
315  wasRefreshed = true;
316  break;
317  }
318  }
319  if (!wasRefreshed) {
320  SMTRAT_LOG_ERROR("smtrat.parser", "Failed to refresh auxiliary variable \"" << aux << "\": " << te);
321  HANDLE_ERROR
322  return false;
323  }
324  if (!instantiate(function, aux, repl, result)) return false;
325  }
326  for (std::size_t i = 0; i < arguments.size(); i++) {
327  if (!instantiate(function, function.arguments[i], arguments[i], result)) return false;
328  }
329  return true;
330  }

332  types::TermType functionCall(const Identifier& identifier, const std::vector<types::TermType>& arguments) {
333  if (settings_parser().disable_theory) return FormulaT();
334  types::TermType result;
335  TheoryError te;
336  if (identifier.symbol == "ite") {
337  if (identifier.indices != nullptr) {
338  SMTRAT_LOG_ERROR("smtrat.parser", "The function \"" << identifier << "\" should not have indices.");
339  HANDLE_ERROR
340  return result;
341  }
342  return handleITE(arguments);
343  } else if (identifier.symbol == "distinct") {
344  if (identifier.indices != nullptr) {
345  SMTRAT_LOG_ERROR("smtrat.parser", "The function \"" << identifier << "\" should not have indices.");
346  HANDLE_ERROR
347  return result;
348  }
349  return handleDistinct(arguments);
350  }
351  auto deffunit = state->defined_functions.find(identifier.symbol);
352  if (deffunit != state->defined_functions.end()) {
353  if (identifier.indices != nullptr) {
354  SMTRAT_LOG_ERROR("smtrat.parser", "The function \"" << identifier << "\" should not have indices.");
355  HANDLE_ERROR
356  return result;
357  }
358  SMTRAT_LOG_DEBUG("smtrat.parser", "Trying to call function \"" << identifier << "\" with arguments " << arguments << ".");
359  if ((*deffunit->second)(arguments, result, te(identifier.symbol))) {
360  SMTRAT_LOG_DEBUG("smtrat.parser", "Success, result is \"" << result << "\".");
361  return result;
362  }
363  SMTRAT_LOG_ERROR("smtrat.parser", "Failed to call function \"" << identifier << "\" with arguments " << arguments << ":" << te);
364  HANDLE_ERROR
365  return result;
366  }
367  auto ideffunit = state->defined_indexed_functions.find(identifier.symbol);
368  if (ideffunit != state->defined_indexed_functions.end()) {
369  if (identifier.indices == nullptr) {
370  SMTRAT_LOG_ERROR("smtrat.parser", "The function \"" << identifier << "\" should have indices.");
371  HANDLE_ERROR
372  return result;
373  }
374  SMTRAT_LOG_DEBUG("smtrat.parser", "Trying to call function \"" << identifier << "\" with arguments " << arguments << ".");
375  if ((*ideffunit->second)(*identifier.indices, arguments, result, te(identifier.symbol))) {
376  SMTRAT_LOG_DEBUG("smtrat.parser", "Success, result is \"" << result << "\".");
377  return result;
378  }
379  SMTRAT_LOG_ERROR("smtrat.parser", "Failed to call function \"" << identifier << "\" with arguments " << arguments << ":" << te);
380  HANDLE_ERROR
381  return result;
382  }
383  auto udeffunit = state->defined_user_functions.find(identifier.symbol);
384  if (udeffunit != state->defined_user_functions.end()) {
385  if (identifier.indices != nullptr) {
386  SMTRAT_LOG_ERROR("smtrat.parser", "The function \"" << identifier << "\" should not have indices.");
387  HANDLE_ERROR
388  return result;
389  }
390  SMTRAT_LOG_DEBUG("smtrat.parser", "Trying to call function \"" << identifier << "\" with arguments " << arguments << ".");
391  if (instantiateUserFunction(*udeffunit->second, arguments, result, te(identifier.symbol))) {
392  SMTRAT_LOG_DEBUG("smtrat.parser", "Success, result is \"" << result << "\".");
393  return result;
394  }
395  SMTRAT_LOG_ERROR("smtrat.parser", "Failed to call function \"" << identifier << "\" with arguments " << arguments << ":" << te);
396  HANDLE_ERROR
397  return result;
398  }
399  for (auto& t: theories) {
400  if (t.second->functionCall(identifier, arguments, result, te(t.first))) return result;
401  }
402  SMTRAT_LOG_ERROR("smtrat.parser", "Failed to call \"" << identifier << "\" with arguments " << arguments << ":" << te);
403  HANDLE_ERROR
404  return result;
405  }
