53  carl::Bitset addToProjection(std::size_t level, const UPoly& p, const Origin::BaseType& origin) {
54  assert(level > 0 && level <= dim());
55  if (cad::projection::canBeRemoved(p)) return carl::Bitset();
56  SMTRAT_LOG_DEBUG("smtrat.cad.projection", "Adding " << p << " to projection level " << level);
57  assert(p.mainVar() == var(level));
58  auto it = polyIDs(level).find(p);
59  if (it != polyIDs(level).end()) {
60  assert(polys(level)[it->second]);
61  polys(level)[it->second]->second += origin;
62  return carl::Bitset();
63  }
64  std::size_t newID = getID(level);;
65  carl::Bitset res;
66  if (level < dim()) {
67  mOperator(Settings::projectionOperator, p, var(level + 1),
68  [&](const UPoly& np){ res |= addToProjection(level + 1, np, Origin::BaseType(level, newID)); }
69  );
70  for (const auto& it: polyIDs(level)) {
71  assert(polys(level)[it.second]);
72  auto newOrigin = Origin::BaseType(level, newID, it.second);
73  mOperator(Settings::projectionOperator, p, it.first, var(level + 1),
74  [&](const UPoly& np){ res |= addToProjection(level + 1, np, newOrigin); }
75  );
76  }
77  }
78  if (newID >= polys(level).size()) {
79  polys(level).resize(newID + 1);
80  }
81  polys(level)[newID] = std::make_pair(p, Origin(origin));
82  polyIDs(level).emplace(p, newID);
83  mLiftingQueues[level - 1].insert(newID);
84  res.set(level);
85  return res;
86  }

103  void removePolynomial(const UPoly&, std::size_t cid, bool) override {
104  SMTRAT_LOG_DEBUG("smtrat.cad.projection", "Removing " << cid);
105  carl::Bitset filter = carl::Bitset().set(cid);
106  for (std::size_t level = 1; level <= dim(); level++) {
107  for (std::size_t lvl = level; lvl <= dim(); lvl++) {
108  for (auto it = polyIDs(level).begin(); it != polyIDs(level).end(); it++) {
109  assert(polys(level)[it->second]);
110  polys(level)[it->second]->second.erase(level, filter);
111  }
112  }
113  carl::Bitset removed;
114  for (auto it = polyIDs(level).begin(); it != polyIDs(level).end();) {
115  std::size_t id = it->second;
116  assert(polys(level)[id]);
117  if (polys(level)[id]->second.empty()) {
118  freeID(level, id);
119  mLiftingQueues[level - 1].erase(id);
120  removed.set(id);
121  polys(level)[id] = boost::none;
122  it = polyIDs(level).erase(it);
123  } else {
124  it++;
125  }
126  }
127  SMTRAT_LOG_TRACE("smtrat.cad.projection", "Calling callback for level " << level << ", removed [" << removed << "]");
128  callRemoveCallback(level, removed);
129  filter = removed;
130  }
131  }
