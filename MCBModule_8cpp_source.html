124  template<typename Settings>
125  void MCBModule<Settings>::collectChoices(const FormulaT& formula) {
126  if (formula.getType() != carl::FormulaType::OR) return;
127 
128  FormulaT::ConstraintBounds cb;
129  collectBounds(cb, formula, false);
130  if (cb.empty()) return;
131 
132  for (const auto& poly: cb) {
133  if (!poly.first.isVariable()) continue;
134  AVar var = poly.first.getSingleVariable();
135  std::vector<std::pair<Rational,FormulaT>> choices;
136  for (const auto& entry: poly.second) {
137  if (entry.second.first != carl::Relation::EQ) break;
138  choices.emplace_back(entry.first, entry.second.second);
139  }
140  if (choices.size() != poly.second.size()) continue;
141  auto& m = mChoices[var];
142  for (const auto& c: choices) {
143  auto it = m.find(c.first);
144  if (it == m.end()) {
145  m.emplace(c.first, std::make_pair(carl::freshBooleanVariable(), c.second));
146  }
147  }
148  }

151  template<typename Settings>
152  FormulaT MCBModule<Settings>::applyReplacements(const FormulaT& f) {
153  if (mChoices.empty()) return f;
154  std::set<AVar> variables;
155  std::map<FormulaT, FormulaT> repl;
156  for (const auto& r: mChoices) {
157  variables.insert(r.first);
158  for (const auto& f: r.second) {
159  BVar v = f.second.first;
160  const FormulaT& form = f.second.second;
161  repl.emplace(form, FormulaT(v));
162  }
163  }
164  carl::FormulaSubstitutor<FormulaT> subs;
165  SMTRAT_LOG_DEBUG("smtrat.mcb", "Applying " << repl << " on \n\t" << f);
166  FormulaT res = subs.substitute(f, repl);
167  SMTRAT_LOG_DEBUG("smtrat.mcb", "Resulting in\n\t" << res);
168 
169  mRemaining.clear();
170  res.allVars(mRemaining);
171  FormulasT equiv;
172  for (const auto& v: variables) {
173  if (mRemaining.count(v) > 0) {
174  // Variable is still in the formula
175  for (const auto& r: mChoices.at(v)) {
176  equiv.push_back(FormulaT(carl::FormulaType::IFF, {FormulaT(r.second.first), r.second.second}));
177  }
178  } else {
179  // Variable has been eliminated
180  ModelVariable var(v);
181  std::map<BVar,Rational> assignment;
182  for (const auto& c: mChoices.at(v)) {
183  assignment.emplace(c.second.first, c.first);
184  }
185  SMTRAT_LOG_DEBUG("smtrat.mcb", "Adding " << var << " = " << assignment);
186  mAssignments.emplace(var, carl::createSubstitution<Rational,Poly,MCBModelSubstitution>(assignment));
187  }
188  for (const auto& c1: mChoices.at(v)) {
189  for (const auto& c2: mChoices.at(v)) {
190  if (c1.second.first >= c2.second.first) continue;
191  equiv.push_back(FormulaT(carl::FormulaType::OR, {FormulaT(c1.second.first).negated(), FormulaT(c2.second.first).negated()}));
192  SMTRAT_LOG_DEBUG("smtrat.mcb", "Adding exclusion " << equiv.back());
193  }
194  }
195  }
196  if (equiv.empty()) return res;
197  SMTRAT_LOG_DEBUG("smtrat.mcb", "Adding equivalences " << equiv);
198  equiv.push_back(res);
199  return FormulaT(carl::FormulaType::AND, std::move(equiv));
200  }
