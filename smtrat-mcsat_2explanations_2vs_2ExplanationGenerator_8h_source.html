34  std::pair<std::vector<carl::Variable>::const_iterator, std::vector<carl::Variable>::const_iterator> getUnassignedVariables() const {
35  std::unordered_set<carl::Variable> freeVariables;
36  for (const auto& constr : mConstraints) {
37  freeVariables.insert(constr.variables().begin(), constr.variables().end());
38  }
39 
40  auto firstVar = std::find(mVariableOrdering.begin(), mVariableOrdering.end(), mTargetVar);
41  assert(firstVar != mVariableOrdering.end());
42  auto lastVar = firstVar;
43  for (auto iter = firstVar; iter != mVariableOrdering.end(); iter++) {
44  if (freeVariables.find(*iter) != freeVariables.end()) {
45  lastVar = iter;
46  }
47  }
48 
49  lastVar++;
50 
51  return std::make_pair(firstVar, lastVar);
52  }

54  boost::optional<FormulaT> eliminateVariable(const FormulaT& inputFormula, const carl::Variable& var) const {
55  // get formula atoms
56  FormulaSetT atoms;
57  helper::getFormulaAtoms(inputFormula, atoms);
58 
59  // generate test candidates
60  std::vector<smtrat::vs::Substitution> testCandidates;
61  if (helper::generateTestCandidates(testCandidates, var, mModel, atoms)) {
62  FormulasT res;
63  res.reserve(testCandidates.size());
64  for (const auto& tc : testCandidates) {
65  FormulaT branchResult = inputFormula;
66 
67  // substitute tc into each input constraint
68  for (const auto& constr : mConstraints) {
69  // check if branchResult still contains constr
70  if (branchResult.contains(constr)) {
71  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Substituting " << tc << " into " << constr);
72 
73  // calculate substitution
74  FormulaT substitutionResult; // TODO reduceConflictConstraints?
75  if (!helper::substitute(constr, tc, mModel, substitutionResult)) {
76  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Substitution failed");
77  return boost::none;
78  }
79 
80  // check if current constraint is part of the conflict
81  if (Settings::reduceConflictConstraints) {
82  carl::ModelValue<Rational,Poly> eval = carl::model::evaluate(substitutionResult, mModel);
83  // If constraint is not fully evaluated or evaluates to false, we take it in.
84  if (!eval.isBool() || !eval.asBool()) {
85  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Use constraint " << constr << " for explanation");
86  }
87  else {
88  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Ignore constraint " << constr << " because it is not part of the conflict");
89  substitutionResult = FormulaT(carl::FormulaType::TRUE);
90  }
91  }
92 
93  // substitute into formula
94  carl::FormulaSubstitutor<FormulaT> substitutor;
95  branchResult = substitutor.substitute(branchResult, constr, substitutionResult);
96  }
97  }
98 
99  // add side condition
100  FormulasT branch;
101  branch.push_back(std::move(branchResult));
102  for (const auto& sc : tc.sideCondition()) {
103  branch.emplace_back(sc);
104  }
105 
106  res.emplace_back(carl::FormulaType::AND, std::move(branch));
107  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Substitution of " << tc << " into formula obtained " << res.back());
108  assert(res.back() != FormulaT(carl::FormulaType::TRUE));
109  }
110 
111  return FormulaT(carl::FormulaType::OR, std::move(res));
112  }
113  else {
114  SMTRAT_LOG_DEBUG("smtrat.mcsat.vs", "Could not generate test candidates");
115  return boost::none;
116  }
117  }

119  boost::optional<FormulaT> eliminateVariables() const {
120  // eliminate unassigned variables
121  boost::optional<FormulaT> res = FormulaT(carl::FormulaType::AND, mConstraints);
122  auto unassignedVariables = getUnassignedVariables();
123  for (auto iter = unassignedVariables.first; iter != unassignedVariables.second; iter++) {
124  res = eliminateVariable(*res, *iter);
125  if (!res) {
126  return boost::none;
127  }
128  }
129 
130  #ifndef NDEBUG
131  carl::ModelValue<Rational,Poly> evalRes = carl::model::evaluate(*res, mModel);
132  assert(evalRes.isBool());
133  assert(!evalRes.asBool());
134  #endif
135 
136  // collect input constraints
137  FormulasT expl;
138  expl.push_back(std::move(*res));
139  for (const auto& c: mConstraints) {
140  expl.emplace_back(c.negated());
141  }
142  return FormulaT(carl::FormulaType::OR, expl);
143  }

150  FormulaT _transformToImplicationChain(const FormulaT& formula, ClauseChain& chain, bool withEquivalences) const /*__attribute__((noinline))*/ {
151  switch(formula.getType()) {
152  case carl::FormulaType::TRUE:
153  case carl::FormulaType::FALSE:
154  case carl::FormulaType::CONSTRAINT:
155  case carl::FormulaType::VARCOMPARE:
156  case carl::FormulaType::VARASSIGN:
157  {
158  return formula;
159  }
160  break;
161 
162  case carl::FormulaType::OR:
163  {
164  FormulasT newFormula;
165  for (const auto& sub : formula.subformulas()) {
166  FormulaT tseitinSub = _transformToImplicationChain(sub, chain, withEquivalences);
167  newFormula.push_back(std::move(tseitinSub));
168  }
169  return FormulaT(carl::FormulaType::OR, std::move(newFormula));
170  }
171  break;
172 
173  case carl::FormulaType::AND:
174  {
175  FormulaT tseitinVar = chain.createTseitinVar(formula);
176  FormulasT newFormula;
177  for (const auto& sub : formula.subformulas()) {
178  FormulaT tseitinSub = _transformToImplicationChain(sub, chain, withEquivalences);
179  newFormula.push_back(std::move(tseitinSub));
180  const auto& lit = newFormula.back();
181 
182  // tseitinVar -> newFormula_1 && ... && newFormula_n
183 
184  carl::ModelValue<Rational,Poly> eval = carl::model::evaluate(sub, mModel);
185  assert(eval.isBool());
186  if (!eval.asBool()) {
187  chain.appendPropagating(FormulaT(carl::FormulaType::OR, FormulasT({lit, tseitinVar.negated()})), tseitinVar.negated());
188  } else {
189  chain.appendOptional(FormulaT(carl::FormulaType::OR, FormulasT({lit, tseitinVar.negated()})));
190  }
191  }
192 
193  if (withEquivalences) {
194  // newFormula_1 && ... && newFormula_n -> tseitinVar
195  FormulasT tmp;
196  std::transform (newFormula.begin(), newFormula.end(), std::back_inserter(tmp), [](const FormulaT& f) -> FormulaT { return f.negated(); } );
197  tmp.push_back(tseitinVar);
198  chain.appendOptional(FormulaT(carl::FormulaType::OR, tmp));
199  }
200 
201  return tseitinVar;
202  }
203 
204  default:
205  SMTRAT_LOG_WARN("smtrat.mcsat.vs", "Invalid formula type " << formula);
206  assert(false);
207  return FormulaT();
208  }
209  }
