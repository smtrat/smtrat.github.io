42  mSamples.emplace_back(RAN::sampleAbove(mRoots.back()));

44  }

138  bool hasAssignment(carl::Variable v) {
139  SMTRAT_LOG_DEBUG("smtrat.nlsat", "Assignment for " << v);
140  RootIndexer ri;
141  std::map<ConstraintT, std::pair<std::list<RAN>, ConstraintT>> rootMap;
142  for (const auto& c: mConstraints) {
143  if (!isUnivariate(c, v)) continue;
144  ConstraintT cnew(carl::model::substitute(c.lhs(), mModel), c.relation());
145  auto list = carl::model::realRoots(cnew.lhs(), v, mModel);
146  ri.add(list);
147  rootMap.emplace(c, std::make_pair(std::move(list), cnew));
148  }
149  ri.process();
150  Covering cover(ri.size() * 2 + 1);
151  Model m = mModel;
152  for (const auto& c: rootMap) {
153  carl::Bitset b;
154  const auto& roots = c.second.first;
155  const auto& constraint = c.second.second;
156  std::size_t last = 0;
157  for (const auto& r: roots) {
158  std::size_t cur = ri[r];
159  SMTRAT_LOG_TRACE("smtrat.nlsat", constraint << " vs " << ri.sampleFrom(2*cur));
160  if (!satisfies(constraint, m, v, ri.sampleFrom(2*cur))) {
161  // Refutes interval left of this root
162  SMTRAT_LOG_TRACE("smtrat.nlsat", constraint << " refutes " << ri.sampleFrom(2*cur) << " -> " << last << ".." << (2*cur));
163  setBitsForInterval(b, last, 2*cur);
164  }
165  SMTRAT_LOG_TRACE("smtrat.nlsat", constraint << " vs " << ri.sampleFrom(2*cur+1));
166  if (!satisfies(constraint, m, v, r)) {
167  // Refutes root
168  SMTRAT_LOG_TRACE("smtrat.nlsat", constraint << " refutes " << r << " -> " << 2*cur+1);
169  setBitsForInterval(b, 2*cur+1, 2*cur+1);
170  }
171  last = 2*cur + 2;
172  }
173  SMTRAT_LOG_TRACE("smtrat.nlsat", constraint << " vs " << ri.sampleFrom(last));
174  if (!satisfies(constraint, m, v, ri.sampleFrom(last))) {
175  // Refutes interval right of largest root
176  SMTRAT_LOG_TRACE("smtrat.nlsat", constraint << " refutes " << ri.sampleFrom(roots.size()*2) << " -> " << last << ".." << (ri.size()*2));
177  setBitsForInterval(b, last, ri.size()*2);
178  }
179  cover.add(c.first, b);
180  }
181  SMTRAT_LOG_TRACE("smtrat.nlsat", cover);
182  if (cover.conflicts()) {
183  mConflictingCore = std::vector<ConstraintT>();
184  cover.buildConflictingCore(*mConflictingCore);
185  return false;
186  } else {
187  mAssignment = ri.sampleFrom(cover.satisfyingInterval());
188  return true;
189  }
190  }
