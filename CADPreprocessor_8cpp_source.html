140  auto sres = simplify(constraints);
141  if (std::holds_alternative<ConstraintT>(sres)) {
142  return sres;
143  }
144  assert(std::holds_alternative<bool>(sres));
145  bool foundNew = std::get<bool>(sres);

150  auto sres = simplify(constraints);

152  return sres;

156  foundNew = foundNew || continueSearch || std::get<bool>(sres);

176 bool ResultantRule::addPoly(const UPoly& poly, std::size_t level, const std::vector<FormulaT>& origin) {
177  if (poly.isNumber()) return true;
178  SMTRAT_LOG_TRACE("smtrat.cad.pp", "Adding poly " << poly << " under ordering " << mVars);
179  Poly mp(poly);
180  UPoly p = poly;
181  assert(p.mainVar() == mVars[level]);
182  while (carl::is_constant(p)) {
183  ++level;
184  p = carl::to_univariate_polynomial(mp, mVars[level]);
185  }
186  SMTRAT_LOG_TRACE("smtrat.cad.pp", "Inserting " << p << " into level " << level);
187  SMTRAT_LOG_TRACE("smtrat.cad.pp", "Into " << mData);
188  if (std::find(mData[level].begin(), mData[level].end(), p) != mData[level].end()) {
189  return true;
190  }
191  mData[level].emplace_back(p);
192  ConstraintT cons(mp, carl::Relation::EQ);
193  mOrigins.add(FormulaT(mp, carl::Relation::EQ), origin);

200 std::optional<std::vector<FormulaT>> ResultantRule::computeResultants(std::size_t level) {
201  for (std::size_t pid = 0; pid < mData[level].size(); ++pid) {
202  for (std::size_t qid = pid + 1; qid < mData[level].size(); ++qid) {
203  auto r = projection::resultant(mVars[level + 1], mData[level][pid], mData[level][qid]);
204  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Resultant(" << mData[level][pid] << ", " << mData[level][qid] << ") = " << r);
205  std::vector<FormulaT> origin;
206  const auto& op = mOrigins.get(FormulaT(Poly(mData[level][pid]), carl::Relation::EQ));
207  origin.insert(origin.end(), op.begin(), op.end());
208  const auto& oq = mOrigins.get(FormulaT(Poly(mData[level][qid]), carl::Relation::EQ));
209  origin.insert(origin.end(), oq.begin(), oq.end());
210  if (!addPoly(r, level + 1, origin)) {
211  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Found direct conflict due to " << origin);
212  return origin;
213  }
214  }
215  }
216  return std::nullopt;
217 }

291  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Adding necessary parts of model to conflict: " << conflict);

338 bool CADPreprocessor::preprocess() {
339  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Starting with:" << std::endl << *this);
340  bool changed = addEqualities(mEqualities);
341  while (changed) {
342  changed = false;
343  if (settings_cadpp().disable_variable_elimination) {
344  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Variable elimination is disabled");
345  } else {
346  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Collecting assignments from:" << std::endl << *this);
347  auto collectResult = mAssignments.collect(mDerivedEqualities);
348  if (std::holds_alternative<ConstraintT>(collectResult)) {
349  mConflict = { FormulaT(std::get<ConstraintT>(collectResult)) };
350  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Immediate conflict due to " << *mConflict);
351  return false;
352  }
353  assert(std::holds_alternative<bool>(collectResult));
354  if (std::get<bool>(collectResult) == true) {
355  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Collected assignments:" << std::endl << *this);
356  for (const auto& de: mDerivedEqualities) {
357  if (mOrigins.mOrigins.find(FormulaT(de.second)) == mOrigins.mOrigins.end()) {
358  mOrigins.add(FormulaT(de.second), {FormulaT(de.first)});
359  }
360  }
361  changed = true;
362  }
363  }
364  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Origins are now:" << std::endl << mOrigins.mOrigins);
365 
366  if (settings_cadpp().disable_resultants) {
367  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Resultant rule is disabled");
368  } else {
369  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Computing Resultants from:" << std::endl << *this);
370  auto conflict = mResultants.compute(mDerivedEqualities);
371  if (conflict.has_value()) {
372  mConflict = std::set<FormulaT>(conflict->begin(), conflict->end());
373  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Immediate conflict due to " << *mConflict);
374  return false;
375  } else {
376  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Computed resultants:" << std::endl << *this << std::endl << "-> " << mResultants.getNewECs());
377  if (addEqualities(mResultants.getNewECs())) {
378  changed = true;
379  }
380  }
381  }
382  }
383  for (auto& c: mInequalities) {
384  carl::model::substituteIn(c.second, mModel);
385  if (c.second.isConsistent() == 0) {
386  mConflict = { FormulaT(c.first) };
387  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Immediate conflict due to " << *mConflict);
388  return false;
389  }
390  }
391  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "After preprocessing:" << std::endl << *this);
392  return true;
393 }

403 void CADPreprocessor::postprocessConflict(std::set<FormulaT>& mis) const {
404  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Postprocessing conflict: " << mis << " based on" << std::endl << *this);
405  if (collectOriginsOfConflict(mis, mInequalities)) {
406  SMTRAT_LOG_DEBUG("smtrat.cad.pp", "Resolved inequalities: " << mis);
