138  void generateExplanation(Callback&& callback) {
139  std::vector<Bound> mLower;
140  std::vector<Bound> mUpper;
141  std::multimap<Rational, Bound> mInequalities;
142  std::vector<Bound> mEqualities;
143  std::vector<std::pair<Bound, Bound>> mBoundPair;
144 
145  // initialize bounds
146  for (const auto& b: mBounds) {
147  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Processing bound " << b);
148 
149  if (!b.hasVariable(mVariable)) {
150  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Discarding bound " << b << " because it does not contain " << mVariable);
151  continue;
152  }
153 
154  if (b.varInfo(mVariable).maxDegree() > 1) {
155  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Discarding bound " << b << " because " << mVariable << " occurs nonlinearly");
156  continue;
157  }
158  auto p = b.coefficient(mVariable, 1);
159  if (carl::isZero(p)) {
160  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Discarding bound " << b << " because it does not contain " << mVariable);
161  continue;
162  }
163 
164  auto evalp = carl::model::evaluate(p, mModel);
165  if (!evalp.isRational()) {
166  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Discarding bound " << b << " because " << p << " did not evaluate to a rational");
167  continue;
168  }
169  assert(evalp.isRational());
170 
171  auto q = p * mVariable - b.lhs();
172  auto evalq = carl::model::evaluate(q, mModel);
173  if (!evalq.isRational()) {
174  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Discarding bound " << b << " because " << q << " did not evaluate to a rational");
175  continue;
176  }
177  assert(evalq.isRational());
178 
179  if (carl::isZero(evalp.asRational())) {
180  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Discarding bound " << b << " because it does not contain " << mVariable << " after we evaluate it");
181 
182  if (!carl::evaluate(Rational(0), b.relation(), evalq.asRational())) {
183  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Bound " << b << " unsat because p is zero and q does not comply");
184  FormulasT expl;
185  expl.emplace_back(b.negation());
186  expl.emplace_back(ConstraintT(p, carl::Relation::EQ).negation());
187  expl.emplace_back(ConstraintT(-q, b.relation()));
188 
189  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Explanation: " << expl[0].negated() << " && " << expl[1].negated() << " -> " << expl[2]);
190  mcsat_yield(callback, expl);
191  }
192 
193  continue;
194  }
195  bool negated = evalp.asRational() < 0;
196 
197  Rational r = evalq.asRational() / evalp.asRational();
198 
199  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Considering bound " << b << " for " << mVariable);
200  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Decomposed into " << p << " * " << mVariable << " ~ " << q << ", " << mVariable << " ~ " << r);
201  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Coefficient is " << evalp.asRational());
202 
203  switch (b.relation()) {
204  case carl::Relation::LESS:
205  case carl::Relation::LEQ:
206  if (negated) {
207  mLower.emplace_back(b, p, q, r, negated);
208  } else {
209  mUpper.emplace_back(b, p, q, r, negated);
210  }
211  break;
212  case carl::Relation::EQ:
213  mLower.emplace_back(b, p, q, r, negated);
214  mUpper.emplace_back(b, p, q, r, negated);
215  mEqualities.emplace_back(b, p, q, r, negated);
216  break;
217  case carl::Relation::NEQ:
218  mInequalities.emplace(r, Bound(b, p, q, r, negated));
219  break;
220  case carl::Relation::GEQ:
221  case carl::Relation::GREATER:
222  if (negated) {
223  mUpper.emplace_back(b, p, q, r, negated);
224  } else {
225  mLower.emplace_back(b, p, q, r, negated);
226  }
227  break;
228  }
229  }
230  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Lower: " << mLower);
231  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Upper: " << mUpper);
232  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Inequ: " << mInequalities);
233 
234  // look for FM constraints
235  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Looking for conflicts between lower and upper bounds");
236 
237  std::sort(mLower.begin(), mLower.end(), comparator);
238  if (comparator.symmetric) {
239  std::sort(mUpper.rbegin(), mUpper.rend(), comparator);
240  }
241  else {
242  std::sort(mUpper.begin(), mUpper.end(), comparator);
243  }
244 
245  for (const Bound& lower : mLower) {
246  for (const Bound& upper : mUpper) {
247  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Combining " << lower << " and " << upper);
248  bool strict = carl::isStrict(lower.constr.relation()) || carl::isStrict(upper.constr.relation());
249 
250  if (lower.r < upper.r) {
251  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Not in conflict");
252  continue;
253  }
254 
255  if (lower.r == upper.r && !strict) {
256  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Not in conflict");
257  mBoundPair.push_back(std::make_pair(lower, upper));
258  continue;
259  }
260 
261  mcsat_yield(callback, conflictLowerAndUpperBound(lower, upper));
262  }
263  }
264 
265  // handle inequalities
266  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Looking for conflicts with inequalities");
267 
268  for (const auto& eq : mEqualities) {
269  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Considering equality " << eq.constr);
270  auto it = mInequalities.find(eq.r);
271  if (it != mInequalities.end()) {
272  mcsat_yield(callback, conflictEqualityAndInequality(eq, it->second));
273  }
274  }
275 
276  for (const auto& bounds : mBoundPair) {
277  SMTRAT_LOG_DEBUG("smtrat.mcsat.fm", "Considering lower bound " << bounds.first << " and upper bound " << bounds.second);
278  auto it = mInequalities.find(bounds.first.r);
279  if (it != mInequalities.end()) {
280  mcsat_yield(callback, conflictInequalitiesAndInequality(bounds.first, bounds.second, it->second));
281  }
282  }
283  }
