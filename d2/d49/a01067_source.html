50  switch (f.getType()) {
51  case carl::FormulaType::CONSTRAINT:
52  mBounds.addBound(f.constraint(), f);
53  if (!f.constraint().isBound()) {
54  mConstraints.emplace(f, f.constraint());
55  }
56  break;

98  template<class Settings>
99  Answer ICEModule<Settings>::processConstraints() {
100  if (mBounds.isConflicting()) {
101  mInfeasibleSubsets.emplace_back(mBounds.getConflict());
102  return Answer::UNSAT;
103  }
104  FHG graph;
105  TermT src;
106  std::vector<TermT> dest;
107  Coefficient coeff;
108  for (const auto& c: mConstraints) {
109  if (isSuitable(c.second, src, dest, coeff)) {
110  auto& edge = graph.newEdge(graph.newVertex(src), EdgeProperty(coeff, c.first));
111  for (const auto& d: dest) {
112  edge.addOut(graph.newVertex(d));
113  }
114  }
115  }
116  if (Settings::dumpAsDot) {
117  SMTRAT_LOG_INFO("smtrat.ice", "Dumping resulting graph to " << Settings::dotFilename);
118  graph.toDot(Settings::dotFilename);
119  }
120 
121  CycleCollector collector;
122  CycleEnumerator<FHG,CycleCollector> enumerator(graph, collector);
123  enumerator.findAll();
124 
125  if (!collector.mInfeasibleSubset.empty()) {
126  SMTRAT_LOG_INFO("smtrat.ice", "Found input to be unsat, subset is " << collector.mInfeasibleSubset);
127  mInfeasibleSubsets.emplace_back(collector.mInfeasibleSubset);
128  return UNSAT;
129  }
130  for (const auto& lemma: collector.mLemmas) {
131  SMTRAT_LOG_DEBUG("smtrat.ice", "Adding " << lemma.first << " with origin " << lemma.second);
132  addSubformulaToPassedFormula(lemma.first, lemma.second);
133  }
134  return SAT;
135  }

137  template<class Settings>
138  bool ICEModule<Settings>::isSuitable(const ConstraintT& c, TermT& src, std::vector<TermT>& dest, Coefficient& coeff) {
139  SMTRAT_LOG_FUNC("smtrat.ice", c);
140  bool invert = false;
141  src = TermT();
142  dest.clear();
143  coeff.strict = false;
144  coeff.r = 0;
145 
146  switch (c.relation()) {
147  case carl::Relation::EQ: break;
148  case carl::Relation::NEQ: return false;
149  case carl::Relation::LESS:
150  invert = true;
151  coeff.strict = true;
152  break;
153  case carl::Relation::LEQ:
154  invert = true;
155  break;
156  case carl::Relation::GREATER:
157  coeff.strict = true;
158  break;
159  case carl::Relation::GEQ: break;
160  }
161  Poly p = c.lhs();
162  if (invert) p = -p;
163 
164  for (const auto& term: p) {
165  if (term.isConstant()) {
166  coeff.r += term.coeff();
167  continue;
168  }
169  if (isZero(term)) {
170  SMTRAT_LOG_WARN("smtrat.ice", "Term " << term << " is zero. We'll ignore it.");
171  continue;
172  }
173  if (isSemiPositive(term)) {
174  if (!carl::isZero(src)) {
175  SMTRAT_LOG_TRACE("smtrat.ice", "No: Already has a source, but " << term << " was found.");
176  return false;
177  }
178  src = term;
179  } else if (isSemiNegative(term)) {
180  dest.push_back(-term);
181  } else {
182  SMTRAT_LOG_TRACE("smtrat.ice", "No: Has indefinite term " << term << ".");
183  return false;
184  }
185  }
186  if (dest.empty()) {
187  SMTRAT_LOG_TRACE("smtrat.ice", "No: No destinations were found.");
188  return false;
189  }
190  SMTRAT_LOG_TRACE("smtrat.ice", "Yes: " << src << " -> " << dest << " with coefficient " << coeff << ".");
191  return true;
192  }
193 }
