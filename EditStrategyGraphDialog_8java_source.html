531  len = condition.getLength( index );
532  condition.removeElement( index );
533  }
534  }

537  int start, end;

539  {
540  start = c.getDot();
541  end = c.getMark();
542  }
543  else
544  {
545  start = c.getMark();
546  end = c.getDot();
547  }
548 
549  int startIndex = condition.calculateIndexPosition( start );

554  }

616  }
617  else

621  double maxX = -1;

627  if( minX == -1 || minX > destPosX )

629  if( maxX == -1 || maxX < destPosX )
630  maxX = destPosX;
631  sumOfY += destPosY;
632  }
633  posY = sumOfY/(double) outEdges.size();
634  double vertexPosX = IOTools.layout.getX( vertex );
635  double xdist = maxX - minX;
636  if( outEdges.size() == 1 )
637  {
638  xdist = 2*GUI.VISUALIZATION_PADDING;
639  }
640  else
641  {
642  xdist = xdist/(outEdges.size()-1);
643  }
644  if( Math.abs( vertexPosX - minX ) < Math.abs( vertexPosX - maxX ) )
645  {
646  posX = minX - xdist;
647  if( posX < GUI.VISUALIZATION_PADDING )
648  {
649  posX = GUI.VISUALIZATION_PADDING;
650  }
651  }
652  else
653  {
654  posX = maxX + xdist;
655  if( posX > GUI.VISUALIZATION_WIDTH - GUI.VISUALIZATION_PADDING )
656  {
657  posX = GUI.VISUALIZATION_WIDTH - GUI.VISUALIZATION_PADDING;
658  }
659  }
