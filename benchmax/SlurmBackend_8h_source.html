42  void parse_result_file(const Jobs& jobs, const std::filesystem::path& file, std::vector<JobData>& results) {
43  BENCHMAX_LOG_DEBUG("benchmax.slurm", "Processing file " << file);
44  std::ifstream in(file);
45  std::string content((std::istreambuf_iterator<char>(in)), std::istreambuf_iterator<char>());
46  auto extension = file.extension();
47  std::regex filere("Executing (.+)\\n# START ([0-9]+) #([^#]*)# END \\2 #(?:([^#]*)# END DATA \\2 #)?");
48 
49  auto reBegin = std::sregex_iterator(content.begin(), content.end(), filere);
50  auto reEnd = std::sregex_iterator();
51  for (auto i = reBegin; i != reEnd; ++i) {
52  std::size_t id = std::stoull((*i)[2]) - 1;
53  bool toolFound = false;
54  std::string cmdline = (*i)[1];
55  for (const auto& tool : jobs.tools()) {
56  auto t = tool->parseCommandline(cmdline);
57  if (t) {
58  results.emplace_back(JobData { tool.get(), std::filesystem::path(*t), BenchmarkResult() });
59  toolFound = true;
60  break;
61  }
62  }
63  if (!toolFound) {
64  BENCHMAX_LOG_WARN("benchmax.slurm", "Could not find tool for " << cmdline);
65  } else {
66  auto& res = std::get<2>(results.back());
67  if (extension == ".out") {
68  res.stdout = (*i)[3];
69  res.exitCode = std::stoi(slurm::parse_result_info((*i)[4], "exitcode"));
70  res.time = std::chrono::milliseconds(std::stoi(slurm::parse_result_info((*i)[4], "time")));
71  BENCHMAX_LOG_DEBUG("benchmax.slurm", "Got " << res << " for task " << id << " from stdout");
72  } else if (extension == ".err") {
73  res.stderr = (*i)[3];
74  BENCHMAX_LOG_DEBUG("benchmax.slurm", "Got " << res << " for task " << id << " from stderr");
75  } else {
76  BENCHMAX_LOG_WARN("benchmax.slurm", "Trying to parse output file with unexpected extension " << extension);
77  }
78  }
79  }
80  }

199  void run(const Jobs& jobs, bool wait_for_termination) {
200  if (load_job_ids().size() > 0) {
201  BENCHMAX_LOG_ERROR("benchmax.slurm", "Benchmax is still running in the specified tmp_dir! If this is not the case, please delete " + settings_slurm().tmp_dir + "/slurmjobs");
202  return;
203  }
204 
205  std::vector<JobData> results;
206  for (const auto& [tool, file]: jobs.randomized()) {
207  results.emplace_back(JobData { tool, file, BenchmarkResult() });
208  }
209  BENCHMAX_LOG_DEBUG("benchmax.slurm", "Gathered " << results.size() << " jobs");
210 
211  std::vector<std::future<void>> tasks;
212  std::size_t count = results.size() / (settings_slurm().array_size * settings_slurm().slice_size) + 1;
213  for (std::size_t i = 0; i < count; ++i) {
214  tasks.emplace_back(std::async(std::launch::async,
215  [i,&results,wait_for_termination,this](){
216  return run_job_async(i, results, wait_for_termination);
217  }
218  ));
219  }
220  for (auto& f: tasks) {
221  f.wait();
222  }
223  if (wait_for_termination) {
224  BENCHMAX_LOG_DEBUG("benchmax.slurm", "All jobs terminated.");
225  } else {
226  BENCHMAX_LOG_DEBUG("benchmax.slurm", "All jobs scheduled.");
227  }
228  }
229 };
