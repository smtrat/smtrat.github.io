59 
60  // Text printed by REDUCE before the evaluation result. This should

62  char *pretext;
63 
64  // The evaluation result as printed by the procedures lr_sprint(u)
65  // or lr_aprint(u) (for symbolic mode or algebraic mode, resp.) in
66  // libreduce.red. They are defined as "prin2 u" and "prin2 reval u",
67  // resp., where the idea is that the application could redefine
68  // them:
69  char *result;
70 
71  // Text printed by REDUCE after the evaluation result but before the
72  // next prompt. I expect this to be exactly the output generated by
73  // "on time", which is switched on in libreduce.red.
74  char *posttext;
75 
76  // The prompt for the next computation:
77  char *nextprompt;
78 
79  // The computation time and gc time for this RedAns in ms as an
80  // integer. This is why libreduce.red says "on time."
81  int time;
82  int gctime;
83 
84  // A flag indicating a REDUCE error when computing this RedAns.
85  // Since there is no evaluation result in case of an error, ALL
86  // REDUCE output including the "on time" information then go into
87  // the pretext field. Besides this, only the nextprompt field
88  // contains a meaningful value. When passing an incomplete line to
89  // RedAns_new() there occurs an error in the sense discussed here.
90  // Anyway, sending the remainder of the line via another
91  // RedAns_new() then leads to a correct computation.
92  int error;
