394  boost::optional<Explanation> explainInconsistency() {
395  if (isConsistent()) {
396  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Trail is still consistent");
397  return boost::none;
398  } else {
399  const auto& conflvar = mInconsistentVariables.front();
400  auto val = mGetter.getBoolVarValue(conflvar);
401  assert(val != l_Undef);
402  const auto confl = (val == l_True) ? mGetter.reabstractVariable(conflvar) : mGetter.reabstractVariable(conflvar).negated();
403 
404  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Inconsistent: " << confl << " evaluates to false");
405  // pick any unassigned variable in confl (it must exist, otherwise the AssignmentFinder is incorrect)
406  carl::Variables vars;
407  confl.arithmeticVars(vars);
408  for (const auto& avar : mBackend.assignedVariables())
409  vars.erase(avar);
410  assert(vars.size() > 0);
411  auto explanation = mBackend.explain(*(vars.begin()), {confl});
412  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Got a conflict: " << explanation);
413  return explanation;
414  }

417  Explanation explainTheoryPropagation(Minisat::Lit literal) {
418  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Current state: " << (*this));
419  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Explaining " << literal << " under " << mBackend.getModel());
420  const auto& f = mGetter.reabstractLiteral(literal);
421 
422  carl::carlVariables vars;
423  f.gatherVariables(vars);
424  for (const auto& v : mBackend.assignedVariables())
425  vars.erase(v);
426  assert(vars.size() == 1);
427  carl::Variable tvar = carl::underlying_variable(*(vars.begin()));
428 
429  auto conflict = mBackend.isInfeasible(tvar, !f);
430  assert(carl::variant_is_type<FormulasT>(conflict));
431  auto& confl = boost::get<FormulasT>(conflict);
432  assert( std::find(confl.begin(), confl.end(), !f) != confl.end() );
433  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Explaining " << f << " from " << confl);
434  auto res = mBackend.explain(tvar, !f, confl);
435  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Explaining " << f << " by " << res);
436  // f is part of the conflict, because the trail is feasible without f:
437  if (carl::variant_is_type<FormulaT>(res)) {
438  if (boost::get<FormulaT>(res).isFalse()) {
439  SMTRAT_LOG_DEBUG("smtrat.sat.mcsat", "Explanation failed.");
440  } else {
441  assert(boost::get<FormulaT>(res).contains(f));
442  }
443  }
444  else {
445  assert(boost::get<ClauseChain>(res).chain().back().clause().contains(f));
446  }
447  return res;
448  }

494  std::size_t computeTheoryLevel(const FormulaT& f) const {
495  SMTRAT_LOG_TRACE("smtrat.sat.mcsat", "Computing theory level for " << f);
496  carl::carlVariables vars;
497  f.gatherVariables(vars);
498  if (vars.empty()) {
499  SMTRAT_LOG_TRACE("smtrat.sat.mcsat", f << " has no variable, thus on level 0");
500  return 0;
501  }
502 
503  Model m = model();
504  if (!carl::model::evaluate(f, m).isBool()) {
505  SMTRAT_LOG_TRACE("smtrat.sat.mcsat", f << " is undecided.");
506  return std::numeric_limits<std::size_t>::max();
507  }
508  for (std::size_t lvl = level(); lvl > 0; lvl--) {
509  if (variable(lvl) == carl::Variable::NO_VARIABLE) continue;
510  m.erase(variable(lvl));
511  if (!vars.has(variable(lvl))) continue;
512  if (!carl::model::evaluate(f, m).isBool()) {
513  return lvl;
514  }
515  }
516  assert(false);
517  return 0;
518  }

576  std::size_t maxDegree(const Minisat::Var& var) {
577  std::size_t v = varid(var);
578  assert(v < mVarPropertyCache.size());
579 
580  if (mVarPropertyCache[v].maxDegree == boost::none) {
581  if (!mGetter.isTheoryAbstraction(var)) {
582  mVarPropertyCache[v].maxDegree = 0;
583  } else {
584  const auto& reabstraction = mGetter.reabstractVariable(var);
585  if (reabstraction.getType() == carl::FormulaType::CONSTRAINT) {
586  const auto& constr = reabstraction.constraint();
587  carl::Variables vars;
588  reabstraction.arithmeticVars(vars);
589  std::size_t maxDeg = 0;
590  for (const auto& tvar : vars) {
591  std::size_t deg = constr.lhs().degree(tvar);
592  if (deg > maxDeg) maxDeg = deg;
593  }
594  mVarPropertyCache[v].maxDegree = maxDeg;
595  } else if (reabstraction.getType() == carl::FormulaType::VARCOMPARE) {
596  mVarPropertyCache[v].maxDegree = std::numeric_limits<std::size_t>::max();
597  } else {
598  assert(false);
599  }
600 
601  }
602  }
603
