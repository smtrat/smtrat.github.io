109  bool isSymbolFree(const std::string& name, bool output = true) {
110  std::stringstream out;
111  if (name == "true" || name == "false") out << "\"" << name << "\" is a reserved keyword.";
112  else if (variables.find(name) != variables.end()) out << "\"" << name << "\" has already been defined as a variable.";
113  else if (bindings.find(name) != bindings.end()) out << "\"" << name << "\" has already been defined as a binding to \"" << bindings[name] << "\".";
114  else if (constants.find(name) != constants.end()) out << "\"" << name << "\" has already been defined as a constant.";
115  else if (declared_functions.find(name) != declared_functions.end()) out << "\"" << name << "\" has already been declared as a function.";
116  else if (defined_functions.find(name) != defined_functions.end()) out << "\"" << name << "\" has already been defined as a function.";
117  else if (defined_indexed_functions.find(name) != defined_indexed_functions.end()) out << "\"" << name << "\" has already been defined as a function.";
118  else if (defined_user_functions.find(name) != defined_user_functions.end()) out << "\"" << name << "\" has already been defined as a function by the user.";
119  else return true;
120  if (output) SMTRAT_LOG_ERROR("smtrat.parser", out.str());
121  return false;
122  }
