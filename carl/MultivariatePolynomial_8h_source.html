213  MultivariatePolynomial coeff(Variable::Arg var, std::size_t exp) const {
214  MultivariatePolynomial res;
215  for (const auto& t: mTerms) {
216  if (t.monomial() == nullptr) {
217  if (exp == 0) res.mTerms.push_back(t);
218  } else if (t.monomial()->exponentOfVariable(var) == exp) {
219  Monomial::Arg newMon = t.monomial()->dropVariable(var);
220  res.mTerms.push_back(TermType(t.coeff(), newMon));
221  }
222  }
223  res.mOrdered = false;
224  res.makeMinimallyOrdered<true,true>();
