28 template<typename Coeff>
29 std::list<UnivariatePolynomial<Coeff>> subresultants(
30  const UnivariatePolynomial<Coeff>& pol1,
31  const UnivariatePolynomial<Coeff>& pol2,
32  SubresultantStrategy strategy
33 ) {
34  /* The algorithm consists of three parts:
35  * Part 1: Initialization, i.e. preparation of the input so that the requirements of the core algorithm in parts 2 and 3 are met.
36  * Part 2: First part of the main loop. If the two subresultants which were added before (initially the two inputs) differ by more
37  * than 1 in their degree, an intermediate subresultant is computed by reducing the last one added with the leading coefficient
38  * of the one before this one.
39  * Part 3: Second part of the main loop. The pseudo remainder of the last two subresultants (the one possibly added in Part 2 disregarded)
40  * is computed and added to the subresultant sequence.
41  */
42 
43  /* Part 1
44  * Check and normalize input, initialize local variables.
45  */
46  assert(pol1.mainVar() == pol2.mainVar());
47  CARL_LOG_TRACE("carl.core.resultant", "subresultants(" << pol1 << ", " << pol2 << ")");
48  std::list<UnivariatePolynomial<Coeff>> subresultants;
49  Variable variable = pol1.mainVar();
50 
51  assert(!pol1.isZero());
52  assert(!pol2.isZero());
53  // We initialize p and q with pol1 and pol2.
54  UnivariatePolynomial<Coeff> p(pol1), q(pol2);
55 
56  // pDeg >= qDeg shall hold, so switch if it does not hold
57  if (p.degree() < q.degree()) {
58  std::swap(p, q);
59  }
60  CARL_LOG_TRACE("carl.core.resultant", "p = " << p);
61  CARL_LOG_TRACE("carl.core.resultant", "q = " << q);
62 
63  subresultants.push_front(p);
64  if (q.isZero()) {
65  CARL_LOG_TRACE("carl.core.resultant", "q is Zero.");
66  return subresultants;
67  }
68  subresultants.push_front(q);
69 
70  // SPECIAL CASE: both, p and q, are constant
71  if (q.isConstant()) {
72  CARL_LOG_TRACE("carl.core.resultant", "q is constant.");
73  return subresultants;
74  }
75 
76  // Explicitly check preconditions
77  assert(p.degree() >= q.degree());
78  assert(q.degree() >= 1);
79 
80  // BUG in Duco's article(?):
81  //ex subresLcoeff = GiNaC::pow( a.lcoeff(), a.degree() - b.degree() ); // initialized on the basis of the smaller-degree polynomial
82  //Coeff subresLcoeff(a.lcoeff()); // initialized on the basis of the smaller-degree polynomial
83  Coeff subresLcoeff = q.lcoeff().pow(p.degree() - q.degree());
84  CARL_LOG_TRACE("carl.core.resultant", "subresLcoeff = " << subresLcoeff);
85 
86  UnivariatePolynomial<Coeff> tmp = q;
87  q = p.prem(-q);
88  p = tmp;
89  CARL_LOG_TRACE("carl.core.resultant", "q = p.prem(-q) = " << q);
90  CARL_LOG_TRACE("carl.core.resultant", "p = " << p);
91  //CARL_LOG_TRACE("carl.core.resultant", "p = q");
92  //CARL_LOG_TRACE("carl.core.resultant", "q = p.prem(-q)");
93 
94  /* Parts 2 and 3
95  * Main loop filling the subresultants chain step by step.
96  */
97  // MAIN: start main loop containing different computation strategies
98  while (true) {
99  CARL_LOG_TRACE("carl.core.resultant", "Looping...");
100  CARL_LOG_TRACE("carl.core.resultant", "p = " << p);
101  CARL_LOG_TRACE("carl.core.resultant", "q = " << q);
102  if (q.isZero()) return subresultants;
103  uint pDeg = p.degree();
104  uint qDeg = q.degree();
105  subresultants.push_front(q);
106 
107  // Part 2
108  assert(pDeg >= qDeg);
109  uint delta = pDeg - qDeg;
110  CARL_LOG_TRACE("carl.core.resultant", "delta = " << delta);
111 
119  UnivariatePolynomial<Coeff> c(variable);
120  if (delta > 1) {
121  // compute c
122  // Notation hints: Compared to [Duc98], here S_{d-1} is b and S_d is a, and S_e is c.
123  switch (strategy) {
124  case SubresultantStrategy::Generic: {
125  CARL_LOG_TRACE("carl.core.resultant", "Part 2: Generic strategy");
126  UnivariatePolynomial<Coeff> reductionCoeff = q.lcoeff().pow(delta - 1) * q;
127  Coeff dividant = subresLcoeff.pow(delta-1);
128  bool res = reductionCoeff.divideBy(dividant, c);
129  if (res) {
130  subresultants.push_front(c);
131  assert(!c.isZero());
132  qDeg = c.degree();
133  } else {
134  c = q;
135  }
136  break;
137  }
138  case SubresultantStrategy::Ducos:
139  case SubresultantStrategy::Lazard: {
140  CARL_LOG_TRACE("carl.core.resultant", "Part 2: Ducos/Lazard strategy");
141  // "dichotomous Lazard": efficient exponentiation
142  uint deltaReduced = delta-1;
143  CARL_LOG_TRACE("carl.core.resultant", "deltaReduced = " << deltaReduced);
144  // should be true by the loop condition
145  assert(deltaReduced > 0);
146 
147  Coeff lcoeffQ = q.lcoeff();
148  UnivariatePolynomial<Coeff> reductionCoeff(variable, lcoeffQ);
149 
150  CARL_LOG_TRACE("carl.core.resultant", "lcoeffQ = " << lcoeffQ);
151  CARL_LOG_TRACE("carl.core.resultant", "reductionCoeff = " << reductionCoeff);
152 
153  uint exponent = highestPower(deltaReduced);
154  deltaReduced -= exponent;
155  CARL_LOG_TRACE("carl.core.resultant", "exponent = " << exponent);
156  CARL_LOG_TRACE("carl.core.resultant", "deltaReduced = " << deltaReduced);
157 
158  while (exponent != 1) {
159  exponent /= 2;
160  CARL_LOG_TRACE("carl.core.resultant", "exponent = " << exponent);
161  bool res = (reductionCoeff*reductionCoeff).divideBy(subresLcoeff, reductionCoeff);
162  if (res && deltaReduced >= exponent) {
163  (reductionCoeff*lcoeffQ).divideBy(subresLcoeff, reductionCoeff);
164  deltaReduced -= exponent;
165  }
166  }
167  CARL_LOG_TRACE("carl.core.resultant", "reductionCoeff = " << reductionCoeff);
168  reductionCoeff *= q;
169  CARL_LOG_TRACE("carl.core.resultant", "reductionCoeff = " << reductionCoeff);
170  bool res = reductionCoeff.divideBy(subresLcoeff, c);
171  if (res) {
172  subresultants.push_front(c);
173  assert(!c.isZero());
174  qDeg = c.degree();
175  CARL_LOG_TRACE("carl.core.resultant", "qDeg = " << qDeg);
176  } else {
177  c = q;
178  }
179  CARL_LOG_TRACE("carl.core.resultant", "c = " << c);
180  break;
181  }
182  }
183  } else {
184  c = q;
185  }
186  if (qDeg == 0) return subresultants;
187 
188  CARL_LOG_TRACE("carl.core.resultant", "Mid");
189  //CARL_LOG_TRACE("carl.core.resultant", "p = " << p);
190  //CARL_LOG_TRACE("carl.core.resultant", "q = " << q);
191 
192  // Part 3
193  switch (strategy) {
194  // Compared to [Duc98], here S_{d-1} is b and S_d is a, S_e is c, and s_d is subresLcoeff.
195  case SubresultantStrategy::Generic:
196  case SubresultantStrategy::Lazard: {
197  CARL_LOG_TRACE("carl.core.resultant", "Part 3: Generic/Lazard strategy");
198  if (p.isZero()) return subresultants;
199 
200  /* If b was constant, the degree properties for subresultants are still met, enforcing us to disregard whether
201  * the above division was successful (in this case, reducedNewB remains unchanged).
202  * If it was successful, the resulting term is safely added to the list, yielding an optimized resultant.
203  */
204  UnivariatePolynomial<Coeff> reducedNewB = p.prem(-q);
205  bool r = reducedNewB.divideBy(subresLcoeff.pow(delta)*p.lcoeff(), q);
206  assert(r);
207  break;
208  }
209  case SubresultantStrategy::Ducos: {
210  CARL_LOG_TRACE("carl.core.resultant", "Part 3: Ducos strategy");
211  // Ducos' optimization
212  Coeff lcoeffQ = q.lcoeff();
213  Coeff lcoeffC = c.lcoeff();
214  std::vector<Coeff> h(pDeg);
215 
216  for (uint d = 0; d < qDeg; d++) {
217  h[d] = lcoeffC * Coeff(variable).pow(d);
218  }
219  if (pDeg != qDeg) { // => aDeg > bDeg
220  h[qDeg] = Coeff(lcoeffC * Coeff(variable).pow(qDeg) - c); // H_e
221  }
222  for (uint d = qDeg + 1; d < pDeg; d++) {
223  Coeff t = h[d-1] * variable;
224  UnivariatePolynomial<Coeff> reducedNewB = t.toUnivariatePolynomial(variable).coefficients()[qDeg] * q;
225  bool res = reducedNewB.divideBy(lcoeffQ, reducedNewB);
226  assert(res || reducedNewB.isConstant());
227  h[d] = Coeff(t - reducedNewB);
228  }
229 
230  UnivariatePolynomial<Coeff> sum(p.mainVar(), h.front() * p.coefficients()[0]);
231  for (uint d = 1; d < pDeg; d++) {
232  sum += h[d] * p.coefficients()[d];
233  }
234  UnivariatePolynomial<Coeff> normalizedSum(p.mainVar());
235  bool res = sum.divideBy(p.lcoeff(), normalizedSum);
236  assert(res || sum.isConstant());
237 
238  UnivariatePolynomial<Coeff> t(variable, {0, h.back()});
239  UnivariatePolynomial<Coeff> reducedNewB = ((t + normalizedSum) * lcoeffQ - t.coefficients()[qDeg].toUnivariatePolynomial(variable));
240  reducedNewB.divideBy(p.lcoeff(), reducedNewB);
241  if (delta % 2 == 0) {
242  q = -reducedNewB;
243  } else {
244  q = reducedNewB;
245  }
246  break;
247  }
248  }
249  p = c;
250  subresLcoeff = p.lcoeff();
251  }

290  const UnivariatePolynomial<Coeff>& p,
291  SubresultantStrategy strategy
292 ) {

298  bool result = res.divideBy(redCoeff, res);
299  assert(result);
300  CARL_LOG_TRACE("carl.cad", "discriminant(" << p << ") = " << res);
