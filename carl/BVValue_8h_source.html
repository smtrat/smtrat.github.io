46  {
47  // Obtain an mpz_t copy of _value
48  mpz_t value;
49  mpz_init(value);
50  mpz_set(value, _value.get_mpz_t());
51 
52  // Remember whether the given value is negative.
53  // The conversion is based on the mpz_export function, which ignores
54  // the sign. Hence, we have to construct the two's complement
55  // ourselves in case of negative numbers.
56  bool valueNegative = (mpz_sgn(value) == -1);
57 
58  if(valueNegative) {
59  // If the value is negative, increase it by 1 before conversion.
60  // This way we only have to flip all bits afterwards.
61  mpz_add_ui(value, value, 1);
62  }
63 
64  // We export into an array of Base::block_type.
65  // Calculate the number of needed blocks upfront
66  std::size_t bitsPerBlock = sizeof(Base::block_type) * CHAR_BIT;
67  std::size_t valueBits = mpz_sizeinbase(value, 2);
68  std::size_t blockCount = (valueBits + bitsPerBlock - 1) / bitsPerBlock;
69 
70  {
71  // The actual conversion from mpz_t to dynamic_bitset blocks
72  auto bits = std::make_unique<Base::block_type[]>(blockCount);
73  mpz_export(&bits[0], &blockCount, -1, sizeof(Base::block_type), -1, 0, value);
74 
75  // Import the blocks into mValue, and resize it afterwards to the desired size
76  mValue.append(&bits[0], &bits[0] + blockCount);
77  mValue.resize(_width);
78  }
79 
80  // If the value was negative, finalize the construction of the two's complement
81  // by inverting all bits
82  if(valueNegative) {
83  mValue.flip();
84  }
85  }
