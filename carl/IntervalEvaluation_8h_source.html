110 inline Interval<Numeric> IntervalEvaluation::evaluate(const FactorizedPolynomial<P>& p, const std::map<Variable, Interval<Numeric>>& map)
111 {
112  if( !existsFactorization( p ) )
113  return Interval<Numeric>( p.coefficient() );
114  if( p.factorizedTrivially() )
115  {
116  return evaluate( p.polynomial(), map ) * Interval<Numeric>( p.coefficient() );
117  }
118  else
119  {
120  Interval<Numeric> result( p.coefficient() );
121  for( const auto& factor : p.factorization() )
122  {
123  Interval<Numeric> factorEvaluated = evaluate( factor.first, map );
124  if( factorEvaluated.isZero() )
125  return factorEvaluated;
126  result *= factorEvaluated.pow( factor.second );
127  }
128  return std::move( result );
129  }

133 inline Interval<Numeric> IntervalEvaluation::evaluate(const UnivariatePolynomial<Coeff>& p, const std::map<Variable, Interval<Numeric>>& map) {
134  CARL_LOG_FUNC("carl.core.monomial", p << ", " << map);
135  assert(map.count(p.mainVar()) > 0);
136  Interval<Numeric> res = Interval<Numeric>::emptyInterval();
137  const Interval<Numeric> varValue = map.at(p.mainVar());
138  Interval<Numeric> exp(1);
139  for (unsigned i = 0; i < p.degree(); i++) {
140  res += p.coefficients()[i] * exp;
141  if( res.isInfinite() )
142  return res;
143  exp = varValue.pow(i+1);
144  }
145  return res;
