201 template<typename Number>
202 inline Number highestPower(const Number& n) {
203  static_assert(std::is_fundamental<Number>::value, "Only works on native types.");
204  uint iterations = 0;
205  // Number has 2^k Bits, we do k iterations
206  if (sizeof(Number) == 2) iterations = 4;
207  else if (sizeof(Number) == 4) iterations = 5;
208  else if (sizeof(Number) == 8) iterations = 6;
209  assert(iterations > 0);
210 
211  Number res = n;
212  for (uint i = 0; i < iterations; i++) {
213  res |= res >> (1 << i);
214  }
215  res -= res >> 1;
216  return res;
217 }
