191  Sign sgn(const Polynomial& p) const {
192  if (isNumeric()) {
193  return carl::sgn(p.evaluate(mValue));
194  } else {
195  if (mIR->polynomial == p) return Sign::ZERO;
196  auto seq = mIR->polynomial.standardSturmSequence(mIR->polynomial.derivative() * p);
197  int variations = Polynomial::countRealRoots(seq, mIR->interval);
198  assert((variations == -1) || (variations == 0) || (variations == 1));
199  switch (variations) {
200  case -1: return Sign::NEGATIVE;
201  case 0: return Sign::ZERO;
202  case 1: return Sign::POSITIVE;
203  }
204  CARL_LOG_ERROR("carl.ran", "Unexpected number of variations, should be -1, 0, 1 but was " << variations);
205  return Sign::ZERO;
206  }
207  }
