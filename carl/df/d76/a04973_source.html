106  /**
107  * Test if the given point satisfies this constraint. The variables of this constraint are substituted by the components of r in ascending order until all variables are substituted, even if r is larger.
108  * @param r test point
109  * @return false if the constraint was not satisfied by the given point, true otherwise.
110  */
111  bool satisfiedBy(const RealAlgebraicPoint<Number>& r, const std::vector<Variable>& _variables) const {
112  assert(_variables.size() == r.dim());
113 
114  auto res = RealAlgebraicNumberEvaluation::evaluate(this->polynomial, r, _variables);
115  CARL_LOG_DEBUG("carl.cad.constraint", *this << " evaluates to " << res << " on " << r);
116  if (this->negated) {
117  return res.sgn() != this->sign;
118  } else {
119  return res.sgn() == this->sign;
120  }
121  }
