99 RealAlgebraicNumber<Number> evaluate(const MultivariatePolynomial<Coeff>& p, const RealAlgebraicPoint<Number>& point, const std::vector<Variable>& variables) {
100  assert(point.dim() == variables.size());
101  RANMap<Number> RANs;
102  MultivariatePolynomial<Coeff> pol(p);
103  for (std::size_t i = 0; i < point.dim(); i++) {
104  if (!pol.has(variables[i])) continue;
105  assert(pol.has(variables[i]));
106  if (point[i].isNumeric()) {
107  // Plug in numeric representations
108  pol.substituteIn(variables[i], MultivariatePolynomial<Coeff>(point[i].value()));
109  } else {
110  // Defer interval representations
111  RANs.emplace(variables[i], point[i]);
112  }
113  }
114  if (pol.isNumber()) {
115  return RealAlgebraicNumber<Number>(pol.constantPart());
116  }
117  return evaluate(pol, RANs);
118 }

159 template<typename Number>
160 RealAlgebraicNumber<Number> evaluateIR(const MultivariatePolynomial<Number>& p, const RANMap<Number>& m) {
161  CARL_LOG_DEBUG("carl.ran", "Evaluating " << p << " on " << m);
162  assert(m.size() > 0);
163  auto poly = p.toUnivariatePolynomial(m.begin()->first);
164  if (m.size() == 1 && m.begin()->second.sgn(poly.toNumberCoefficients()) == Sign::ZERO) {
165  return RealAlgebraicNumber<Number>(poly.mainVar());
166  }
167  Variable v = freshRealVariable();
168  // compute the result polynomial and the initial result interval
169  std::map<Variable, Interval<Number>> varToInterval;
170  UnivariatePolynomial<Number> res = evaluatePolynomial(UnivariatePolynomial<MultivariatePolynomial<Number>>(v, {MultivariatePolynomial<Number>(-p), MultivariatePolynomial<Number>(1)}), m, varToInterval);
171  assert(!varToInterval.empty());
172  poly = p.toUnivariatePolynomial(varToInterval.begin()->first);
173  CARL_LOG_DEBUG("carl.ran", "res = " << res);
174  CARL_LOG_DEBUG("carl.ran", "varToInterval = " << varToInterval);
175  CARL_LOG_DEBUG("carl.ran", "poly = " << poly);
176  Interval<Number> interval = IntervalEvaluation::evaluate(poly, varToInterval);
177  CARL_LOG_DEBUG("carl.ran", "-> " << interval);
178 
179  auto sturmSeq = res.standardSturmSequence();
180  // the interval should include at least one root.
181  assert(!res.isZero());
182  assert(
183  res.sgn(interval.lower()) == Sign::ZERO ||
184  res.sgn(interval.upper()) == Sign::ZERO ||
185  res.countRealRoots(sturmSeq, interval) >= 1
186  );
187  while (
188  res.sgn(interval.lower()) == Sign::ZERO ||
189  res.sgn(interval.upper()) == Sign::ZERO ||
190  res.countRealRoots(sturmSeq, interval) != 1) {
191  // refine the result interval until it isolates exactly one real root of the result polynomial
192  for (auto it = m.begin(); it != m.end(); it++) {
193  it->second.refine();
194  if (it->second.isNumeric()) {
195  return evaluate(p, m);
196  } else if (it->second.isInterval()) {
197  varToInterval[it->first] = it->second.getInterval();
198  } else {
199  CARL_LOG_WARN("carl.ran", "Unknown type of RAN.");
200  }
201  }
202  interval = IntervalEvaluation::evaluate(poly, varToInterval);
203  }
204  CARL_LOG_DEBUG("carl.ran", "Result is " << RealAlgebraicNumber<Number>(res, interval, sturmSeq));
205  return RealAlgebraicNumber<Number>(res, interval, sturmSeq);
206 }

209 template<typename Number, typename Coeff>
210 UnivariatePolynomial<Number> evaluatePolynomial(
211  const UnivariatePolynomial<Coeff>& p,
212  const std::map<Variable, RealAlgebraicNumber<Number>>& m,
213  std::map<Variable, Interval<Number>>& varToInterval
214 ) {
215  CARL_LOG_DEBUG("carl.ran", "Evaluating " << p << " on " << m);
216  Variable v = p.mainVar();
217  UnivariatePolynomial<Coeff> tmp = p;
218  for (const auto& i: m) {
219  if (!tmp.has(i.first)) {
220  if (p.has(i.first)) {
221  // Variable vanished, add it to varToInterval
222  varToInterval[i.first] = i.second.getInterval();
223  }
224  continue;
225  }
226  if (i.second.isNumeric()) {
227  CARL_LOG_DEBUG("carl.ran", "Direct substitution: " << i.first << " = " << i.second);
228  tmp.substituteIn(i.first, Coeff(i.second.value()));
229  } else if (i.second.isInterval()) {
230  CARL_LOG_DEBUG("carl.ran", "IR substitution: " << i.first << " = " << i.second);
231  i.second.simplifyByPolynomial(i.first, MultivariatePolynomial<Number>(tmp));
232  UnivariatePolynomial<Coeff> p2(i.first, i.second.getIRPolynomial().template convert<Coeff>().coefficients());
233  CARL_LOG_DEBUG("carl.ran", "Simplifying " << tmp.switchVariable(i.first) << " with " << p2);
234  tmp = tmp.switchVariable(i.first).prem(p2);
235  CARL_LOG_DEBUG("carl.ran", "Using " << p2 << " with " << tmp);
236  tmp = carl::resultant(tmp, p2);
237  CARL_LOG_DEBUG("carl.ran", "-> " << tmp);
238  varToInterval[i.first] = i.second.getInterval();
239  } else {
240  CARL_LOG_WARN("carl.ran", "Unknown type of RAN.");
241  }
242  CARL_LOG_DEBUG("carl.ran", "Substituted " << i.first << " -> " << i.second << ", result: " << tmp);
243  }
244  CARL_LOG_DEBUG("carl.ran", "Result: " << tmp.switchVariable(v).toNumberCoefficients());
245  return tmp.switchVariable(v).toNumberCoefficients();
246 }
