81  CoCoA::RingElem convert(const Poly& p) const {
82  CoCoA::RingElem res(mRing);
83  for (const auto& t: p) {
84  if (!t.monomial()) {
85  res += convert(t.coeff());
86  continue;
87  }
88  std::vector<long> exponents(mSymbolBack.size());
89  for (const auto& p: *t.monomial()) {
90  auto it = mSymbolThere.find(p.first);
91  assert(it != mSymbolThere.end());
92  long indetIndex;
93  if (CoCoA::IsIndet(indetIndex, it->second)) {
94  exponents[std::size_t(indetIndex)] = long(p.second);
95  } else {
96  assert(false && "The symbol is not an inderminant.");
97  }
98  }
99  res += CoCoA::monomial(mRing, convert(t.coeff()), exponents);
100  }
101  return res;
102  }

104  Poly convert(const CoCoA::RingElem& p) const {
105  Poly res;
106  for (CoCoA::SparsePolyIter i = CoCoA::BeginIter(p); !CoCoA::IsEnded(i); ++i) {
107  typename Poly::CoeffType coeff;
108  convert(coeff, CoCoA::coeff(i));
109  if (CoCoA::IsOne(CoCoA::PP(i))) {
110  res += coeff;
111  } else {
112  std::vector<long> exponents;
113  CoCoA::exponents(exponents, CoCoA::PP(i));
114  Monomial::Content monContent;
115  std::size_t tdeg = 0;
116  for (std::size_t i = 0; i < exponents.size(); ++i) {
117  if (exponents[i] == 0) continue;
118  monContent.emplace_back(mSymbolBack[i], exponents[i]);
119  tdeg += std::size_t(exponents[i]);
120  }
121  res += typename Poly::TermType(std::move(coeff), createMonomial(std::move(monContent), tdeg));
122  }
123  }
124  return res;
