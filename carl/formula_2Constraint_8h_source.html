858  template<typename Pol>
859  signed compare( const Constraint<Pol>& _constraintA, const Constraint<Pol>& _constraintB )
860  {
861  /*
862  * Check whether it holds that
863  *
864  * _constraintA = a_1*m_1+...+a_k*m_k + c ~ 0
865  * and
866  * _constraintB = b_1*m_1+...+b_k*m_k + d ~ 0,
867  *
868  * where a_1,..., a_k, b_1,..., b_k, c, d are rational coefficients,
869  * m_1,..., m_k are non-constant monomials and
870  * exists a rational g such that
871  *
872  * a_i = g * b_i for all 1<=i<=k
873  * or b_i = g * a_i for all 1<=i<=k
874  */
875  typename Pol::NumberType one_divided_by_a = _constraintA.lhs().coprimeFactorWithoutConstant();
876  typename Pol::NumberType one_divided_by_b = _constraintB.lhs().coprimeFactorWithoutConstant();
877  typename Pol::NumberType c = _constraintA.lhs().constantPart();
878  typename Pol::NumberType d = _constraintB.lhs().constantPart();
879  assert( carl::isOne(carl::getNum(carl::abs(one_divided_by_b))) );
880  Pol tmpA = (_constraintA.lhs() - c) * one_divided_by_a;
881  Pol tmpB = (_constraintB.lhs() - d) * one_divided_by_b;
882 // std::cout << "tmpA = " << tmpA << std::endl;
883 // std::cout << "tmpB = " << tmpB << std::endl;
884  if( tmpA != tmpB ) return 0;
885  bool termACoeffGreater = false;
886  bool signsDiffer = (one_divided_by_a < carl::constant_zero<typename Pol::NumberType>::get()) != (one_divided_by_b < carl::constant_zero<typename Pol::NumberType>::get());
887  typename Pol::NumberType g;
888  if( carl::getDenom(one_divided_by_a) > carl::getDenom(one_divided_by_b) )
889  {
890  g = typename Pol::NumberType(carl::getDenom(one_divided_by_a))/carl::getDenom(one_divided_by_b);
891  if( signsDiffer )
892  g = -g;
893  termACoeffGreater = true;
894  d *= g;
895  }
896  else
897  {
898  g = typename Pol::NumberType(carl::getDenom(one_divided_by_b))/carl::getDenom(one_divided_by_a);
899  if( signsDiffer )
900  g = -g;
901  c *= g;
902  }
903  // Apply the multiplication by a negative g to the according relation symbol, which
904  // has to be turned around then.
905  Relation relA = _constraintA.relation();
906  Relation relB = _constraintB.relation();
907  if( g < 0 )
908  {
909  if( termACoeffGreater )
910  {
911  switch( relB )
912  {
913  case Relation::LEQ:
914  relB = Relation::GEQ;
915  break;
916  case Relation::GEQ:
917  relB = Relation::LEQ;
918  break;
919  case Relation::LESS:
920  relB = Relation::GREATER;
921  break;
922  case Relation::GREATER:
923  relB = Relation::LESS;
924  break;
925  default:
926  break;
927  }
928  }
929  else
930  {
931  switch( relA )
932  {
933  case Relation::LEQ:
934  relA = Relation::GEQ;
935  break;
936  case Relation::GEQ:
937  relA = Relation::LEQ;
938  break;
939  case Relation::LESS:
940  relA = Relation::GREATER;
941  break;
942  case Relation::GREATER:
943  relA = Relation::LESS;
944  break;
945  default:
946  break;
947  }
948  }
949  }
950 // std::cout << "c = " << c << std::endl;
951 // std::cout << "d = " << d << std::endl;
952 // std::cout << "g = " << g << std::endl;
953 // std::cout << "relA = " << relA << std::endl;
954 // std::cout << "relB = " << relB << std::endl;
955  // Compare the adapted constant parts.
956  switch( relB )
957  {
958  case Relation::EQ:
959  switch( relA )
960  {
961  case Relation::EQ: // p+c=0 and p+d=0
962  if( c == d ) return A_IFF_B;
963  return NOT__A_AND_B;
964  case Relation::NEQ: // p+c!=0 and p+d=0
965  if( c == d ) return A_XOR_B;
966  return B_IMPLIES_A;
967  case Relation::LESS: // p+c<0 and p+d=0
968  if( c < d ) return B_IMPLIES_A;
969  return NOT__A_AND_B;
970  case Relation::GREATER: // p+c>0 and p+d=0
971  if( c > d ) return B_IMPLIES_A;
972  return NOT__A_AND_B;
973  case Relation::LEQ: // p+c<=0 and p+d=0
974  if( c <= d ) return B_IMPLIES_A;
975  return NOT__A_AND_B;
976  case Relation::GEQ: // p+c>=0 and p+d=0
977  if( c >= d ) return B_IMPLIES_A;
978  return NOT__A_AND_B;
979  default:
980  return false;
981  }
982  case Relation::NEQ:
983  switch( relA )
984  {
985  case Relation::EQ: // p+c=0 and p+d!=0
986  if( c == d ) return A_XOR_B;
987  return A_IMPLIES_B;
988  case Relation::NEQ: // p+c!=0 and p+d!=0
989  if( c == d ) return A_IFF_B;
990  return 0;
991  case Relation::LESS: // p+c<0 and p+d!=0
992  if( c >= d ) return A_IMPLIES_B;
993  return 0;
994  case Relation::GREATER: // p+c>0 and p+d!=0
995  if( c <= d ) return A_IMPLIES_B;
996  return 0;
997  case Relation::LEQ: // p+c<=0 and p+d!=0
998  if( c > d ) return A_IMPLIES_B;
999  if( c == d ) return A_AND_B__IFF_C;
1000  return 0;
1001  case Relation::GEQ: // p+c>=0 and p+d!=0
1002  if( c < d ) return A_IMPLIES_B;
1003  if( c == d ) return A_AND_B__IFF_C;
1004  return 0;
1005  default:
1006  return 0;
1007  }
1008  case Relation::LESS:
1009  switch( relA )
1010  {
1011  case Relation::EQ: // p+c=0 and p+d<0
1012  if( c > d ) return A_IMPLIES_B;
1013  return NOT__A_AND_B;
1014  case Relation::NEQ: // p+c!=0 and p+d<0
1015  if( c <= d ) return B_IMPLIES_A;
1016  return 0;
1017  case Relation::LESS: // p+c<0 and p+d<0
