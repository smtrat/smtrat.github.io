89  template<>
90  inline bool AlmostEqual2sComplement<double>(const double& A, const double& B, unsigned maxUlps)
91  {
92  // Make sure maxUlps is non-negative and small enough that the
93  // default NAN won't compare as equal to anything.
94  assert(maxUlps > 0 && maxUlps < 4 * 1024 * 1024);
95  sint aInt = *reinterpret_cast<const sint*>(&A);
96  // Make aInt lexicographically ordered as a twos-complement int
97  if (aInt < 0)
98  aInt = static_cast<sint>(0x8000000000000000) - aInt;
99  // Make bInt lexicographically ordered as a twos-complement int
100  sint bInt = *reinterpret_cast<const sint*>(&B);
101  if (bInt < 0)
102  bInt = static_cast<sint>(0x8000000000000000) - bInt;
103  auto intDiff = static_cast<uint>(std::abs(aInt - bInt));
104  return intDiff <= maxUlps;
105  }
