29  CARL_LOG_DEBUG("carl.model.evaluation", "Evaluating " << f << " on " << m);
30  assert(f.getType() == FormulaType::VARCOMPARE);
31  const auto& vc = f.variableComparison();
32 
33  ModelValue<Rational,Poly> cmp = vc.value();
34  if (cmp.isSubstitution()) {
35  // If assigned directly, the shared_ptr<Substitution> goes out of scope before the result is copied into cmp.
36  // Therefore, we start by copying the data and overwriting it afterwards.
37  CARL_LOG_DEBUG("carl.model.evaluation", "Evaluating " << cmp.asSubstitution() << " on " << m);
38  auto res = cmp.asSubstitution()->evaluate(m);
39  if (res.isBool() && !res.asBool()) {
40  CARL_LOG_DEBUG("carl.model.evaluation", "MVRoot does not exist, returning false");
41  if (vc.negated()) {
42  f = Formula<Poly>(FormulaType::TRUE);
43  } else {
44  f = Formula<Poly>(FormulaType::FALSE);
45  }
46  return;
47  }
48  CARL_LOG_DEBUG("carl.model.evaluation", "Evaluated substitution " << cmp << " -> " << res);
49  cmp = res;
50  }
51  if (cmp.isSubstitution()) {
52  CARL_LOG_DEBUG("carl.model.evaluation", "MVRoot is still a substitution, cannot evaluate.");
53  return;
54  }
55  assert(cmp.isRational() || cmp.isRAN());
56  RealAlgebraicNumber<Rational> val = cmp.isRational() ? RealAlgebraicNumber<Rational>(cmp.asRational()) : cmp.asRAN();
57  CARL_LOG_DEBUG("carl.model.evaluation", "rhs is " << val);
58 
59  auto it = m.find(vc.var());
60  if (it == m.end()) {
61  CARL_LOG_DEBUG("carl.model.evaluation", "Could not evaluate " << vc << " as " << vc.var() << " is not part of the model");
62  return;
63  }
64  const auto& value = m.evaluated(vc.var());
65  assert(value.isRational() || value.isRAN());
66  RealAlgebraicNumber<Rational> reference = value.isRational() ? RealAlgebraicNumber<Rational>(value.asRational()) : value.asRAN();
67  CARL_LOG_DEBUG("carl.model.evaluation", "Reference value: " << vc.var() << " == " << reference);
68 
69  f = Formula<Poly>(FormulaType::FALSE);

71  switch (vc.relation()) {
72  case Relation::EQ:
73  if (reference == val.changeVariable(vc.var())) f = Formula<Poly>(FormulaType::TRUE);
74  break;
75  case Relation::NEQ:
76  if (reference != val.changeVariable(vc.var())) f = Formula<Poly>(FormulaType::TRUE);
77  break;
78  case Relation::LESS:
79  if (reference < val.changeVariable(vc.var())) f = Formula<Poly>(FormulaType::TRUE);
80  break;
81  case Relation::LEQ:
82  if (reference <= val.changeVariable(vc.var())) f = Formula<Poly>(FormulaType::TRUE);
83  break;
84  case Relation::GREATER:
85  if (reference > val.changeVariable(vc.var())) f = Formula<Poly>(FormulaType::TRUE);
86  break;
87  case Relation::GEQ:
88  if (reference >= val.changeVariable(vc.var())) f = Formula<Poly>(FormulaType::TRUE);
89  break;
90  }
91  if (vc.negated()) {

117  template<typename Rational, typename Poly>
118  void substituteIn(Formula<Poly>& f, const Model<Rational,Poly>& m) {
119  switch (f.getType()) {
120  case FormulaType::ITE: {
121  f = Formula<Poly>(FormulaType::ITE, substitute(f.condition(), m), substitute(f.firstCase(), m), substitute(f.secondCase(), m));
122  break;
123  }
124  case FormulaType::EXISTS:
125  CARL_LOG_WARN("carl.model.evaluation", "Evaluation of exists not yet implemented.");
126  break;
127  case FormulaType::FORALL:
128  CARL_LOG_WARN("carl.model.evaluation", "Evaluation of forall not yet implemented.");
129  break;
130  case FormulaType::TRUE: break;
131  case FormulaType::FALSE: break;
132  case FormulaType::BOOL: {
133  auto it = m.find(f.boolean());
134  if (it == m.end()) break;
135  if (it->second.isBool()) {
136  f = Formula<Poly>(it->second.asBool() ? FormulaType::TRUE : FormulaType::FALSE);
137  } else {
138  CARL_LOG_WARN("carl.model.evaluation", "Could not evaluate " << it->first << " as a boolean, value is " << it->second);
139  }
140  break;
141  }
142  case FormulaType::NOT: {
143  f = Formula<Poly>(FormulaType::NOT, substitute(f.subformula(), m));
144  break;
145  }
146  case FormulaType::IMPLIES: {
147  f = Formula<Poly>(FormulaType::IMPLIES, substitute(f.premise(), m), substitute(f.conclusion(), m));
148  break;
149  }
150  case FormulaType::AND: substituteSubformulas(f, m); break;
151  case FormulaType::OR: substituteSubformulas(f, m); break;
152  case FormulaType::XOR: substituteSubformulas(f, m); break;
153  case FormulaType::IFF: substituteSubformulas(f, m); break;
154  case FormulaType::CONSTRAINT: {
155  auto res = evaluate(f.constraint(), m);
156  if (res.isBool()) {
157  if (res.asBool()) f = Formula<Poly>(FormulaType::TRUE);
158  else f = Formula<Poly>(FormulaType::FALSE);
159  } else {
160  assert(res.isSubstitution());
161  auto subs = res.asSubstitution();
162  auto fsubs = static_cast<ModelFormulaSubstitution<Rational,Poly>*>(subs.get());
163  f = fsubs->getFormula();
164  }
165  break;
166  }
167  case FormulaType::VARCOMPARE: {
168  evaluateVarCompare(f, m);
169  break;
170  }
171  case FormulaType::VARASSIGN: {
172  evaluateVarAssign(f, m);
173  break;
174  }
175  case FormulaType::BITVECTOR: {
176  BVConstraint bvc = substitute(f.bvConstraint(), m);
177  if (bvc.isAlwaysConsistent()) f = Formula<Poly>(FormulaType::TRUE);
178  else if (bvc.isAlwaysInconsistent()) f = Formula<Poly>(FormulaType::FALSE);
179  else f = Formula<Poly>(bvc);
180  break;
181  }
182  case FormulaType::UEQ: {
183  std::set<UVariable> vars;
184  f.uequality().collectUVariables(vars);
185  if (m.contains(vars)) {
186  auto val = evaluate(f.uequality(), m);
187  assert(val.isBool());
188  f = Formula<Poly>(val.asBool() ? FormulaType::TRUE : FormulaType::FALSE);
189  } else {
190  CARL_LOG_WARN("carl.model.evaluation", "Could not evaluate " << f.uequality() << " as some variables are missing from the model.");
191  }
192  break;
193  }
194  case FormulaType::PBCONSTRAINT: {
196  PBConstraint<Poly> pbc = substitute(f.pbConstraint(), m);
197  break;
198  }
199  }
