644  template<typename It>
645  bool isSampleConsistent(It node) const {
646  bool lastRoot = false;
647  for (auto cur = sampleTree.begin_children(node); cur != sampleTree.end_children(node); cur++) {
648  if (cur->isRoot() && lastRoot) return false;
649  lastRoot = cur->isRoot();
650  if (!isSampleConsistent(cur)) return false;
651  }
652  return true;
653  }

657  if (!isOk) {
658  CARL_LOG_ERROR("carl.cad", "SampleTree: " << this->sampleTree);
659  }
660  return true;
661  assert(isOk);
