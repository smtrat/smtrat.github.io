428  const FormulaContent<Pol>* create( const FormulasMulti<Pol>& _subformulas )
429  {
430  if( _subformulas.empty() ) return falseFormula();
431  if( _subformulas.size() == 1 )
432  {
433  return _subformulas.begin()->mpContent;
434  }
435  Formulas<Pol> subFormulas;
436  auto lastSubFormula = _subformulas.begin();
437  auto subFormula = lastSubFormula;

441  {
442  if( *lastSubFormula == *subFormula )
443  {
444  ++counter;
445  }
