48  template<typename Poly>
49  Poly convertMV(const CoCoA::RingElem& p, const ConversionInfo& ci) const {
50  Poly res;
51  for (CoCoA::SparsePolyIter i = CoCoA::BeginIter(p); !CoCoA::IsEnded(i); ++i) {
52  Poly coeff;
53  CoCoA::BigRat numcoeff;
54  if (CoCoA::IsRational(numcoeff, CoCoA::coeff(i))) {
55  coeff = Poly(convert<typename Poly::CoeffType>(numcoeff));
56  } else {
57  coeff = convertMV<Poly>(CoCoA::CanonicalRepr(CoCoA::coeff(i)), ci);
58  }
59  if (CoCoA::IsOne(CoCoA::PP(i))) {
60  res += coeff;
61  } else {
62  std::vector<long> exponents;
63  CoCoA::exponents(exponents, CoCoA::PP(i));
64  Monomial::Content monContent;
65  std::size_t tdeg = 0;
66  for (std::size_t i = 0; i < exponents.size(); ++i) {
67  if (exponents[i] == 0) continue;
68  const auto& ring = CoCoA::owner(p);
69  monContent.emplace_back(ci.mSymbolBack.at(std::make_pair(CoCoA::RingID(ring),i)), exponents[i]);
70  tdeg += std::size_t(exponents[i]);
71  }
72  res += coeff * createMonomial(std::move(monContent), tdeg);
73  }
74  }
75  return res;
76  }
