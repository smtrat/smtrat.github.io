482  std::map<Variable, VariableInformation<false, FactorizedPolynomial<P>>> resultVarInfos;

494  VariablesInformation<true, FactorizedPolynomial<P>> getVarInfo() const
495  {
496  if( existsFactorization( *this ) )
497  {
498  // TODO: Maybe we should use the factorization for collecting degrees and coefficients.
499  VariablesInformation<true, P> vi = polynomial().template getVarInfo<true>();
500  VariablesInformation<true, FactorizedPolynomial<P>> result;
501  for( const auto& varViPair : vi )
502  {
503  const auto& varI = varViPair.second;
504  std::map<unsigned,FactorizedPolynomial<P>> coeffs;
505  for( const auto& expCoeffPair : varI.coeffs() )
506  {
507  if( expCoeffPair.second.isConstant() )
508  {
509  coeffs.insert( coeffs.end(), std::make_pair( expCoeffPair.first, FactorizedPolynomial<P>( expCoeffPair.second.constantPart() * mCoefficient ) ) );
510  }
511  else
512  {
513  coeffs.insert( coeffs.end(), std::make_pair( expCoeffPair.first, FactorizedPolynomial<P>( expCoeffPair.second, mpCache ) * mCoefficient ) );
514  }
515  }
516  VariableInformation<true, FactorizedPolynomial<P>> viFactorized( varI.maxDegree(), varI.minDegree(), varI.occurence(), std::move( coeffs ) );
517  result.insert( result.end(), std::make_pair( varViPair.first, std::move( viFactorized ) ) );
518  }
519  return result;
520  }
521  return VariablesInformation<false, FactorizedPolynomial<P>>();
522  }
