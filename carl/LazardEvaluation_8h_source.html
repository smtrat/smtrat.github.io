28  Poly convertMV(const CoCoA::RingElem& p, const ConversionInfo& ci) const {
29  Poly res;
30  for (CoCoA::SparsePolyIter i = CoCoA::BeginIter(p); !CoCoA::IsEnded(i); ++i) {
31  Poly coeff;
32  CoCoA::BigRat numcoeff;
33  if (CoCoA::IsRational(numcoeff, CoCoA::coeff(i))) {
34  coeff = Poly(convert(numcoeff));
35  } else {
36  coeff = convertMV<Poly>(CoCoA::CanonicalRepr(CoCoA::coeff(i)), ci);
37  }
38  if (CoCoA::IsOne(CoCoA::PP(i))) {
39  res += coeff;
40  } else {
41  std::vector<long> exponents;
42  CoCoA::exponents(exponents, CoCoA::PP(i));
43  Monomial::Content monContent;
44  std::size_t tdeg = 0;
45  for (std::size_t i = 0; i < exponents.size(); ++i) {
46  if (exponents[i] == 0) continue;

48  monContent.emplace_back(ci.mSymbolBack.at(std::make_pair(CoCoA::RingID(ring),i)), exponents[i]);
49  tdeg += std::size_t(exponents[i]);
50  }
51  res += coeff * createMonomial(std::move(monContent), tdeg);
52  }
53  }
54  return res;
