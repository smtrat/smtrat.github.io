366  std::string toString(const std::string& _init = "", bool _oneline = true, bool _infix = false, bool _friendlyNames = true) const
367  {
368  if(isInvalid()) {
369  return _init + "%invalid%";
370  }
371  else if (mType == BVTermType::CONSTANT) {
372  return _init + getValue().toString();
373  }
374  else if (mType == BVTermType::VARIABLE) {
375  return _init + getVariable().toString(_friendlyNames);
376  }
377  else {
378  std::string operatorStr = carl::toString(mType);
379  std::string operatorPrefix = operatorStr;
380  std::string operatorInfix = operatorStr;
381 
382  std::string argFirst;
383  std::string argSecond;
384  std::string argThird;
385 
386  // Rewrite operator strings for indexed (parameterized) operators
387  if(mType == BVTermType::EXTRACT) {
388  operatorPrefix = "(_ " + operatorStr + " " + std::to_string(getExtract().mHighest) + " " + std::to_string(getExtract().mLowest) + ")";
389  operatorInfix = operatorStr + "_{" + std::to_string(getExtract().mHighest) + "," + std::to_string(getExtract().mLowest) + "}";
390  } else if(mType == BVTermType::LROTATE || mType == BVTermType::RROTATE
391  || mType == BVTermType::EXT_U || mType == BVTermType::EXT_S || mType == BVTermType::REPEAT) {
392  operatorPrefix = "(_ " + operatorStr + " " + std::to_string(getUnary().mIndex) + ")";
393  operatorInfix = operatorStr + "_" + std::to_string(getUnary().mIndex);
394  }
395 
396  // Fill arg* variables
397  if (mType == BVTermType::EXTRACT) {
398  argFirst = getExtract().mOperand.toString((_oneline ? "" : _init + " "), _oneline, _infix, _friendlyNames);
399  }
400  else if (typeIsUnary(mType)) {
401  argFirst = getUnary().mOperand.toString((_oneline ? "" : _init + " "), _oneline, _infix, _friendlyNames);
402  }
403  else if (typeIsBinary(mType)) {
404  argFirst = getBinary().mFirst.toString((_oneline ? "" : _init + " "), _oneline, _infix, _friendlyNames);
405  argSecond = getBinary().mSecond.toString((_oneline ? "" : _init + " "), _oneline, _infix, _friendlyNames);
406  }
407  else {
408  assert(false);
409  }
410 
411  if(_infix && !argFirst.empty() && !argSecond.empty() && argThird.empty()) {
412  // Infix notation is only really infix if we have exactly two arguments
413  return _init + "("
414  + (_oneline ? "" : "\n")
415  + argFirst
416  + (_oneline ? " " : ("\n" + _init))
417  + operatorInfix
418  + (_oneline ? " " : "\n")
419  + argSecond
420  + (_oneline ? "" : ("\n" + _init))
421  + ")";
422  } else {
423  return _init
424  + (_infix ? (operatorInfix + "(") : ("(" + operatorPrefix + (_oneline ? " " : "")))
425  + (_oneline ? "" : "\n") + argFirst
426  + (argSecond.empty() ? "" : ((_oneline ? " " : "\n") + argSecond))
427  + (argThird.empty() ? "" : ((_oneline ? " " : "\n") + argThird))
