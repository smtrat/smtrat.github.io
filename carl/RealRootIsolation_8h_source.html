63  bool isolate_roots_trivially() {
64  CARL_LOG_DEBUG("carl.core.rootfinder", "Trying to trivially solve mPolynomial " << mPolynomial);
65  switch (mPolynomial.degree()) {
66  case 0: {
67  CARL_LOG_TRACE("carl.core.rootfinder", "Constant polynomial, thus no roots");
68  break;
69  }
70  case 1: {
71  CARL_LOG_DEBUG("carl.core.rootfinder", "Trivially solving linear mPolynomial " << mPolynomial);
72  const auto& a = mPolynomial.coefficients()[1];
73  const auto& b = mPolynomial.coefficients()[0];
74  assert(!carl::isZero(a));
75  add_trivial_root(-b / a);
76  break;
77  }
78  case 2: {
79  CARL_LOG_DEBUG("carl.core.rootfinder", "Trivially solving quadratic mPolynomial " << mPolynomial);
80  const auto& a = mPolynomial.coefficients()[2];
81  const auto& b = mPolynomial.coefficients()[1];
82  const auto& c = mPolynomial.coefficients()[0];
83  assert(!carl::isZero(a));
84  /* Use this formulation of p-q-formula:
85  * x = ( -b +- \sqrt{ b*b - 4*a*c } ) / (2*a)
86  */
87  Number rad = b*b - 4*a*c;
88  if (rad == 0) {
89  add_trivial_root(-b / (2*a));
90  } else if (rad > 0) {
91  std::pair<Number, Number> res = carl::sqrt_fast(rad);
92  if (res.first == res.second) {
93  // Root could be calculated exactly
94  add_trivial_root((-b - res.first) / (2*a));
95  add_trivial_root((-b + res.first) / (2*a));
96  } else {
97  // Root is within interval (res.first, res.second)
98  Interval<Number> r(res.first, BoundType::STRICT, res.second, BoundType::STRICT);
99  add_trivial_root((Number(-b) - r) / Number(2*a));
100  add_trivial_root((Number(-b) + r) / Number(2*a));
101  }
102  } else {
103  // No root.
104  }
105  break;
106  }
107  default:
108  return false;
109  }
110  return true;
111  }

176  void bisect_by_approximation(std::deque<Interval<Number>>& queue) {
177  assert(queue.empty());
178  // Convert polynomial coeffs to double
179  std::vector<double> coeffs;
180  for (const auto& n: mPolynomial.coefficients()) {
181  coeffs.emplace_back(toDouble(n));
182  }
183  CARL_LOG_DEBUG("carl.core.rootfinder", "Double coeffs: " << coeffs);
184 
185  // Get approximations of the roots
186  std::vector<double> approx = eigen::root_approximation(coeffs);
187  // Sort and make them unique
188  for (double& a: approx) {
189  a = std::round(a*1000) / 1000;
190  }
191  std::sort(approx.begin(), approx.end());
192  approx.erase(std::unique(approx.begin(), approx.end()), approx.end());
193  CARL_LOG_DEBUG("carl.core.rootfinder", "Double approximations: " << approx);
194  // Convert to rationals
195  std::vector<Number> roots;
196  for (double r: approx) {
197  if (!isNumber(r)) continue;
198  Number n = carl::rationalize<Number>(r);
199  if (!mInterval.contains(n)) continue;
200  if (carl::is_root_of(mPolynomial, n)) {
201  add_root(n);
202  }
203  roots.emplace_back(n);
204  }
205  CARL_LOG_DEBUG("carl.core.rootfinder", "Approx roots: " << roots);
206 
207  // Get interval endpoints
208  std::vector<Number> endpoints;
209  endpoints.emplace_back(mInterval.lower());
210  if (roots.size() > 0) {
211  for (std::size_t i = 0; i < roots.size() - 1; ++i) {
212  auto tmp = carl::sample(Interval<Number>(roots[i], BoundType::STRICT, roots[i+1], BoundType::STRICT));
213  if (carl::is_root_of(mPolynomial, tmp)) {
214  add_root(tmp);
215  }
216  endpoints.emplace_back(tmp);
217  }
218  }
219  endpoints.emplace_back(mInterval.upper());
220  CARL_LOG_DEBUG("carl.core.rootfinder", "Endpoints: " << endpoints);
221 
222  // Fill queue based on the endpoints
223  for (std::size_t i = 0; i < endpoints.size() - 1; ++i) {
224  queue.emplace_back(endpoints[i], BoundType::STRICT, endpoints[i+1], BoundType::STRICT);
225  }
226  CARL_LOG_DEBUG("carl.core.rootfinder", "Queue: " << queue);
227  }

230  void isolate_by_bisection() {
231  std::deque<Interval<Number>> queue;
232  if (initialize_bisection_by_approximation) {
233  bisect_by_approximation(queue);
234  } else {
235  queue.emplace_back(mInterval);
236  }
237 
238  while (!queue.empty()) {
239  auto cur = queue.front();
240  queue.pop_front();
241 
242  auto variations = carl::sign_variations(mPolynomial, cur);
243 
244  if (variations == 0) {
245  CARL_LOG_DEBUG("carl.core.rootfinder", "No root within " << cur);
246  continue;
247  }
248  if (variations == 1) {
249  CARL_LOG_DEBUG("carl.core.rootfinder", "A single root within " << cur);
250  assert(!carl::is_root_of(mPolynomial, cur.lower()));
251  assert(!carl::is_root_of(mPolynomial, cur.upper()));
252  assert(count_real_roots(mPolynomial, cur) == 1);
253  add_root(cur);
254  continue;
255  }
256 
257  auto pivot = carl::sample(cur);
258  if (carl::is_root_of(mPolynomial, pivot)) {
259  add_root(pivot);
260  }
261  CARL_LOG_DEBUG("carl.core.rootfinder", "Splitting " << cur << " at " << pivot);
262  queue.emplace_back(cur.lower(), BoundType::STRICT, pivot, BoundType::STRICT);
263  queue.emplace_back(pivot, BoundType::STRICT, cur.upper(), BoundType::STRICT);
264  }
265  }

268  void compute_roots() {
269  // Check for p(0) == 0
270  eliminate_zero_roots();
271  // Handle zero polynomial
272  if (carl::isZero(mPolynomial)) {
273  return;
274  }
275  // Handle other easy cases
276  while (true) {
277  // Degree of at most 2 -> use p-q-formula
278  if (isolate_roots_trivially()) {
279  return;
280  }
281  // Use bounds to make interval smaller
282  update_root_bounds();
283  // Check whether interval bounds are roots
284  if (!check_interval_bounds()) {
285  break;
286  }
287  }
288 
289  // Now do actual bisection
290  isolate_by_bisection();
291  }
