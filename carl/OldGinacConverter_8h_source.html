67  GiNaC::ex convertToGinac(const typename Poly::PolyType& poly, const std::map<carl::Variable, GiNaC::ex>& vars)
68  {
69  std::lock_guard<std::recursive_mutex> lock( mMutex );
70  GiNaC::ex result = 0;
71  for(auto term = poly.begin(); term != poly.end(); ++term)
72  {
73  GiNaC::ex factor = GiNaC::ex( GiNaC::numeric( carl::convert<typename Poly::PolyType::CoeffType, cln::cl_RA>(term->coeff()) ) );
74  if((*term).monomial())
75  {
76  for (auto it: *(term->monomial())) {
77  auto carlToGinacVar = vars.find(it.first);
78  assert(carlToGinacVar != vars.end());
79  factor *= GiNaC::pow(carlToGinacVar->second, static_cast<unsigned long>(it.second));
80  }
81  }
82  result += factor;
83  }
84  return result;
85  }

87  template<typename P = Poly, EnableIf<is_factorized<P>> = dummy>
88  GiNaC::ex convertToGinac(const P& poly, const std::map<carl::Variable, GiNaC::ex>& vars)
89  {
90  std::lock_guard<std::recursive_mutex> lock( mMutex );
91  if( existsFactorization( poly ) )
92  {
93  if( poly.factorizedTrivially() )
94  return convertToGinac( poly.polynomial(), vars );
95  GiNaC::ex result = GiNaC::ex( GiNaC::numeric( carl::rationalize<cln::cl_RA>(PreventConversion<typename Poly::PolyType::CoeffType>( poly.coefficient() ))));
96  for( const auto& factor : poly.factorization() )
97  {
98  result *= GiNaC::pow(convertToGinac( factor.first, vars ), factor.second );
99  }
100  return result;
101  }
102  return GiNaC::ex( GiNaC::numeric( carl::rationalize<cln::cl_RA>(PreventConversion<typename Poly::PolyType::CoeffType>( poly.coefficient() ))));
103  }
