352  bool divisible(const Monomial::Arg& m) const
353  {
354  if(!m) return true;
355  assert(isConsistent());
356  if(m->mTotalDegree > mTotalDegree) return false;
357  if(m->nrVariables() > nrVariables()) return false;
358  // Linear, as we expect small monomials.
359  auto itright = m->mExponents.begin();
360  for (const auto& itleft: mExponents) {
361  // Done with division
362  if(itright == m->mExponents.end())
363  {
364  return true;
365  }
366  // Variable is present in both monomials.
367  if(itleft.first == itright->first)
368  {
369  if(itright->second > itleft.second)
370  {
371  // Underflow, itright->exp was larger than itleft->exp.
372  return false;
373  }
374  itright++;
375  }
376  // Variable is not present in lhs, division fails.
377  else if(itleft.first > itright->first)
378  {
379  return false;
380  }
381  else
382  {
383  assert(itright->first > itleft.first);
384  }
385  }
386  // If there remain variables in the m, it fails.
387  return itright == m->mExponents.end();
388  }

587  static std::hash<carl::Variable> h;
588  size_t result = 0;
589  for (const auto& it: c) {
590  // perform a circular shift by 5 bits.
