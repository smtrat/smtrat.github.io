295 template<typename Number>
296 void RealAlgebraicNumberIR<Number>::normalizeInterval() {
297  if (this->interval.lower() == 0 && this->interval.upper() == 0) return; // already normalized
298  if (this->isNumeric()) return;
299  assert(this->polynomial.countRealRoots(this->interval) == 1); // the interval should be isolating for this number
300  // shift the right border below 0 or set the zero interval
301  if (this->interval.contains(0) && this->polynomial.sgn(0) == Sign::ZERO) {
302  this->interval.set(0,0);
303  this->interval.setLowerBoundType(BoundType::WEAK);
304  this->interval.setUpperBoundType(BoundType::WEAK);
305  this->mValue = 0;
306  this->mIsNumeric = true;
307  } else if (this->interval.meets(0)) {
308  // one of the bounds might be 0
309  // the separation is computed following [p.329,Algorithmic Algebra ISBN 3-540-94090-1]
310  Number sep = (Number)1 / (1 + this->polynomial.maximumNorm());
311  assert(this->polynomial.sgn(sep) != Sign::ZERO);
312  if ((this->interval.upper() == 0) ||
313  (this->interval.lower() < -sep && this->polynomial.countRealRoots(Interval<Number>(this->interval.lower(), BoundType::STRICT, -sep, BoundType::STRICT)) > 0)) {
314  this->interval.setUpper(-sep);
315  } else {
316  this->interval.setLower(sep);
317  }
318  }
319  assert( ( !this->interval.meets(0) && this->polynomial.countRealRoots(this->interval) == 1 ) || (this->interval.lower() == 0 && this->interval.upper() == 0)); // otherwise, the interval is not suitable for this real algebraic number
320 }

323 void RealAlgebraicNumberIR<Number>::coarsen(const Interval<Number>& i) const {
324  if (i.lower() < this->interval.lower()) { // coarsen the left bound

340 template<typename Number>
341 void RealAlgebraicNumberIR<Number>::refine(RealAlgebraicNumberSettings::RefinementStrategy strategy) {
342  if (this->isNumeric()) {
343  // This RANIR already has a numeric value
344  assert(this->lower() != this->value());
345  assert(this->upper() != this->value());
346  this->setLower(Interval<Number>(this->lower(), BoundType::STRICT, this->value(), BoundType::STRICT).sample());
347  this->setUpper(Interval<Number>(this->value(), BoundType::STRICT, this->upper(), BoundType::STRICT).sample());
348  return;
349  }
350 
351  Number m;
352  switch (strategy) {
353  case RealAlgebraicNumberSettings::RefinementStrategy::GENERIC:
354  m = this->getInterval().center();
355  break;
356  case RealAlgebraicNumberSettings::RefinementStrategy::BINARYSAMPLE:
357  m = this->getInterval().sample();
358  break;
359  case RealAlgebraicNumberSettings::RefinementStrategy::BINARYNEWTON:
360  m = this->getInterval().sample();
361  break;
362  }
363  assert(this->interval.contains(m));
364  if (this->getPolynomial().isRoot(m)) {
365  this->setLower(Interval<Number>(this->lower(), BoundType::STRICT, m, BoundType::STRICT).sample());
366  this->setUpper(Interval<Number>(m, BoundType::STRICT, this->upper(), BoundType::STRICT).sample());
367  this->mValue = m;
368  this->mIsNumeric = true;
369  } else {
370  if (this->getPolynomial().countRealRoots(Interval<Number>(this->lower(), BoundType::STRICT, m, BoundType::STRICT)) > 0) {
371  this->setUpper(m);
372  } else {
373  this->setLower(m);
374  }
375  }
376 
377  this->refinementCount++;
378  assert(this->lower() < this->upper());

469  // this number is defined as a root of the given polynomial.

471  }
