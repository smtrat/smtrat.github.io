48  struct PolynomialPairIsLess {
49  std::less<UPolynomial> less;
50  unsigned int length(const PolynomialPair& p) {
51  if (p.first == nullptr && p.second == nullptr) return 0;
52  if (p.first == nullptr || p.second == nullptr) return 1;
53  return 2;
54  }
55  bool operator()(const PolynomialPair& a, const PolynomialPair& b) {
56  unsigned int l1 = this->length(a), l2 = this->length(b);
57  if (l1 < l2) return true;
58  if (l2 < l1) return false;
59  if (less(a.first, b.first)) return true;
60  if (a.first == b.first) return less(a.second, b.second);
61  return false;
62  }
