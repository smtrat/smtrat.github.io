46  template<typename Poly>
47  void SqrtEx<Poly>::normalize()
48  {
49 // std::cout << *this << std::endl;
50  Poly gcdA;
51  if( mFactor.isZero() )
52  {
53  gcdA = mConstantPart;
54  }
55  else
56  {
57  Poly sqrtOfRadicand;
58  if( mRadicand.sqrt( sqrtOfRadicand ) )
59  {
60  mConstantPart += mFactor * sqrtOfRadicand;
61  mFactor = 0;
62  mRadicand = 0;
63  }
64  else
65  {
66  assert( !radicand().isZero() );
67  Rational absOfLCoeff = abs( radicand().coprimeFactor() );
68  Rational sqrtResult = 0;
69  if( carl::sqrt_exact( absOfLCoeff, sqrtResult ) )
70  {
71  mFactor *= constant_one<Rational>::get()/sqrtResult;
72  mRadicand *= absOfLCoeff;
73  }
74  }
75  if( mFactor.isZero() )
76  {
77  gcdA = mConstantPart;
78  }
79  else
80  {
81  if( mConstantPart.isZero() )
82  {
83  gcdA = mFactor;
84  }
85  else
86  {
87  Rational ccConstantPart = mConstantPart.coprimeFactor();
88  Poly cpConstantPart = mConstantPart * ccConstantPart;
89  Rational ccFactor = mFactor.coprimeFactor();
90  Poly cpFactor = mFactor * ccFactor;
91  gcdA = carl::gcd( cpConstantPart, cpFactor )*carl::gcd(ccConstantPart,ccFactor);
92  }
93  }
94  }
95  if( gcdA.isZero() ) return;
96  Rational ccGcdA = gcdA.coprimeFactor();
97  Poly cpGcdA = gcdA * ccGcdA;
98  Rational ccDenominator = mDenominator.coprimeFactor();
99  Poly cpDenominator = mDenominator * ccDenominator;
100  gcdA = carl::gcd( cpGcdA, cpDenominator )*carl::gcd(ccGcdA,ccDenominator);
101  // Make sure that the polynomial to divide by cannot be negative, otherwise the sign of the square root expression could change.
102  if( !(gcdA == constant_one<Poly>::get()) && gcdA.definiteness() == carl::Definiteness::POSITIVE_SEMI )
103  {
104  if( !mConstantPart.isZero() )
105  {
106  mConstantPart.divideBy( gcdA, mConstantPart );
107  }
108  if( !mFactor.isZero() )
109  {
110  mFactor.divideBy( gcdA, mFactor );
111  }
112  mDenominator.divideBy( gcdA, mDenominator );
113  }
114  Rational numGcd = constant_zero<Rational>::get();
115  Rational denomLcm = constant_one<Rational>::get();
116  if( factor().isZero() )
117  {
118  if( !constantPart().isZero() )
119  {
120  Rational cpOfConstantPart = constantPart().coprimeFactor();
121  numGcd = carl::getNum( cpOfConstantPart );
122  denomLcm = carl::getDenom( cpOfConstantPart );
123  }
124  else
125  {
126 // cout << " to " << *this << endl;
127  return; // Sqrt expression corresponds to 0.
128  }
129  }
130  else
131  {
132  Rational cpOfFactorPart = factor().coprimeFactor();
133  if( constantPart().isZero() )
134  {
135  numGcd = carl::getNum( cpOfFactorPart );
136  denomLcm = carl::getDenom( cpOfFactorPart );
137  }
138  else
139  {
140  Rational cpOfConstantPart = constantPart().coprimeFactor();
141  numGcd = carl::gcd( carl::getNum( cpOfConstantPart ), carl::getNum( cpOfFactorPart ) );
142  denomLcm = carl::lcm( carl::getDenom( cpOfConstantPart ), carl::getDenom( cpOfFactorPart ) );
143  }
144  }
145  assert( numGcd != constant_zero<Rational>::get() );
146  Rational cpFactor = numGcd/denomLcm;
147  mConstantPart *= cpFactor;
148  mFactor *= cpFactor;
149  Rational cpOfDenominator = denominator().coprimeFactor();
150  mDenominator *= cpOfDenominator;
151  Rational sqrtExFactor = (denomLcm*carl::getNum( cpOfDenominator ))/(numGcd*carl::getDenom( cpOfDenominator ));
152  mConstantPart *= carl::getNum( sqrtExFactor );
153  mFactor *= carl::getNum( sqrtExFactor );
154  mDenominator *= carl::getDenom( sqrtExFactor );

156  //TODO: implement this method further

294  template<typename Poly>
295  bool SqrtEx<Poly>::evaluate(Rational& _result, const std::map<Variable, Rational>& _evalMap, int _rounding) const
296  {
297  Poly radicandEvaluated = radicand().substitute( _evalMap );
298  assert( radicandEvaluated.isConstant() );
299  Rational radicandValue = radicandEvaluated.constantPart();
300  assert( radicandValue >= 0 );
301  Poly factorEvaluated = factor().substitute( _evalMap );
302  assert( factorEvaluated.isConstant() );
303  Rational factorValue = factorEvaluated.constantPart();
304  Poly constantPartEvaluated = constantPart().substitute( _evalMap );
305  assert( constantPartEvaluated.isConstant() );
306  Rational constantPartValue = constantPartEvaluated.constantPart();
307  Poly denomEvaluated = denominator().substitute( _evalMap );
308  assert( denomEvaluated.isConstant() );
309  Rational denomValue = denomEvaluated.constantPart();
310  assert( !carl::isZero( denomValue ) );
311  // Check whether the resulting assignment is integer.
312  bool rounded = true;
313  Rational sqrtExValue;
314  if( !carl::sqrt_exact( radicandValue, sqrtExValue ) )
315  {
316  assert( _rounding != 0 );
317  rounded = false;
318  assert( factorValue != 0 );
319  double dbSqrt = sqrt( carl::toDouble( radicandValue ) );
320  sqrtExValue = carl::rationalize<Rational>( dbSqrt ) ;
321  // As there is no rational number representing the resulting square root we have to round.
322  if( _rounding < 0 ) // If the result should round down in this case.
323  {
324  if( factorValue > 0 && (sqrtExValue*sqrtExValue) > radicandValue )
325  {
326  // The factor of the resulting square root is positive, hence force rounding down.
327  dbSqrt = std::nextafter( dbSqrt, -INFINITY );
328  sqrtExValue = carl::rationalize<Rational>( dbSqrt );
329  assert( !((sqrtExValue*sqrtExValue) > radicandValue) );
330  }
331  else if( factorValue < 0 && (sqrtExValue*sqrtExValue) < radicandValue )
332  {
333  // The factor of the resulting square root is negative, hence force rounding up.
334  dbSqrt = std::nextafter( dbSqrt, INFINITY );
335  sqrtExValue = carl::rationalize<Rational>( dbSqrt );
336  assert( !((sqrtExValue*sqrtExValue) < radicandValue) );
337  }
338  }
339  else if( _rounding > 0 ) // If the result should round up in this case.
340  {
341  if( factorValue < 0 && (sqrtExValue*sqrtExValue) > radicandValue )
342  {
343  // The factor of the resulting square root is negative, hence force rounding down.
344  dbSqrt = std::nextafter( dbSqrt, -INFINITY );
345  sqrtExValue = carl::rationalize<Rational>( dbSqrt );
346  assert( !((sqrtExValue*sqrtExValue) > radicandValue) );
347  }
348  else if( factorValue > 0 && (sqrtExValue*sqrtExValue) < radicandValue )
349  {
350  // The factor of the resulting square root is positive, hence force rounding up.
351  dbSqrt = std::nextafter( dbSqrt, INFINITY );
352  sqrtExValue = carl::rationalize<Rational>( dbSqrt );
353  assert( !((sqrtExValue*sqrtExValue) < radicandValue) );
354  }
355  }
356  }
357  _result = (constantPartValue + factorValue * sqrtExValue) / denomValue;
358  return rounded;
359  }

380  template<typename Poly>
381  SqrtEx<Poly> SqrtEx<Poly>::subBySqrtEx( const Poly& _substituteIn, const carl::Variable& _varToSubstitute, const SqrtEx<Poly>& _substituteBy )
382  {
383  if( !_substituteIn.has( _varToSubstitute ) )
384  return SqrtEx<Poly>( _substituteIn );
385  /*
386  * We have to calculate the result of the substitution:
387  *
388  * q+r*sqrt{t}
389  * (a_n*x^n+...+a_0)[------------ / x]
390  * s
391  * being:
392  *
393  * sum_{k=0}^n (a_k * (q+r*sqrt{t})^k * s^{n-k})
394  * ----------------------------------------------
395  * s^n
396  */
397  auto varInfo = _substituteIn.getVarInfo<true>( _varToSubstitute );
398  const auto& coeffs = varInfo.coeffs();
399  // Calculate the s^k: (0<=k<=n)
400  auto coeff = coeffs.begin();
401  carl::uint lastDegree = varInfo.maxDegree();
402  std::vector<Poly> sk;
403  sk.push_back(constant_one<Poly>::get());
404  for( carl::uint i = 1; i <= lastDegree; ++i )
405  {
406  // s^i = s^l * s^{i-l}
407  sk.push_back( sk.back() * _substituteBy.denominator() );
408  }
409  // Calculate the constant part and factor of the square root of (q+r*sqrt{t})^k
410  std::vector<Poly> qk;
411  qk.push_back( _substituteBy.constantPart() );
412  std::vector<Poly> rk;
413  rk.push_back( _substituteBy.factor() );
414  // Let (q+r*sqrt{t})^l be (q'+r'*sqrt{t})
415  // then (q+r*sqrt{t})^l+1 = (q'+r'*sqrt{t}) * (q+r*sqrt{t}) = ( q'*q+r'*r't + (q'*r+r'*q) * sqrt{t} )
416  for( carl::uint i = 1; i < lastDegree; ++i )
417  {
418  // q'*q+r'*r't
419  qk.push_back( qk.back() * _substituteBy.constantPart() + rk.back() * _substituteBy.factor() * _substituteBy.radicand() );
420  // q'*r+r'*q
421  rk.push_back( rk.back() * _substituteBy.constantPart() + qk.at( i - 1 ) * _substituteBy.factor() );
422  }
423  // Calculate the result:
424  Poly resFactor = constant_zero<Poly>::get();
425  Poly resConstantPart = constant_zero<Poly>::get();
426  if( coeff->first == 0 )
427  {
428  resConstantPart += sk.back() * coeff->second;
429  ++coeff;
430  }
431  for( ; coeff != coeffs.end(); ++coeff )
432  {
433  resConstantPart += coeff->second * qk.at( coeff->first - 1 ) * sk.at( lastDegree - coeff->first );
434  resFactor += coeff->second * rk.at( coeff->first - 1 ) * sk.at( lastDegree - coeff->first );
435  }
436  return SqrtEx( resConstantPart, resFactor, sk.back(), _substituteBy.radicand() );
437  }
