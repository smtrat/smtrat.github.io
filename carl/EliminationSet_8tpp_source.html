186 template<typename Coefficient>
187 size_t EliminationSet<Coefficient>::erase(const UPolynomial* p) {
188  if (p == nullptr) return 0;
189 
190  // remove the child for each parent from the children mapping
191  for (auto i: this->parentsPerChild[p]) {
192  if (i.first != nullptr) this->childrenPerParent[i.first].erase( p );
193  if (i.second != nullptr) this->childrenPerParent[i.second].erase( p );
194  }
195  // remove the child from the parents mapping
196  this->parentsPerChild.erase(p);
197  // remove from lifting and elimination queues
198  auto queuePosition = std::lower_bound(this->mLiftingQueue.begin(), this->mLiftingQueue.end(), p, this->liftingOrder);
199 
200  if (queuePosition != this->mLiftingQueue.end() && *queuePosition == p ) {
201  this->mLiftingQueue.erase(queuePosition);
202  }
203  queuePosition = std::lower_bound(mLiftingQueueReset.begin(), mLiftingQueueReset.end(), p, this->liftingOrder);
204  if( queuePosition != mLiftingQueueReset.end() && *queuePosition == p )
205  mLiftingQueueReset.erase( queuePosition );
206  queuePosition = std::lower_bound(mSingleEliminationQueue.begin(), mSingleEliminationQueue.end(), p, this->eliminationOrder);
207  if( queuePosition != mSingleEliminationQueue.end() && *queuePosition == p )
208  mSingleEliminationQueue.erase(queuePosition);
209  queuePosition = std::lower_bound(mPairedEliminationQueue.begin(), mPairedEliminationQueue.end(), p, this->eliminationOrder);
210  if( queuePosition != mPairedEliminationQueue.end() && *queuePosition == p )
211  mPairedEliminationQueue.erase(queuePosition);
212  // remove from main structure
213  return this->polynomials.erase(p);

216 template<typename Coefficient>
217 std::forward_list<const typename EliminationSet<Coefficient>::UPolynomial*> EliminationSet<Coefficient>::removeByParent(const UPolynomial* parent) {
218  std::forward_list<const UPolynomial*> deleted;
219  if (parent == nullptr) // zero parent is reserved for "no parent"
220  return deleted;
221  // remove all lone children of parent
222  auto position = this->childrenPerParent.find(parent);
223  if (position == this->childrenPerParent.end())
224  return deleted; // parent not found
225  for (const UPolynomial* child: position->second) {
226  // for all children of parent check their parents
227  auto parents = this->parentsPerChild.find(child);
228  if (parents == this->parentsPerChild.end() || parents->second.empty())
229  continue; // nothing to be done for this child
230  typename parentbucket::const_iterator p = std::find_if( parents->second.begin(), parents->second.end(), PolynomialPairContains(parent));
231  while (p != parents->second.end()) {
232  // search matching parents
233  parents->second.erase(p); // remove either single matching parent or parents which got divorced by the removed parent
234  p = std::find_if( parents->second.begin(), parents->second.end(), PolynomialPairContains(parent));
235  }
236 
237  if (parents->second.empty()) {
238  // no parent was left for the child, so delete it
239  this->parentsPerChild.erase( parents );
240  auto queuePosition = std::lower_bound( mLiftingQueue.begin(), mLiftingQueue.end(), child, this->liftingOrder );
241  if( queuePosition != mLiftingQueue.end() )
242  mLiftingQueue.erase( queuePosition );
243  queuePosition = std::lower_bound( mLiftingQueueReset.begin(), mLiftingQueueReset.end(), child, this->liftingOrder );
244  if( queuePosition != mLiftingQueueReset.end() )
245  mLiftingQueueReset.erase( queuePosition );
246  queuePosition = std::lower_bound( mSingleEliminationQueue.begin(), mSingleEliminationQueue.end(), child, this->eliminationOrder );
247  if( queuePosition != mSingleEliminationQueue.end() )
248  mSingleEliminationQueue.erase( queuePosition );
249  queuePosition = std::lower_bound( mPairedEliminationQueue.begin(), mPairedEliminationQueue.end(), child, this->eliminationOrder );
250  if( queuePosition != mPairedEliminationQueue.end() )
251  mPairedEliminationQueue.erase( queuePosition );
252  deleted.push_front(child);
253  this->polynomials.erase(child);
254  }
255  }
256  // remove the information of parent itself
257  this->childrenPerParent.erase(position);
258  return deleted;
259 }
