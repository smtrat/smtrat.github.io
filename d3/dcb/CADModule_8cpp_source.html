64  /**
65  * This method just adds the respective constraint of the subformula, which ought to be one real constraint,
66  * to the local list of constraints. Moreover, the list of all variables is updated accordingly.
67  *
68  * Note that the CAD object is not touched here, the respective calls to CAD::addPolynomial and CAD::check happen in isConsistent.
69  * @param _subformula
70  * @return returns false if the current list of constraints was already found to be unsatisfiable (in this case, nothing is done), returns true previous result if the constraint was already checked for consistency before, otherwise true
71  */
72  template<typename Settings>
73  bool CADModule<Settings>::addCore(ModuleInput::const_iterator _subformula)
74  {
75  SMTRAT_LOG_FUNC("smtrat.cad", _subformula->formula());
76  switch (_subformula->formula().getType()) {
77  case carl::FormulaType::TRUE:
78  return true;
79  case carl::FormulaType::FALSE: {
80  this->hasFalse = true;
81  FormulaSetT infSubSet;
82  infSubSet.insert(_subformula->formula());
83  mInfeasibleSubsets.push_back(infSubSet);
84  return false;
85  }
86  case carl::FormulaType::CONSTRAINT: {
87  SMTRAT_LOG_FUNC("smtrat.cad", _subformula->formula().constraint());
88  if (this->hasFalse) {
89  mSubformulaQueue.push_back(_subformula->formula());
90  return false;
91  } else {
92  return this->addConstraintFormula(_subformula->formula());
93  }
94  }
95  default:
96  SMTRAT_LOG_ERROR("smtrat.cad", "Asserted " << _subformula->formula());
97  assert(false);
98  return true;
99  }
100  }

102  /**
103  * All constraints asserted (and not removed) so far are now added to the CAD object and checked for consistency.
104  * If the result is false, a minimal infeasible subset of the original constraint set is computed.
105  * Otherwise a sample value is available.
106  * @return SAT if consistent, UNSAT otherwise
107  */
108  template<typename Settings>
109  Answer CADModule<Settings>::checkCore()
110  {
111  SMTRAT_LOG_FUNC("smtrat.cad", mFullCheck);
112  if (!mFullCheck) {
113  SMTRAT_LOG_WARN("smtrat.cad", "UNKNOWN due to !mFullCheck");
114  return UNKNOWN;
115  }
116 
117  assert(mConstraints.size() == mConstraintsMap.size());
118 #ifdef SMTRAT_DEVOPTION_Statistics
119  mStats.addCall();
120 #endif
121  if (this->hasFalse) return UNSAT;
122  else {
123  for (const auto& f: mSubformulaQueue) {
124  addConstraintFormula(f);
125  }
126  mSubformulaQueue.clear();
127  }
128  if (!rReceivedFormula().isRealConstraintConjunction() && !rReceivedFormula().isIntegerConstraintConjunction()) {
129  SMTRAT_LOG_WARN("smtrat.cad", "UNKNOWN due to invalid constraints");
130  return UNKNOWN;
131  }
132  if (!mInfeasibleSubsets.empty())
133  return UNSAT; // there was no constraint removed which was in a previously generated infeasible subset
134  // check the extended constraints for satisfiability
135  mCAD.prepareElimination();
136 
137  if (variableBounds().isConflicting()) {
138  mInfeasibleSubsets.push_back(variableBounds().getConflict());
139  mRealAlgebraicSolution = carl::RealAlgebraicPoint<smtrat::Rational>();
140  SMTRAT_LOG_DEBUG("smtrat.cad", "Returning UNSAT due to bound conflict.");
141  return UNSAT;
142  }
143  carl::CAD<smtrat::Rational>::BoundMap boundMap;
144  std::map<carl::Variable, carl::Interval<smtrat::Rational>> eiMap = mVariableBounds.getEvalIntervalMap();
145  std::vector<carl::Variable> variables = mCAD.getVariables();
146  for (unsigned v = 0; v < variables.size(); ++v)
147  {
148  auto vPos = eiMap.find(variables[v]);
149  if (vPos != eiMap.end())
150  boundMap[v] = vPos->second;
151  }
152  carl::cad::Answer status = mCAD.check(mConstraints, mRealAlgebraicSolution, mConflictGraph, boundMap, false, true);
153  //std::cout << mCAD.getVariables() << " = " << mRealAlgebraicSolution << std::endl;
154  if (anAnswerFound()) return ABORTED;
155  if (status == carl::cad::Answer::False) {
156  SMTRAT_LOG_DEBUG("smtrat.cad", "Conflict Graph: " << mConflictGraph);
157  cad::MISGeneration<Settings::mis_heuristic> mis;
158  mis(*this, mInfeasibleSubsets);
159  //std::cout << "Infeasible Subset: " << *mInfeasibleSubsets.begin() << std::endl;
160  //std::cout << "From " << constraints() << std::endl;
161 
162  if (Settings::checkMISForMinimality) {
163  Module::checkInfSubsetForMinimality(mInfeasibleSubsets.begin());
164  }
165  mRealAlgebraicSolution = carl::RealAlgebraicPoint<smtrat::Rational>();
166  SMTRAT_LOG_DEBUG("smtrat.cad", "Returning UNSAT due to theory conflict.");
167  SMTRAT_LOG_DEBUG("smtrat.cad", "Subset: " << mInfeasibleSubsets.back());
168  return UNSAT;
169  }
170  if (status == carl::cad::Answer::Unknown) {
171  // Pass on branch from CAD.
172  const std::vector<carl::Variable>& vars = mCAD.getVariables();
173  std::size_t d = vars.size() - mRealAlgebraicSolution.dim();
174  assert(vars[d].type() == carl::VariableType::VT_INT);
175  auto r = mRealAlgebraicSolution[0].branchingPoint();
176  assert(!carl::isInteger(r));
177  SMTRAT_LOG_DEBUG("smtrat.cad", "Variables: " << vars);
178  SMTRAT_LOG_DEBUG("smtrat.cad", "Branching at " << vars[d] << " = " << r);
179  if (mFinalCheck) branchAt(vars[d], r);
180  return UNKNOWN;
181  }
182  SMTRAT_LOG_TRACE("smtrat.cad", "#Samples: " << mCAD.samples().size());
183  SMTRAT_LOG_TRACE("smtrat.cad", "Elimination sets:");
184  for (unsigned i = 0; i != mCAD.getEliminationSets().size(); ++i) {
185  SMTRAT_LOG_TRACE("smtrat.cad", "\tLevel " << i << " (" << mCAD.getEliminationSet(i).size() << "): " << mCAD.getEliminationSet(i));
186  }
187  SMTRAT_LOG_DEBUG("smtrat.cad", "Result: true");
188  SMTRAT_LOG_DEBUG("smtrat.cad", "CAD complete: " << mCAD.isComplete());
189  SMTRAT_LOG_DEBUG("smtrat.cad", "Solution point: " << mRealAlgebraicSolution);
190  SMTRAT_LOG_DEBUG("smtrat.cad", "Variables: " << mCAD.getVariables());
191  mInfeasibleSubsets.clear();
192  if (Settings::integerHandling == carl::cad::IntegerHandling::SPLIT_ASSIGNMENT) {
193 #ifdef SMTRAT_DEVOPTION_Statistics
194  {
195  std::size_t branches = 0;
196  for (std::size_t d = 0; d < mRealAlgebraicSolution.dim(); d++) {
197  if (mCAD.getVariables()[d].type() != carl::VariableType::VT_INT) continue;
198  if (carl::isInteger(mRealAlgebraicSolution[d])) continue;
199  branches++;
200  }
201  mStats.addBBStats(branches);
202  }
203 #endif
204  cad::SplitVariableSelector<Settings::splitHeuristic> svs;
205  int d = svs.select(mCAD.getVariables(), mRealAlgebraicSolution);
206  if (d != -1) {
207  if (mFinalCheck) branchAt(mCAD.getVariables()[std::size_t(d)], mRealAlgebraicSolution[std::size_t(d)].branchingPoint(), true, true, true);
208  SMTRAT_LOG_DEBUG("smtrat.cad", "Returning unknown with split");
209  return UNKNOWN;
210  }
211  } else if (Settings::integerHandling == carl::cad::IntegerHandling::SPLIT_PATH) {
212  // Check whether the found assignment is integer. Split path to first non-integral assignment
213  const std::vector<carl::Variable>& vars = mCAD.getVariables();
214  FormulasT formulas;
215  FormulasT exclusion;
216  for (std::size_t dim = mRealAlgebraicSolution.dim(); dim > 0; dim--) {
217  std::size_t d = dim - 1;
218  if (!validateIntegrality(vars, d)) {
219  // Assemble lemma
220  if (mFinalCheck) {
221  FormulaT lemma(carl::OR, std::move(formulas));
222  FormulaT enforcer(carl::AND, std::move(exclusion));
223  SMTRAT_LOG_DEBUG("smtrat.cad", "Lifting lemma " << lemma);
224  SMTRAT_LOG_DEBUG("smtrat.cad", "Lifting exclusion " << enforcer);
225  SMTRAT_LOG_DEBUG("smtrat.cad", "For sample tree" << mCAD.getSampleTree());
226  addLemma(FormulaT(carl::OR, {FormulaT(rReceivedFormula()).negated(), std::move(lemma)}));
227  addLemma(enforcer);
228  }
229  SMTRAT_LOG_DEBUG("smtrat.cad", "Returning unknown with lemma");
230  return UNKNOWN;
231  }
232  auto r = this->mRealAlgebraicSolution[d].branchingPoint();
233  Poly p = vars[d] - r;
234  FormulaT c1(ConstraintT(p + carl::constant_one<Rational>::get(), carl::Relation::LEQ));
235  FormulaT c2(ConstraintT(-p + carl::constant_one<Rational>::get(), carl::Relation::LEQ));
236  formulas.push_back(c1);
237  formulas.push_back(c2);
238  exclusion.push_back(FormulaT(carl::OR, {c1.negated(), c2.negated()}));
239  }
240  } else if (Settings::integerHandling == carl::cad::IntegerHandling::NONE) {
241  const std::vector<carl::Variable>& vars = mCAD.getVariables();
242  for (std::size_t d = 0; d < mRealAlgebraicSolution.dim(); d++) {
243  if (!validateIntegrality(vars, d)) {
244  SMTRAT_LOG_DEBUG("smtrat.cad", "Returning unknown without split");
245  return UNKNOWN;
246  }
247  }
248  }
249  assert(checkSatisfiabilityOfAssignment());
250  SMTRAT_LOG_DEBUG("smtrat.cad", "Returning sat");
251  return SAT;
252  }

254  template<typename Settings>
255  void CADModule<Settings>::removeCore(ModuleInput::const_iterator _subformula)
256  {
257  SMTRAT_LOG_FUNC("smtrat.cad", _subformula->formula());
258  switch (_subformula->formula().getType()) {
259  case carl::FormulaType::TRUE:
260  return;
261  case carl::FormulaType::FALSE:
262  this->hasFalse = false;
263  return;
264  case carl::FormulaType::CONSTRAINT: {
265  SMTRAT_LOG_FUNC("smtrat.cad", _subformula->formula().constraint());
266  auto it = std::find(mSubformulaQueue.begin(), mSubformulaQueue.end(), _subformula->formula());
267  if (it != mSubformulaQueue.end()) {
268  mSubformulaQueue.erase(it);
269  return;
270  }
271 
272  mVariableBounds.removeBound(_subformula->formula().constraint(), _subformula->formula());
273 
274  ConstraintIndexMap::iterator constraintIt = mConstraintsMap.find(_subformula->formula());
275  if (constraintIt == mConstraintsMap.end())
276  return; // there is nothing to remove
277  carl::cad::Constraint<smtrat::Rational> constraint = mConstraints[constraintIt->second];
278 
279  SMTRAT_LOG_TRACE("smtrat.cad", "---- Constraint removal (before) ----");
280  SMTRAT_LOG_TRACE("smtrat.cad", "Elimination sets:");
281  for (unsigned i = 0; i != mCAD.getEliminationSets().size(); ++i) {
282  SMTRAT_LOG_TRACE("smtrat.cad", "\tLevel " << i << " (" << mCAD.getEliminationSet(i).size() << "): " << mCAD.getEliminationSet(i));
283  }
284  SMTRAT_LOG_TRACE("smtrat.cad", "#Samples: " << mCAD.samples().size());
285  SMTRAT_LOG_TRACE("smtrat.cad", "-----------------------------------------");
286  SMTRAT_LOG_TRACE("smtrat.cad", "Removing " << constraint << "...");
287 
288  unsigned constraintIndex = constraintIt->second;
289  // remove the constraint in mConstraintsMap
290  mConstraintsMap.erase(constraintIt);
291  // update the constraint / index map, i.e., decrement all indices above the removed one
292  updateConstraintMap(constraintIndex, true);
293  // remove the corresponding constraint node with index constraintIndex
294  mConflictGraph.removeConstraint(constraint);
295 
296  // remove the constraint from the list of constraints
297  assert(mConstraints.size() > constraintIndex); // the constraint to be removed should be stored in the local constraint list
298  mConstraints.erase(mConstraints.begin() + constraintIndex); // erase the (constraintIt->second)-th element
299 
300  // remove the corresponding polynomial from the CAD if it is not occurring in another constraint
301  bool doDelete = true;
302  for (const auto& c: mConstraints) {
303  if (constraint.getPolynomial() == c.getPolynomial()) {
304  SMTRAT_LOG_TRACE("smtrat.cad", "Not removing " << constraint.getPolynomial() << " due to " << c);
305  doDelete = false;
306  break;
307  }
308  }
309  if (doDelete) {
310  // no other constraint claims the polynomial, hence remove it from the list and the cad
311  mCAD.removePolynomial(constraint.getPolynomial());
312  }
313 
314 
315  SMTRAT_LOG_TRACE("smtrat.cad", "---- Constraint removal (afterwards) ----");
316  SMTRAT_LOG_TRACE("smtrat.cad", "New constraint set: " << mConstraints);
317  SMTRAT_LOG_TRACE("smtrat.cad", "Elimination sets:");
318  for (unsigned i = 0; i != mCAD.getEliminationSets().size(); ++i) {
319  SMTRAT_LOG_TRACE("smtrat.cad", "\tLevel " << i << " (" << mCAD.getEliminationSet(i).size() << "): " << mCAD.getEliminationSet(i));
320  }
321  SMTRAT_LOG_TRACE("smtrat.cad", "#Samples: " << mCAD.samples().size());
322  SMTRAT_LOG_TRACE("smtrat.cad", "-----------------------------------------");
323  return;
324  }
325  default:
326  return;
327  }

333  template<typename Settings>
334  void CADModule<Settings>::updateModel() const
335  {
336  SMTRAT_LOG_FUNC("smtrat.cad", "Updating model");
337  clearModel();
338  if (this->solverState() == SAT) {
339  // bound-independent part of the model
340  std::vector<carl::Variable> vars(mCAD.getVariables());
341  for (unsigned varID = 0; varID < vars.size(); ++varID) {
342  if (varID == mRealAlgebraicSolution.dim()) break;
343  ModelValue ass = mRealAlgebraicSolution[varID];
344  mModel.insert(std::make_pair(vars[varID], ass));
345  }
346  // bounds for variables which were not handled in the solution point
347  for (auto b: mVariableBounds.getIntervalMap()) {
348  // add an assignment for every bound of a variable not in vars (Caution! Destroys vars!)
349  std::vector<carl::Variable>::iterator v = std::find(vars.begin(), vars.end(), b.first);
350  if (v != vars.end()) {
351  vars.erase(v); // shall never be found again
352  } else {
353  // variable not handled by CAD, use the midpoint of the bounding interval for the assignment
354  ModelValue ass = carl::center(b.second);
355  mModel.insert(std::make_pair(b.first, ass));
356  }
357  }
358  }
359  excludeNotReceivedVariablesFromModel(); // TODO: try to avoid this
360  SMTRAT_LOG_DEBUG("smtrat.cad", "Model of CAD: " << mModel);
361  }

374  template<typename Settings>
375  bool CADModule<Settings>::addConstraintFormula(const FormulaT& f) {
376  assert(f.getType() == carl::FormulaType::CONSTRAINT);
377  mVariableBounds.addBound(f.constraint(), f);
378  // add the constraint to the local list of constraints and memorize the index/constraint assignment if the constraint is not present already
379  if (mConstraintsMap.find(f) != mConstraintsMap.end())
380  return true; // the exact constraint was already considered
381  carl::cad::Constraint<smtrat::Rational> constraint = convertConstraint(f.constraint());
382  mConstraints.push_back(constraint);
383  mConstraintsMap[f] = (unsigned)(mConstraints.size() - 1);
384  mCFMap[constraint] = f;
385 #ifdef __VS
386  mCAD.addPolynomial(Poly::PolyType(constraint.getPolynomial()), constraint.getVariables());
387 #else
388  mCAD.addPolynomial(typename Poly::PolyType(constraint.getPolynomial()), constraint.getVariables());
389 #endif
390 
391  return solverState() != UNSAT;
392  }

394  /**
395  * Converts the constraint types.
396  * @param c constraint of the SMT-RAT
397  * @return constraint of GiNaCRA
398  */
399  template<typename Settings>
400  inline const carl::cad::Constraint<smtrat::Rational> CADModule<Settings>::convertConstraint( const smtrat::ConstraintT& c )
401  {
402  // convert the constraints variable
403  std::vector<carl::Variable> variables;
404  for (auto i: c.variables().underlyingVariables()) {
405  variables.push_back(i);
406  }
407  carl::Sign signForConstraint = carl::Sign::ZERO;
408  bool cadConstraintNegated = false;
409  switch (c.relation()) {
410  case carl::Relation::EQ:
411  break;
412  case carl::Relation::LEQ:
413  signForConstraint = carl::Sign::POSITIVE;
414  cadConstraintNegated = true;
415  break;
416  case carl::Relation::GEQ:
417  signForConstraint = carl::Sign::NEGATIVE;
418  cadConstraintNegated = true;
419  break;
420  case carl::Relation::LESS:
421  signForConstraint = carl::Sign::NEGATIVE;
422  break;
423  case carl::Relation::GREATER:
424  signForConstraint = carl::Sign::POSITIVE;
425  break;
426  case carl::Relation::NEQ:
427  cadConstraintNegated = true;
428  break;
429  default: assert(false);
430  }
431 #ifdef __VS
432  return carl::cad::Constraint<smtrat::Rational>((Poly::PolyType)c.lhs(), signForConstraint, variables, cadConstraintNegated);
433 #else
434  return carl::cad::Constraint<smtrat::Rational>((typename Poly::PolyType)c.lhs(), signForConstraint, variables, cadConstraintNegated);
435 #endif
436  }

438  /**
439  * Converts the constraint types.
440  * @param c constraint of the GiNaCRA
441  * @return constraint of SMT-RAT
442  */
443  template<typename Settings>
444  inline ConstraintT CADModule<Settings>::convertConstraint( const carl::cad::Constraint<smtrat::Rational>& c )
445  {
446  carl::Relation relation = carl::Relation::EQ;
447  switch (c.getSign()) {
448  case carl::Sign::POSITIVE:
449  if (c.isNegated()) relation = carl::Relation::LEQ;
450  else relation = carl::Relation::GREATER;
451  break;
452  case carl::Sign::ZERO:
453  if (c.isNegated()) relation = carl::Relation::NEQ;
454  else relation = carl::Relation::EQ;
455  break;
456  case carl::Sign::NEGATIVE:
457  if (c.isNegated()) relation = carl::Relation::GEQ;
458  else relation = carl::Relation::LESS;
459  break;
460  default: assert(false);
461  }
462  return ConstraintT(c.getPolynomial(), relation);
463  }
