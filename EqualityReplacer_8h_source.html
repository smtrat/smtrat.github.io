103  bool considerDuplicateAssignments() {
104  for (auto it = mDuplicateAssignments.begin(); it != mDuplicateAssignments.end();) {
105  SMTRAT_LOG_DEBUG("smtrat.cad.eqreplacer", "Looking at duplicate " << it->first);
106  if (mAssignments.find(it->second.first) == mAssignments.end()) {

126  int commitConstraint(C& c) {
127  auto res = carl::model::substitute(c.first, mModel);
128  if (c.second) {
129  // Constraint has already been simplified
130  assert(c.second != boost::none);
131  if (res == c.second) return 1;
132  // Old simplication is invalid, remove it.
133  SMTRAT_LOG_DEBUG("smtrat.cad.eqreplacer", "Actually Erasing " << *c.second);
134  mCAD.removeConstraint(*c.second);
135  mForwardedConstraints.erase(*c.second);
136  c.second = boost::none;
137  }
138  if (mForwardedConstraints.find(res) != mForwardedConstraints.end()) {
139  // The (simplified?) constraint has already been forwarded.
140  SMTRAT_LOG_DEBUG("smtrat.cad.eqreplacer", "Not considering " << res);
141  return 2;
142  }
143  SMTRAT_LOG_INFO("smtrat.cad.eqreplacer", c.first << " -> " << res);
144  if (res.isConsistent() == 0) {
145  // The constraint is trivially false under the model.
146  SMTRAT_LOG_DEBUG("smtrat.cad.eqreplacer", "Got immediate conflict");
147  mConflict = c.first;
148  return -1;
149  }
150  if (res.isConsistent() == 2) {
151  // The constraint is forwarded to the CAD.
152  c.second = res;
153  SMTRAT_LOG_DEBUG("smtrat.cad.eqreplacer", "Actually Adding " << *c.second);
154  mCAD.addConstraint(*c.second);
155  mForwardedConstraints.insert(*c.second);
156  }
157  return 1;
158  }

163  bool commit() {
164  SMTRAT_LOG_DEBUG("smtrat.cad.eqreplacer", "Commit ");
165  if (!considerDuplicateAssignments()) return false;
166  SMTRAT_LOG_INFO("smtrat.cad.eqreplacer", "Using " << mModel << " to simplify");
167  std::vector<ConstraintT> toReconsider;
168  for (auto& c: mConstraints) {
169  int res = commitConstraint(c);
170  if (res == -1) return false;
171  if (res == 2) toReconsider.emplace_back(c.first);
172  }
173  for (const auto& c: toReconsider) {
174  int res = commitConstraint(*mConstraints.find(c));
175  if (res == -1) return false;
176  }
177  for (const auto& ass: mAssignments) {
178  // Forward the assignment to the CAD.
179  if (mForwardedConstraints.find(ass.second) != mForwardedConstraints.end()) continue;
180  SMTRAT_LOG_DEBUG("smtrat.cad.eqreplacer", "Adding Assignment " << ass.second);
181  mCAD.addConstraint(ass.second);
182  mForwardedConstraints.insert(ass.second);
183  }
184  return true;
185  }
