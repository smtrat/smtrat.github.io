52  bool insertRootSamples(Iterator parent, std::vector<Sample>& samples) {
53  if (samples.empty()) return false;
54  std::sort(samples.begin(), samples.end());
55  samples.erase(std::unique(samples.begin(), samples.end()), samples.end());
56  bool gotNewSamples = false;
57  auto tbegin = mTree.begin_children(parent);
58  auto tit = tbegin;
59  auto tend = mTree.end_children(parent);
60  // Insert roots
61  for (const auto& s: samples) {
62  while (tit != tend && *tit < s) tit++;
63  if (tit == tend) {
64  // Append as last sample
65  auto it = mTree.append(parent, s);
66  addToQueue(it);
67  gotNewSamples = true;
68  } else if (tit->value() == s.value()) {
69  // Replace non-root sample
70  if (!tit->isRoot()) gotNewSamples = true;
71  tit->merge(s);
72  } else {
73  // Insert before sample
74  auto it = mTree.insert(tit, s);
75  addToQueue(it);
76  gotNewSamples = true;
77  }
78  }
79  return gotNewSamples;
80  }

82  bool mergeRootSamples(Iterator parent, std::vector<Sample>& samples) {
83  SMTRAT_LOG_TRACE("smtrat.cad.lifting", "Merging " << samples << " into" << std::endl << mTree);
84  bool gotNewSamples = insertRootSamples(parent, samples);
85  if (!gotNewSamples) {
86  SMTRAT_LOG_TRACE("smtrat.cad.lifting", "Nothing to merge");
87  if (mTree.is_leaf(parent)) {
88  auto var = mVariables[parent.depth()];
89  auto interval = mConstraints.bounds().getInterval(var);
90  SMTRAT_LOG_TRACE("smtrat.cad.lifting", "Inserting zero node for " << var << " from " << interval);
91  auto it = mTree.append(parent, Sample(RAN(sample(interval))));
92  addToQueue(it);
93  }
94  return false;
95  }
96  // Add samples
97  auto tbegin = mTree.begin_children(parent);
98  auto tend = mTree.end_children(parent);
99  auto tit = tbegin, tlast = tbegin;
100  if (tbegin->isRoot()) {
101  auto it = mTree.insert(tbegin, Sample(carl::sampleBelow(tlast->value())));
102  addToQueue(it);
103  }
104  while (true) {
105  tit++;
106  if (tit == tend) break;
107  if (tlast->isRoot() && tit->isRoot()) {
108  auto it = mTree.insert(tit, Sample(carl::sampleBetween(tlast->value(), tit->value(), Settings::sampleHeuristic)));
109  addToQueue(it);
110  }
111  tlast = tit;
112  }
113  if (tlast->isRoot()) {
114  auto it = mTree.append(parent, Sample(carl::sampleAbove(tlast->value())));
115  addToQueue(it);
116  }
117 
118  SMTRAT_LOG_TRACE("smtrat.cad.lifting", "Result: " << std::endl << mTree);
119  return true;
120  }

177  bool liftSample(Iterator sample, const UPoly& p, std::size_t pid) {
178  assert(isConsistent());
179  auto m = extractSampleMap(sample);
180  SMTRAT_LOG_DEBUG("smtrat.cad.lifting", "Lifting " << m << " on " << p);
181  std::vector<Sample> newSamples;
182  // TODO: Check whether the polynomials becomes zero (check if McCallum is safe)
183  auto roots = carl::rootfinder::realRoots(p, m, RationalInterval::unboundedInterval());
184  if (roots.empty()) roots = { RAN(0) };
185  for (const auto& r: roots) {
186  SMTRAT_LOG_DEBUG("smtrat.cad.lifting", "\tnew root sample: " << r);
187  newSamples.emplace_back(r, pid);
188  }
189  auto bounds = mConstraints.bounds().getInterval(mVariables[sample.depth()]);
190  if (bounds.lowerBoundType() != carl::BoundType::INFTY) {
191  newSamples.emplace_back(RAN(bounds.lower()), true);
192  }
193  if (bounds.upperBoundType() != carl::BoundType::INFTY) {
194  newSamples.emplace_back(RAN(bounds.upper()), true);
195  }
196  SMTRAT_LOG_DEBUG("smtrat.cad.lifting", "\tmerging " << newSamples);
197  return mergeRootSamples(sample, newSamples);
198  }
199  bool addTrivialSample(Iterator sample) {
200  if (!mTree.is_leaf(sample)) return false;
201  SMTRAT_LOG_DEBUG("smtrat.cad.lifting", "Variables: " << mVariables);
202  SMTRAT_LOG_DEBUG("smtrat.cad.lifting", "For " << printSample(sample));
203  auto variable = mVariables[sample.depth()];
204  auto center = carl::sample(mConstraints.bounds().getInterval(variable));
205  SMTRAT_LOG_DEBUG("smtrat.cad.lifting", "Selecting " << center << " for " << variable << " from " << mConstraints.bounds().getInterval(variable));
206  auto it = mTree.append(sample, Sample(RAN(center), false));
207  addToQueue(it);
208 
209  for (std::size_t i = 1; i < Settings::trivialSampleRadius; i++) {
210  auto itpos = mTree.append(sample, Sample(RAN(center + i), false));
211  addToQueue(itpos);
212  auto itneg = mTree.append(sample, Sample(RAN(center - i), false));
213  addToQueue(itneg);
214  }
215  return true;
216  }

218  SMTRAT_LOG_TRACE("smtrat.cad.lifting", "Extracting sample from" << std::endl << mTree);

230  void removedPolynomialsFromLevel(std::size_t level, const carl::Bitset& mask) {
231  SMTRAT_LOG_DEBUG("smtrat.cad.lifting", "Cleanup after removing " << mask << " from level " << level);
232  for (auto it = mTree.begin_depth(level - 1); it != mTree.end_depth(); it++) {
233  it->liftedWith() -= mask;
234  }
235  std::vector<Iterator> deleteQueue;
236  for (auto it = mTree.begin_depth(level); it != mTree.end_depth(); it++) {
237  if (!it->isRoot()) continue;
238  it->rootOf() -= mask;
239  if (it->rootOf().none()) {
240  deleteQueue.emplace_back(it);

242  }
243  }

246  mTree.erase(it);
