124 public:
125  /**
126  * Add a CellDescription to the covering.
127  *
128  * The added cell needs to be the rightmost cell of the already added cells and not be contained in any of these cells (or vice versa).
129  *
130  * * The first cell needs to have -oo as left bound.
131  * * The last cell needs to have oo as right bound.
132  * * All cells need to cover the real line under an appropriate sample.
133  * * Evaluated under an appropriate sample, the left bound of the added cell c is strictly greater than the left bounds of already added cells (considering also "strictness" of the bounds).
134  * * The right bound of the added cell c needs to be greater than all right bounds of already added cells (considering also "strictness" of the bounds).
135  */
136  void add(const CellDescription& c) {
137  assert(!m_data.empty() || (c.is_sector() && !c.lower()));
138  assert(m_data.empty() || c.is_section() || c.lower());
139  assert(m_data.empty() || m_data.back().is_section() || m_data.back().upper());
140  m_data.push_back(c);
141  }
