93 /**
94  * This class parses a whole smtlib file into a hierarchy of Node objects.
95  */
96 class Parser {
97  /// Parses symbols.
98  SymbolParser symbol;
99  qi::rule<Iterator, Node(), Skipper> token;
100  qi::rule<Iterator, Node(), Skipper> tpl;
101  qi::rule<Iterator, Node(), Skipper> function;
102  qi::rule<Iterator, Node(), Skipper> expression;
103  qi::rule<Iterator, std::vector<Node>(), Skipper> expressionlist;
104  qi::rule<Iterator, Skipper> linePrefix;
105  qi::rule<Iterator, Skipper> lineAddr;
106  qi::rule<Iterator, Node(), Skipper> lineContent;
107  qi::rule<Iterator, Node(), Skipper> line;
108  qi::rule<Iterator, std::vector<Node>(), Skipper> main;
109  // Error handler.
110  boost::phoenix::function<ErrorHandler> errorHandler;
111 
112 public:
113  /**
114  * Constructs the parsing rules.
115  */
116  Parser() {
117  token = qi::as_string[qi::lexeme[ *(~qi::char_("<>()") | qi::blank) ]];
118  token.name("token");
119  tpl = token >> "<" >> expressionlist >> ">";
120  tpl.name("template");
121  function = token >> "(" >> expressionlist >> ")";
122  function.name("function");
123  expression = tpl | function | token;
124  expression.name("expression");
125  expressionlist = expression % ", ";
126  linePrefix = "==" >> +qi::digit >> "==" >> *qi::blank;
127  linePrefix.name("line prefix");
128  lineAddr = (qi::lit("at 0x") | qi::lit("by 0x")) >> +qi::alnum;
129  lineAddr.name("line address");
130  lineContent = qi::as_string[qi::no_skip[ *(qi::char_ - qi::eol) ]];
131  lineContent.name("line content");
132  line = linePrefix >> (
133  (lineAddr >> ": " >> expression)
134  | lineContent
135  );
136  line.name("line");
137  main = *line >> qi::eoi;
138  main.name("main");
139  qi::debug(main);
140  qi::debug(line);
141  qi::debug(lineContent);
142  qi::debug(expression);
143  qi::debug(tpl);
144  qi::on_error<qi::fail>(main, errorHandler(qi::_1, qi::_2, qi::_3));
145  qi::on_success(main, std::cout << qi::_val << std::endl);
146  qi::on_success(line, std::cout << qi::_val << std::endl);
147  qi::on_success(expression, std::cout << qi::_val << std::endl);
148  }
