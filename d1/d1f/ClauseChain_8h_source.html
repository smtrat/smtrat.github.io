9 /**
10  * An explanation is either a single clause or a chain of clauses, satisfying the following properties:
11  * - If the clauses are inserted and propagated in the chain's order, at least the last clause should be conflicting.
12  * - Each conflicting clause is already conflicting after preceding clauses have been inserted and propagated.
13  * - Tseitin vars should be used so that a Tseitin variable C is equivalent to its corresponding subformula F (or at least C -> F).
14  */

122  /**
123  * Performs resolution on the chain. Uses the last clause for resolution.
124  * @return A single clause conflicting under the current assignment.
125  */
126  FormulaT resolve() const {
127  FormulasT result;
128  std::unordered_set<FormulaT> toProcess;
129 
130  // initialize with conflicting clause
131  assert(mClauseChain.back().isConflicting());
132  if (mClauseChain.back().clause().isNary()) {
133  for (const auto& lit : mClauseChain.back().clause()) {
134  if (isTseitinVar(lit)) {
135  toProcess.insert(lit);
136  } else {
137  result.push_back(lit);
138  }
139  }
140  } else {
141  if (isTseitinVar(mClauseChain.back().clause())) {
142  toProcess.insert(mClauseChain.back().clause());
143  } else {
144  result.push_back(mClauseChain.back().clause());
145  }
146  }
147 
148 
149  // resolve using propagations
150  for (auto iter = mClauseChain.rbegin()+1; iter != mClauseChain.rend(); iter++) {
151  if (iter->isPropagating()) {
152  // check if any tseitin variable can be resolved using this clause
153  if (toProcess.find(iter->impliedTseitinLiteral().negated()) == toProcess.end()) {
154  continue;
155  }
156 
157  // remove the processed tseitin variable
158  toProcess.erase(iter->impliedTseitinLiteral().negated());
159 
160  // add literals of clauses to respective sets
161  if (iter->clause().isNary()) {
162  for (const auto& lit : iter->clause().subformulas()) {
163  if (lit != iter->impliedTseitinLiteral()) {
164  if (isTseitinVar(lit)) {
165  toProcess.insert(lit);
166  } else {
167  result.push_back(lit);
168  }
169  }
170  }
171  }
172 
173  if (toProcess.empty()) {
174  break;
175  }
176  }
177  }
178  assert(toProcess.empty());
179 
180  return FormulaT(carl::FormulaType::OR, std::move(result));
181  }
