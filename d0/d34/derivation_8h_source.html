268 template<typename Properties>
269 DerivationRef<Properties> make_derivation(Projections& proj, const Assignment& assignment, size_t level) {
270  const auto& vars = proj.polys().var_order();
271 
272  auto zero_base = std::make_shared<BaseDerivation<Properties>>(proj, BaseDerivationRef<Properties>(nullptr), 0);
273  auto zero_delineated = std::make_shared<DelineatedDerivation<Properties>>(zero_base);
274  DerivationRef<Properties> current = std::make_shared<SampledDerivation<Properties>>(zero_delineated, RAN(0));
275 
276  for (size_t i = 1; i <= level; i++) {
277  auto base = std::make_shared<BaseDerivation<Properties>>(proj, current, i);
278  auto delineated = std::make_shared<DelineatedDerivation<Properties>>(base);
279  if (assignment.find(vars[i-1]) != assignment.end()) {
280  current = std::make_shared<SampledDerivation<Properties>>(delineated, assignment.at(vars[i-1]));
281  } else {
282  current = delineated;
283  }
284  }
285 
286  return current;
287 }

297 void merge_underlying(std::vector<std::reference_wrapper<SampledDerivation<Properties>>>& derivations) {
298  std::set<DerivationRef<Properties>> underlying;
